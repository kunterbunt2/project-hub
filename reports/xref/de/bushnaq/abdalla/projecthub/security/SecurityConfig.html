35  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;

47  @EnableWebSecurity
48  @EnableMethodSecurity(securedEnabled = true, prePostEnabled = true)
49  @Configuration
50  public class SecurityConfig extends VaadinWebSecurity {
51      Logger logger = LoggerFactory.getLogger(this.getClass());
52  
53      @Autowired
54      private SecurityUserDetailsService userDetailsService;
55  
56      @Bean
57      public AuthenticationManager authenticationManager() {
58          // Create authentication provider for the test users
59          DaoAuthenticationProvider testProvider = new DaoAuthenticationProvider();
60          testProvider.setPasswordEncoder(passwordEncoder());
61          testProvider.setUserDetailsService(testUsers());
62  
63          // Create authentication provider for the regular users
64          DaoAuthenticationProvider regularProvider = new DaoAuthenticationProvider();
65          regularProvider.setPasswordEncoder(passwordEncoder());
66          regularProvider.setUserDetailsService(userDetailsService);
67  
68          // Return a provider manager with both providers
69          return new ProviderManager(testProvider, regularProvider);
70      }
71  
72      /**
73       * Separate security configuration specifically for API endpoints
74       * This uses HTTP Basic Authentication for API security
75       */
76      @Bean
77      @Order(2) // Lower precedence than the OAuth2 API security filter chain
78      public SecurityFilterChain basicAuthApiSecurityFilterChain(HttpSecurity http) throws Exception {
79          logger.info(">>> Configuring security chain (3/4) basic authentication for REST API endpoints");
80          return http
81                  .securityMatcher("/api/**") // Apply this configuration only to API endpoints
82                  .authorizeHttpRequests(authorize -> authorize
83                          .anyRequest().authenticated()
84                  )
85                  .csrf(AbstractHttpConfigurer::disable) // Disable CSRF for API endpoints
86                  .httpBasic() // Enable HTTP Basic Auth for APIs
87                  .and()
88                  .exceptionHandling(handling -> handling
89                          // Return 401 for unauthenticated requests instead of redirecting
90                          .authenticationEntryPoint((request, response, authException) -> {
91                              response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
92                              response.getWriter().write("Authentication required");
93                          })
94                          // Return 403 for unauthorized requests
95                          .accessDeniedHandler((request, response, accessDeniedException) -> {
96                              response.setStatus(HttpServletResponse.SC_FORBIDDEN);
97                              response.getWriter().write("Access denied");
98                          })
99                  )
100                 .authenticationManager(authenticationManager()) // Use our combined authentication manager
101                 .build();
102     }
103 
104     @Override
105     protected void configure(HttpSecurity http) throws Exception {
106         logger.info(">>> Configuring security chain (4/4) basic authentication for vaadin");
107         // Configure for all non-API endpoints (Vaadin UI)
108 
109         // Allow for H2 console
110         http//
111                 .authorizeHttpRequests(authorize -> authorize//
112                         .requestMatchers(new AntPathRequestMatcher("/h2-console/**")).permitAll()
113                 )//
114                 .csrf(csrf -> csrf//
115                         .ignoringRequestMatchers(new AntPathRequestMatcher("/h2-console/**"))
116                 )//
117                 .headers(headers -> headers//
118                         .frameOptions(frameOptions -> frameOptions.sameOrigin())
119                 );
120 
121 
122         // Set the login view
123         setLoginView(http, LoginView.class);
124 
125         http.authorizeHttpRequests(authorize -> authorize
126                         .requestMatchers(new AntPathRequestMatcher("/")).permitAll()
127                         .requestMatchers(new AntPathRequestMatcher("/" + LoginView.ROUTE)).permitAll()
128                         .requestMatchers(new AntPathRequestMatcher("/VAADIN/**")).permitAll()
129                         .requestMatchers(new AntPathRequestMatcher("/ui/icons/**")).permitAll()
130                         .requestMatchers(new AntPathRequestMatcher("/ui/images/**")).permitAll()
131                         .requestMatchers(new AntPathRequestMatcher("/frontend/**")).permitAll()
132                         .requestMatchers(new AntPathRequestMatcher("/frontend-es5/**")).permitAll()
133                         .requestMatchers(new AntPathRequestMatcher("/frontend-es6/**")).permitAll()
134 //                .requestMatchers(new AntPathRequestMatcher("/oauth2/**")).permitAll()
135 //                .requestMatchers(new AntPathRequestMatcher("/login/oauth2/**")).permitAll()
136         );
137 
138         // Call the parent configuration to handle Vaadin-specific security
139         super.configure(http);
140 
141         // Set the default success URL after login to ProductListView
142         http.formLogin(formLogin -> formLogin.defaultSuccessUrl("/ui/product-list", true));
143     }
144 
145     @Bean
146     public PasswordEncoder passwordEncoder() {
147         return new BCryptPasswordEncoder();
148     }
149 
150     /**
151      * Configure test users for API testing with appropriate roles
152      */
153     @Bean
154     public InMemoryUserDetailsManager testUsers() {
155         // Create an admin user for admin role tests
156         UserDetails adminUser = User.builder()
157                 .username("admin-user")
158                 .password(passwordEncoder().encode("test-password"))
159                 .roles("ADMIN")
160                 .build();
161 
162         // Create a regular user for USER role tests
163         UserDetails regularUser = User.builder()
164                 .username("user")
165                 .password(passwordEncoder().encode("test-password"))
166                 .roles("USER")  // Only USER role, no ADMIN privileges
167                 .build();
168 
169         logger.info("Created default test user/admin users.");
170 
171         return new InMemoryUserDetailsManager(adminUser, regularUser);
172     }
173 
174     /**
175      * Configure a combined UserDetailsService that checks both test users and regular users
176      */
177     @Bean
178     @Primary
179     public UserDetailsService userDetailsService() {
180         return username -> {
181             try {
182                 // First try the test users
183                 return testUsers().loadUserByUsername(username);
184             } catch (UsernameNotFoundException e) {
185                 // If not found among test users, try the regular user service
186                 return userDetailsService.loadUserByUsername(username);
187             }
188         };
189     }
190 }
