81          super(dao);
82          this.request = dao.sprint;
83          processingInit(dao);
84          // populateGuide(context, start, end, request);
85          initGanttGuide(context, dao.start, dao.end);
86          maxActualWorked = dao.maxWorked;
87          if (isPlannedBurnDownGuideAvailable()) {
88              // was the estimation in the gantt chart actually bigger than the current estimation?

93              request.exceptions.add(new WarnException(String.format("%s %d times.", WORK_OUTSIDE_ALLOWED_TIME_BOUNDARIES_OCCURRED, numberOfWorkExceptions)));

97      public void calculateAuthorContribution(List<Worklog> worklogList, List<WorklogRemaining> worklogRemaining,
98                                              AuthorsContribution authorsContribution) {
99          if (worklogRemaining != null) {
100             for (WorklogRemaining work : worklogRemaining) {
101                 AuthorContribution ac = authorsContribution.get(work.getAuthor());
102                 if (ac == null) {
103                     ac = new AuthorContribution();
104                     authorsContribution.put(work.getAuthor(), ac);
105                 }
106                 ac.addRemaining(work.getRemaining());
107             }
108         }
109         if (worklogList != null) {
110             for (Worklog work : worklogList) {
111                 AuthorContribution ac = authorsContribution.get(request.getuser(work.getAuthorId()));
112                 if (ac == null) {
113                     ac = new AuthorContribution();
114                     authorsContribution.put(request.getuser(work.getAuthorId()), ac);
115                 }
116                 ac.addWorked(work.getTimeSpent());
117             }
118         }

121     protected int calculateAuthorGraphHeight(Duration authorDelta, Duration authorEstimated, Duration sumEstimated) {
122         long maxAuthorGraphHeight = (diagram.height * authorEstimated.getSeconds()) / sumEstimated.getSeconds();
123         return (int) ((authorEstimated.minus(authorDelta).getSeconds() * maxAuthorGraphHeight) / (authorEstimated.getSeconds()));
124     }

132     @Override
133     protected void calculateDayWidth() {
134         super.calculateDayWidth();
135         if (chartWidth == 0) {
136             //big chart with fixed day width
137             calendarXAxes.dayOfWeek.setWidth(MAX_DAY_WIDTH);
138         } else {
139             calendarXAxes.dayOfWeek.setWidth((chartWidth - Y_AXIS_WIDTH) / days);
140             calendarXAxes.dayOfWeek.setWidth(Math.min(calendarXAxes.dayOfWeek.getWidth(), MAX_DAY_WIDTH));
141             calendarXAxes.dayOfWeek.setWidth(Math.max(calendarXAxes.dayOfWeek.getWidth(), 1));
142             days = (chartWidth - Y_AXIS_WIDTH) / calendarXAxes.dayOfWeek.getWidth();
143         }
144     }

150     private void calculateWorkPerDay(Context context, Task task, BurnDownGuide guide) throws Exception {
151         LocalDateTime start = task.getStart();
152         LocalDateTime stop  = task.getFinish();
153         if (!task.isMilestone() && task.getResourceId() != null) {
154             if (GanttUtil.isValidTask(task) && !task.isMilestone() && task.getChildTasks().isEmpty()) {
155                 if (stop.isBefore(start) || stop.isEqual(start)) {
156                     request.exceptions.add(new ErrorException(String.format("Task %s finish time has to be after start time. Ignoring task.", task.getName())));
157                 } else {
158                     //                    Duration duration0 = task.getDuration();
159 //                    for (User assignment : task.getAssignedUser())
160                     {
161 //                        Resource resource =  assignment.getResource();
162 
163 //                        Number availability = assignment.getUnits();
164                         Number availability = task.getAssignedUser().getAvailabilities().getLast().getAvailability();
165                         if (/*resource != null &&*/ availability != null && context.debug.filterResource(task.getAssignedUser())) {
166                             int  startDayIndex = calculateDayIndex(start);
167                             int  stopDayIndex  = calculateDayIndex(stop);
168                             long oneDay        = 75 * SECONDS_PER_HOUR / 10;
169                             //                            long oneHour = SECONDS_PER_HOUR;
170                             if (stopDayIndex == startDayIndex) {
171                                 // ---We assume that a task is worked on every day from 8:00 - 12:00, 13:00 - 16:30 (7.5h, with lunch time at 12:00)
172                                 // start time might not be in the morning at exactly 8:00
173                                 // stop time might not be in the afternoon 15:30
174                                 LocalDateTime lunchStartTime = DateUtil.calculateLunchStartTime(start);
175                                 LocalDateTime lunchStopTime  = DateUtil.calculateLunchStopTime(start);
176                                 if (start.isBefore(lunchStartTime) || start.isEqual(lunchStartTime)) {
177                                     // if(start <= 12:00)
178                                     if (stop.isBefore(lunchStartTime) || stop.isEqual(lunchStartTime)) {
179                                         // if(stop <= 12:00)
180                                         // (stop - start)/7.5
181                                         double   fraction = (double) Duration.between(start, stop).getSeconds() / oneDay;
182                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
183                                         guide.add(startDayIndex, work);
184                                     } else if (stop.isAfter(lunchStopTime) || stop.isEqual(lunchStopTime)) {
185                                         // if(stop >= 13:00)
186                                         // ( stop - start -1h )/7.5
187                                         double   fraction = (double) Duration.between(start, stop).minusHours(1).getSeconds() / oneDay;
188                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
189                                         guide.add(startDayIndex, work);
190                                     } else {
191                                         throw new Exception(String.format("Task %s stop within lunch time", task.getName()));
192                                     }
193                                 } else if (start.isAfter(lunchStopTime) || start.isEqual(lunchStopTime)) {
194                                     // if(start >= 13:00 && stop >= 13:00)
195                                     // (stop - Start))/7.5
196                                     double   fraction = ((double) Duration.between(start, stop).getSeconds()) / oneDay;
197                                     Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
198                                     guide.add(startDayIndex, work);
199                                 } else {
200                                     throw new Exception(String.format("Task %s stop before start", task.getName()));
201                                 }
202 
203                             } else {
204                                 // start
205                                 {
206                                     // ---We assume that a task is worked on every day from 8:00 - 12:00, 13:00 -  16:30 (7.5h, with lunch time at 12:00)
207                                     // start time might not be in the morning at exactly 8:00
208                                     LocalDateTime lunchStartTime = DateUtil.calculateLunchStartTime(start);
209                                     LocalDateTime lunchStopTime  = DateUtil.calculateLunchStopTime(start);
210                                     if (start.isBefore(lunchStartTime) || start.isEqual(lunchStartTime)) {
211                                         double   fraction = ((double) Duration.between(start.toLocalTime(), LocalTime.of(16, 30)).minusHours(1).getSeconds()) / oneDay;
212                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
213                                         guide.add(startDayIndex, work);
214                                     } else if (start.isAfter(lunchStopTime) || start.isEqual(lunchStopTime)) {
215                                         double   fraction = ((double) Duration.between(start.toLocalTime(), LocalTime.of(16, 30)).minusHours(1).getSeconds()) / oneDay;
216                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
217                                         guide.add(startDayIndex, work);
218                                     } else {
219                                         throw new Exception(String.format("Task %s start within lunch time", task.getName()));
220                                     }
221                                 }
222                                 // end
223                                 {
224                                     // ---We assume that a task is worked on every day from 8:00 - 12:00, 13:00 - 16:30 (7.5h, with lunch time at 12:00)
225                                     // last day, stop time might not be in the afternoon 16:30
226                                     LocalDateTime lunchStartTime = DateUtil.calculateLunchStartTime(stop);
227                                     LocalDateTime lunchStopTime  = DateUtil.calculateLunchStopTime(stop);
228                                     if (stop.isBefore(lunchStartTime) || stop.isEqual(lunchStartTime)) {
229                                         // if(stop <= 12:00)
230                                         // (stop - 8:00)/7.5
231                                         double   fraction = ((double) Duration.between(LocalTime.of(8, 0), stop.toLocalTime()).getSeconds()) / oneDay;
232                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
233                                         guide.add(stopDayIndex, work);
234                                     } else if (stop.isAfter(lunchStopTime) || stop.isEqual(lunchStopTime)) {
235                                         // if(stop >= 13:00)
236                                         // (stop - 8:00 -1h)/7.5
237                                         double   fraction = ((double) Duration.between(LocalTime.of(8, 0), stop.toLocalTime()).minusHours(1).getSeconds()) / oneDay;
238                                         Duration work     = Duration.ofSeconds((long) ((fraction * availability.doubleValue() * SECONDS_PER_WORKING_DAY)));
239                                         guide.add(stopDayIndex, work);
240                                     } else {
241                                         throw new Exception(String.format("Task %s stop within lunch time", task.getName()));
242                                     }
243                                 }
244                             }
245                             if (startDayIndex + 1 < stopDayIndex) {
246                                 for (int index = startDayIndex + 1; index < stopDayIndex; index++) {
247                                     LocalDate today = calculateDayFromIndex(index);
248                                     if (task.getEffectiveCalendar().isWorkingDate(today))
249 //                                    if (isResourceWorkingDay(context, task.getAssignedUser(), today))
250                                     {
251                                         Duration work = Duration.ofSeconds((long) (availability.doubleValue() * SECONDS_PER_WORKING_DAY));
252                                         guide.add(index, work);
253                                     }
254                                 }
255                             }
256                         }
257                     }
258 
259                 }
260             }
261         }
262     }

268     protected void createBurnDownChart() {
269         graphics2D.setStroke(new BasicStroke(STANDARD_LINE_STROKE_WIDTH));
270 
271         LocalDate firstDay  = milestones.firstMilestone /*- (long) (60 * ONE_DAY_MILLIS / dayWidth)*/;
272         int       firstDayX = diagram.x + calendarXAxes.dayOfWeek.getWidth() / 2;
273         int       startX    = firstDayX + DateUtil.calculateDays(firstDay, milestones.get("S").time) * calendarXAxes.dayOfWeek.getWidth();
274 
275         drawAuthorLegend();
276         drawLegend();
277         if (maxWorked != null && !maxWorked.isZero()) {
278             // y axis markings
279             drawYAxes(startX, maxWorked);
280             if (worklog != null && worklog.size() != 0) {
281                 drawBurnDown(firstDay, firstDayX, maxWorked);
282             }
283             if (sprintClosed) {
284                 drawWatermark(startX - calendarXAxes.dayOfWeek.getWidth() / 2 + 10,
285                         chartHeight - calendarXAxes.getHeight() - calendarXAxes.year.getHeight() - 10, "CLOSED", graphicsTheme.watermarkColor);
286             } else if (isAbandonedProject()) {
287                 drawWatermark(startX - calendarXAxes.dayOfWeek.getWidth() / 2 + 10,
288                         chartHeight - calendarXAxes.getHeight() - calendarXAxes.year.getHeight() - 10, "ABANDONED", graphicsTheme.watermarkColor);
289             }
290             // Optimal burn down rate
291 
292             if (isPlannedBurnDownGuideAvailable()) {
293                 drawPlannedBurnDownGuide(firstDayX, ganttWorkWithoutBufferPerDayAccumulated);
294                 drawPlannedBurnDownGuide(firstDayX, ganttWorkWithBufferPerDayAccumulated);
295             } else {
296                 // ---liniar guide
297                 drawOptimalBurnDownGuide(firstDayX, eBestWork);
298             }
299 
300             if (eWorstWork != null) {
301                 drawOptimalBurnDownGuide(firstDayX, eWorstWork);
302             }
303 
304             // release extrapolation (Velocity)
305             if (milestones.get("R") != null) {
306                 graphics2D.setColor(extrapolationColor);
307                 int x1 = startX - calendarXAxes.dayOfWeek.getWidth() / 2 + 1;
308                 int y1 = diagram.y + diagram.height - calculateGraphHight(maxActualWorked);
309                 {
310                     int x = firstDayX + DateUtil.calculateDays(firstDay, milestones.get("R").time) * calendarXAxes.dayOfWeek.getWidth();
311                     graphics2D.drawLine(x1, y1, x, diagram.y + diagram.height);
312                 }
313             }
314         } else {
315             // ---Maybe this is a test case ticket?
316         }
317

320     protected void createMilestones(LocalDateTime start, LocalDateTime now, LocalDateTime end, LocalDateTime firstWorklog, LocalDateTime lastWorklog,
321                                     LocalDateTime release, boolean completed) {
322         if (start != null) {
323             milestones.add(start.toLocalDate(), "S", "Start (Start of project)", Color.blue);
324         }
325         milestones.add(now.toLocalDate(), "N", "Now (current date)", Color.blue);
326         if (end != null) {
327             milestones.add(end.toLocalDate(), "E", "End (End of project)", Color.blue);
328         }
329         if (release != null) {
330             milestones.add(release.toLocalDate(), "R", "Release (Estimated release date)", Color.blue);
331         }
332         if (isHideNow(now, end, completed)) {
333             milestones.remove("N");
334         }
335         if (firstWorklog != null && (start == null || !firstWorklog.toLocalDate().isEqual(start.toLocalDate()))) {
336             milestones.add(firstWorklog.toLocalDate(), "F", "First punch-in", Color.blue);
337         }
338         if (lastWorklog != null && (end == null || !lastWorklog.toLocalDate().isEqual(end.toLocalDate()))) {
339             milestones.add(lastWorklog.toLocalDate(), "L", "Last punch-out", Color.blue);
340         }
341         milestones.calculate();
342         // milestones.print();//debugging code
343     }

371     private void drawBurnDown(LocalDate firstDay, int firstDayX, Duration estimatedWork) {
372         // burn down graph
373         if (context.parameters.detailed) {
374             Integer[] graphHeight;
375             {
376                 int days = DateUtil.calculateDays(milestones.firstMilestone, DateUtil.max(milestones.lastMilestone, milestones.get("N").time)) + 1;
377                 graphHeight = new Integer[days + 3];
378                 for (int i = 0; i < days + 3; i++) {
379                     graphHeight[i] = 0;
380                 }
381             }
382             int authorIndex = 0;
383             // int sum = 0;
384             for (User user : usersTotalContribution.getSortedKeyList()) {
385                 AuthorContribution ac                  = usersTotalContribution.get(user);
386                 Duration           authorEstimatedWork = ac.worked.plus(ac.remaining);
387                 if (!authorEstimatedWork.isZero()) {
388                     // (worked + remaining));
389                     int yesterdayX  = 0;
390                     int yesterdayY  = 0;
391                     int yesterdayY2 = 0;
392                     int lastX       = 0;
393                     int lastY       = 0;
394                     int lastY2      = 0;
395                     int x           = 0;
396                     int y           = 0;
397                     int nowDayIndex = (DateUtil.calculateDays(milestones.firstMilestone, DateUtil.min(milestones.lastMilestone, milestones.get("N").time)));
398                     {
399                         int maxDayIndex = nowDayIndex;// Math.max(authorWorkPerDay.get(user).length, nowDayIndex);
400                         // logger.info(DateUtil.createDateString(calculateDayFromIndex(maxDayIndex), sdfMMMdd));
401                         for (int dayIndex = 0; dayIndex <= maxDayIndex + 2; dayIndex++) {
402                             x = firstDayX + dayIndex * calendarXAxes.dayOfWeek.getWidth();
403                             int                currentDayAuthorGraphHeight = 0;
404                             List<List<String>> transactions                = null;
405                             if (dayIndex <= maxDayIndex + 2) {
406                                 currentDayAuthorGraphHeight = calculateAuthorGraphHeight(usersWorkPerDayAccumulated.get(user)[dayIndex].duration,
407                                         authorEstimatedWork, maxWorked);
408                                 y                           = diagram.y + diagram.height - graphHeight[dayIndex] - currentDayAuthorGraphHeight;
409                                 if (dayIndex > 0) {
410                                     transactions = usersWorkPerDayAccumulated.get(user)[dayIndex - 1].transactions;
411                                 }
412                             }
413                             if (x != lastX) {
414                                 // ---a new day started, so we can draw the polygon of last day
415                                 if (yesterdayX != 0 && lastX != 0) {
416 //                                    User      author               = authors.getIdMap().get(user.getId());
417                                     LocalDate calendarFromDayIndex = calendarFromDayIndex(dayIndex - 2);
418                                     drawPolygon(yesterdayX, yesterdayY, yesterdayY2, lastX, lastY, lastY2,
419                                             authorIndex == usersTotalContribution.getSortedKeyList().size() - 1, DateUtil.isWorkDay(calendarFromDayIndex),
420                                             graphicsTheme.burnDownBorderColor, generateBurnDownColor(user.getColor()), transactions, user.getName());
421                                 }
422                                 yesterdayX  = lastX;
423                                 yesterdayY  = lastY;
424                                 yesterdayY2 = lastY2;
425                             }
426                             lastX = x;
427                             lastY = y;
428                             if (dayIndex <= maxDayIndex + 2) {
429                                                          lastY2 = diagram.y + diagram.height - graphHeight[dayIndex];
430                                 graphHeight[dayIndex] += currentDayAuthorGraphHeight;
431                             }
432                         }
433                     }
434                 }
435                 authorIndex++;
436             }
437         }
438         // draw border
439         if (!context.parameters.detailed) {
440             graphics2D.setStroke(new BasicStroke(STANDARD_LINE_STROKE_WIDTH));
441             int      lastX      = 0;
442             int      lastY      = 0;
443             Duration worked     = Duration.ZERO;
444             int      yesterdayX = 0;
445             int      yesterdayY = 0;
446             for (Worklog work : worklog) {
447                 if (work.getStart().toLocalDate().isBefore(milestones.get("N").time)) {
448                     worked = worked.plus(work.getTimeSpent());
449                     int x = firstDayX + DateUtil.calculateDays(firstDay, DateUtil.toDayPrecision(work.getStart())) * calendarXAxes.dayOfWeek.getWidth();
450                     if (x < firstDayX) {
451                         x = firstDayX;
452                     }
453                     int y = diagram.y + diagram.height - calculateGraphHight(estimatedWork.minus(worked));
454                     if (x != lastX) {
455                         // ---a new day started, so we can draw the polygon of last day
456                         if (yesterdayX != 0 && lastX != 0) {
457                             drawBorder(lastX, lastY, yesterdayX, yesterdayY);
458                         }
459                         yesterdayX = lastX;
460                         yesterdayY = lastY;
461                     }
462                     lastX = x;
463                     lastY = y;
464                 }
465             }
466             // ---Draw last polygon
467             drawBorder(lastX, lastY, yesterdayX, yesterdayY);
468             // to now
469             if (worklog.size() != 0) {
470                 int x = firstDayX + DateUtil.calculateDays(firstDay, milestones.get("N").time) * calendarXAxes.dayOfWeek.getWidth();
471                 if (x != lastX) {
472                     drawBorder(lastX, lastY, x, lastY);
473                 }
474             }
475         }
476     }

482     protected void drawOptimalBurnDownGuide(int firstDayX, Duration estimatedWork) {
483         LocalDate firstDay          = milestones.get("S").time;
484         LocalDate lastDay           = milestones.get("E").time;
485         int       workingDays       = DateUtil.calculateWorkingDaysIncluding(firstDay, lastDay);

621     public void init() throws IOException {
622         initSize(Y_AXIS_WIDTH, 0, true);
623         if (milestones.get("R") != null && milestones.get("R").time.isAfter(milestones.get("E").time)) {
624             extrapolationColor = graphicsTheme.delayEventColor;
625         } else {
626             extrapolationColor = graphicsTheme.inTimeColor;
627         }
628 
629         calculateAuthorContribution(worklog, worklogRemaining, usersTotalContribution);
630 
631         Map<User, Duration> authorWorkSum = new HashMap<>();// total work done by any author
632         for (User user : usersTotalContribution.getSortedKeyList()) {
633             if (authorWorkSum.get(user) == null) {
634                 authorWorkSum.put(user, Duration.ZERO);
635             }
636         }
637         int days = DateUtil.calculateDays(milestones.firstMilestone, milestones.lastMilestone) + 1;
638         if (worklog != null) {
639             int lastDayIndexWithValue = 0;// the last day any author has data
640             for (Worklog work : worklog) {
641                 Duration  aws     = authorWorkSum.get(request.getuser(work.getAuthorId()));
642                 DayWork[] dayWork = usersWorkPerDayAccumulated.get(request.getuser(work.getAuthorId()));
643                 if (dayWork == null) {
644                     // create a list of work per day for authors that have worked
645                     dayWork    = new DayWork[days + 3];// first value is 0
646                     dayWork[0] = new DayWork();
647                     usersWorkPerDayAccumulated.put(request.getuser(work.getAuthorId()), dayWork);
648                 }
649                 aws = aws.plus(work.getTimeSpent());
650                 authorWorkSum.put(request.getuser(work.getAuthorId()), aws);
651                 int day = (DateUtil.calculateDays(milestones.firstMilestone, DateUtil.toDayPrecision(work.getStart())));
652                 if (day < 0) {
653                     // ignore any data before first day of sprint
654                     day = 0;
655                 }
656                 if (day < days) {
657                     {
658                         if (dayWork[day + 1] != null) {
659                             dayWork[day + 1].setDuration(aws);
660                             dayWork[day + 1].add(work);
661                         } else {
662                             dayWork[day + 1] = new DayWork(aws);// every day has the amount of work done at that day and all days before that
663                             dayWork[day + 1].add(work);
664                         }
665                         if (!aws.isZero()) {
666                             lastDayIndexWithValue = Math.max(day, lastDayIndexWithValue);
667                         }
668                     }
669                 } else {
670                     numberOfWorkExceptions++;
671                     logger.error(WORK_OUTSIDE_ALLOWED_TIME_BOUNDARIES_OCCURRED);
672                 }
673 
674             }
675             milestones.add(calendarFromDayIndex(lastDayIndexWithValue + 1), "L", "last value + 1", Color.red, true);
676         }
677         milestones.calculate();
678         int nowDayIndex = (DateUtil.calculateDays(milestones.firstMilestone, DateUtil.min(milestones.lastMilestone, milestones.get("N").time)));
679         for (User user : usersTotalContribution.getSortedKeyList()) {
680             Duration  last = Duration.ZERO;
681             DayWork[] aw   = usersWorkPerDayAccumulated.get(user);
682             if (aw == null) {
683                 // create a list of work per day for authors that have no work done yet and thus
684                 // where missed in the above use case involving worklog
685                 aw    = new DayWork[days + 3];
686                 aw[0] = new DayWork();
687                 usersWorkPerDayAccumulated.put(user, aw);
688             }
689             // fill in empty parts of the days list
690             for (int i = 1; i < usersWorkPerDayAccumulated.get(user).length; i++) {
691                 if (i <= nowDayIndex + 2) {
692                     if (usersWorkPerDayAccumulated.get(user)[i] == null
693                             || last.toSeconds() > usersWorkPerDayAccumulated.get(user)[i].duration.toSeconds()) {
694                         usersWorkPerDayAccumulated.get(user)[i] = new DayWork(last);
695                     }
696                     last = usersWorkPerDayAccumulated.get(user)[i].duration;
697                 } else {
698                     // ignore anything after today (only makes sense in a test scenario, where we
699                     // simulate a now time in the past)
700                     if (usersWorkPerDayAccumulated.get(user)[i] == null) {
701                         usersWorkPerDayAccumulated.get(user)[i] = new DayWork(Duration.ZERO);
702                     }
703                 }
704             }
705         }
706 
707         if (usersTotalContribution.size() != 0) {
708             for (User user : usersTotalContribution.getSortedKeyList()) {
709                 authors.add(user);
710             }
711         }
712 //        authors.calculateColors(graphicsTheme, true);
713     }
