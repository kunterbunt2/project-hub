18  package de.bushnaq.abdalla.projecthub.report.burndown;
19  
20  import de.bushnaq.abdalla.projecthub.dto.Worklog;
21  import de.bushnaq.abdalla.util.date.DateUtil;
22  
23  import java.time.Duration;
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  public class DayWork {
28      public Duration           duration     = Duration.ZERO;
29      public List<List<String>> transactions = new ArrayList<>();
30  
31      public DayWork() {
32      }
33  
34      public DayWork(Duration duration) {
35          this.duration = duration;
36      }
37  
38      public void add(Worklog work) {
39          List<String> list = new ArrayList<>();
40          //TODO get key
41  //        list.add(String.format("%s", work.getTaskId() );
42          list.add(String.format("<b>%s</b>", DateUtil.create24hDurationString(work.getTimeSpent(), false, true, false)));
43          list.add(String.format("%s", work.getComment()));
44          transactions.add(list);
45      }
46  
47      public void setDuration(Duration duration) {
48          this.duration = duration;
49      }
50  
51      public static String transactionsToTooltips(List<List<String>> transactions, String authorName) {
52          if (transactions != null && !transactions.isEmpty()) {
53              String tooltip = authorName + " <table><tr> <th><b>Key</b></th> <th><b>Work</b></th> <th><b>Summary</b></th></tr>";
54  
55              for (List<String> ll : transactions) {
56                  tooltip += "<tr>";
57                  for (String s : ll) {
58                      tooltip += String.format("<td>%s</td>", s);
59                  }
60                  tooltip += "</tr>";
61              }
62              tooltip += "</table>";
63              return tooltip;
64          }
65          return "";
66      }

68  }
