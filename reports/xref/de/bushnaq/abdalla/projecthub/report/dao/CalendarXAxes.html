80      protected int calculateDayX(LocalDate date) {
81          int firstMilestoneX = x + dayOfWeek.getWidth() / 2;
82          return firstMilestoneX + (DateUtil.calculateDays(milestones.firstMilestone, date) + preRun) * dayOfWeek.getWidth();
83      }

85      public void drawCalendar(boolean drawDays) {
86          LocalDate firstDay = DateUtil.addDay(milestones.firstMilestone, -preRun);
87          //TODO why are we ignoring postRun days?
88          LocalDate lastDay           = DateUtil.max(milestones.lastMilestone, DateUtil.addDay(milestones.firstMilestone, parent.days - 1));
89          boolean   yearWasDrawn      = false;
90          boolean   monthWasDrawn     = false;
91          boolean   firstWeekWasDrawn = false;
92  
93          // logger.trace(String.format("%s first=%s, last=%s, day.width=%d",
94          // father.getImageName(), DateUtil.createDateString(firstDay, sdtmf),
95          // DateUtil.createDateString(lastDay, sdtmf), dayOfWeek.width));
96          for (int phase = 0; phase < 5; phase++) {
97              for (LocalDate currentDay = firstDay; currentDay.isBefore(lastDay) || currentDay.isEqual(lastDay); currentDay = currentDay.plusDays(1)) {
98                  // if (phase == 4)
99                  // logger.trace(String.format("first=%s, last=%s, current=%s",
100                 // Util.createDateString(firstDay, sdtmf), Util.createDateString(lastDay,
101                 // sdtmf), Util.createDateString(currentDay, sdtmf)));
102                 int       daysX    = calculateDayX(currentDay);
103                 String[]  weekDays = {"M", "T", "W", "T", "F", "S", "S"};
104                 String[]  months   = {"Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
105                 LocalDate startCal = currentDay;
106 
107                 if (phase == 4 && ((startCal.getDayOfMonth() == 1 && startCal.getMonth() == Month.JANUARY) || !yearWasDrawn)) {
108                     // --year
109                     {
110                         LocalDate end = startCal.withMonth(12).withDayOfMonth(31);// new years eve
111                         //                        end.set(Calendar.MONTH, 11); // 11 = december
112                         //                        end.set(Calendar.DAY_OF_MONTH, 31); // new years eve
113                         if (end.isAfter(lastDay)) {
114                             end = lastDay;
115                         }
116                         int x2 = calculateDayX(end) - dayOfWeek.getWidth() / 2;
117                         drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), x2 + dayOfWeek.getWidth(), year.getY(), year.getHeight(),
118                                 String.valueOf(startCal.getYear()), parent.graphicsTheme.ganttYearTextColor, parent.graphicsTheme.ganttYearBackgroundColor,
119                                 parent.graphicsTheme.ganttYearBoderColor, year.getFont(), false);
120                         yearWasDrawn = true;
121                     }
122                 } else if (phase == 3 && (startCal.getDayOfMonth() == 1 || !monthWasDrawn) && isMonthVisible()) {
123                     // --month
124                     {
125                         LocalDate end = startCal.plusMonths(1).withDayOfMonth(1).minusDays(1);
126                         if (end.isAfter(lastDay)) {
127                             end = lastDay;
128                         }
129                         int   x2              = calculateDayX(end) - dayOfWeek.getWidth() / 2;
130                         Color backgroundColor = parent.graphicsTheme.monthColor[startCal.getMonth().getValue() - 1];
131                         drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), x2 + dayOfWeek.getWidth(), month.getY(), month.getHeight(),
132                                 months[startCal.getMonth().getValue() - 1], parent.graphicsTheme.ganntMonthTextColor, backgroundColor,
133                                 parent.graphicsTheme.ganttMonthBorderColor, month.getFont(), false);
134                         monthWasDrawn = true;
135                     }
136                 } else if (phase == 2 && (startCal.getDayOfWeek() == DayOfWeek.MONDAY || !firstWeekWasDrawn) && isWeekVisible()) {
137                     // --week of the year
138                     {
139                         LocalDate end = DateUtil.getWeekSunday(startCal);
140                         if (end.isAfter(lastDay)) {
141                             end = lastDay;
142                         }
143                         int    x2 = calculateDayX(end) - dayOfWeek.getWidth() / 2;
144                         String calendarWeek;
145                         if (isDayOfWeekVisible()) {
146                             TemporalField woy = WeekFields.of(Locale.CANADA).weekOfWeekBasedYear();
147                             calendarWeek = "W" + currentDay.get(woy);
148                         } else {
149                             calendarWeek = DateUtil.createDateString(currentDay, sdf);
150                         }
151 //                        drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), x2 + dayOfWeek.getWidth(), week.getY(), week.getHeight(), calendarWeek,
152 //                                parent.graphicsTheme.weekTextColor, parent.graphicsTheme.weekBackgroundColor, parent.graphicsTheme.weekBoderColor,
153 //                                week.getFont(), false);
154                         drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), x2 + dayOfWeek.getWidth(), week.getY(), week.getHeight() - 1, calendarWeek,
155                                 parent.graphicsTheme.ganttWeekDayTextColor, parent.graphicsTheme.ganttWeekDayBgColor, parent.graphicsTheme.ganttWeekBoderColor,
156                                 week.getFont(), false);
157                         firstWeekWasDrawn = true;
158                     }
159                 } else if (phase == 1 && isDayOfWeekVisible()) {
160                     // --day of Month
161                     {
162                         Color color;
163                         if (startCal.getDayOfWeek() == DayOfWeek.SATURDAY || startCal.getDayOfWeek() == DayOfWeek.SUNDAY) {
164 //                            color = GraphColorUtil.getDayOfWeekColor(parent.graphicsTheme, startCal);
165 //                            drawTextBox(daysX - (dayOfMonth.getWidth() / 2 - 1), daysX - (dayOfMonth.getWidth() / 2 - 1) + (dayOfMonth.getWidth() - 1),
166 //                                    dayOfMonth.getY(), dayOfMonth.getHeight(), "" + startCal.getDayOfMonth(), Color.BLACK, color,
167 //                                    parent.graphicsTheme.dayOfMonthBorderColor, dayOfMonth.getFont(), true);
168                             drawTextBox(daysX - (dayOfMonth.getWidth() / 2 - 1), daysX - (dayOfMonth.getWidth() / 2 - 1) + (dayOfMonth.getWidth() - 1),
169                                     dayOfMonth.getY(), dayOfMonth.getHeight(), "" + startCal.getDayOfMonth(), parent.graphicsTheme.ganttWeekendTextColor, parent.graphicsTheme.ganttSundayStripeColor,
170                                     parent.graphicsTheme.ganttDayOfMonthBorderColor, dayOfMonth.getFont(), true);
171                         } else {
172 //                            color = parent.graphicsTheme.weekDayBgColor;
173 //                            drawTextBox(daysX - (dayOfMonth.getWidth() / 2 - 1), daysX - (dayOfMonth.getWidth() / 2 - 1) + (dayOfMonth.getWidth() - 1),
174 //                                    dayOfMonth.getY(), dayOfMonth.getHeight(), "" + startCal.getDayOfMonth(), parent.graphicsTheme.dayOfMonthTextColor, color,
175 //                                    parent.graphicsTheme.dayOfMonthBorderColor, dayOfMonth.getFont(), true);
176                             drawTextBox(daysX - (dayOfMonth.getWidth() / 2 - 1), daysX - (dayOfMonth.getWidth() / 2 - 1) + (dayOfMonth.getWidth() - 1),
177                                     dayOfMonth.getY(), dayOfMonth.getHeight(), "" + startCal.getDayOfMonth(), parent.graphicsTheme.ganttNormalDayTextColor, parent.graphicsTheme.ganttWeekDayBgColor,
178                                     parent.graphicsTheme.ganttDayOfMonthBorderColor, dayOfMonth.getFont(), true);
179                         }
180 
181                         // father.graphics2D.setColor(graphicsTheme.dayDiagramBorderColor);
182                     }
183 
184                     // --day of week
185                     {
186                         Color color = GraphColorUtil.getDayOfWeekStripeColor(parent.graphicsTheme, startCal);
187 //                        drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), daysX - (dayOfWeek.getWidth() / 2 - 1) + (dayOfWeek.getWidth() - 1),
188 //                                dayOfWeek.getY(), dayOfWeek.getHeight(), weekDays[startCal.getDayOfWeek().getValue() - 1], parent.graphicsTheme.dayTextColor,
189 //                                color, parent.graphicsTheme.dayBorderColor, dayOfWeek.getFont(), true);
190                         drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), daysX - (dayOfWeek.getWidth() / 2 - 1) + (dayOfWeek.getWidth() - 1),
191                                 dayOfWeek.getY(), dayOfWeek.getHeight(), weekDays[startCal.getDayOfWeek().getValue() - 1], parent.graphicsTheme.ganttWeekDayTextColor,
192                                 color, parent.graphicsTheme.ganttDayBorderColor, dayOfWeek.getFont(), true);
193 //                        parent.graphics2D.setColor(parent.graphicsTheme.dayDiagramBorderColor);
194 //                        parent.graphics2D.fillRect(daysX - (dayOfWeek.getWidth() / 2 - 1) + (dayOfWeek.getWidth() - 1), milestone.flagY, 1, milestone.flagHeight - 1);
195                     }
196                 } else if (phase == 0) {
197                     {
198                         if (drawDays && isDayBarsVisible()) {
199                             parent.drawDayBars(currentDay);
200                         }
201                     }
202                     if (milestonesVisible()) {
203                         Color color = GraphColorUtil.getDayOfWeekStripeColor(parent.graphicsTheme, startCal);
204                         parent.graphics2D.setColor(color);
205                         //                        father.graphics2D.setColor(Color.red);
206 //                        parent.graphics2D.fillRect(daysX - (dayOfWeek.getWidth() / 2 - 1), milestone.flagY, dayOfWeek.getWidth() - 1, milestone.flagHeight - 1);
207                         drawTextBox(daysX - (dayOfWeek.getWidth() / 2 - 1), daysX - (dayOfWeek.getWidth() / 2 - 1) + (dayOfWeek.getWidth() - 1),
208                                 milestone.flagY, milestone.flagHeight, null, parent.graphicsTheme.ganttWeekDayTextColor,

220     public String drawMilestone(Milestone m, LocalDate time, int x, int y, Color fillColor, String text, boolean drawMilestone, Integer flagY,
221                                 Color flagTextColor, boolean drawFlag, boolean drawNowLine) {
222         String imageMap = "";
223         int    centerX  = x;
224         int    centerY  = y;
225         if (text.startsWith("N")) {
226             // now line
227             if (drawNowLine) {
228                 parent.graphics2D.setColor(GraphicsTheme.COLOR_DARK_RED);
229                 parent.graphics2D.fillRect(x, parent.diagram.y, 2, parent.diagram.height);
230                 int d = Math.max(dayOfWeek.getWidth() / 3, 6);
231                 if (calendarAtBottom) {
232                     parent.graphics2D.fillOval(x + 1 - d / 2, parent.diagram.y - d / 2, d, d);
233                 } else {
234                     parent.graphics2D.fillOval(x + 1 - d / 2, parent.diagram.y + parent.diagram.height - d, d, d);
235                 }
236             }
237         }
238         if (drawMilestone) {
239             // draw milestone
240             parent.graphics2D.setFont(milestone.font);
241             parent.graphics2D.setColor(fillColor);
242             parent.graphics2D.fillRect(centerX - milestone.width / 2, centerY, milestone.width, milestone.height - 1);
243             //            if (m != null) {
244             //                String lable = DateUtil.createDateString(time, imageMapSdf);
245             //                String toolTip = String.format("<b>%s</b> = %s<br>%s.<br>", text, m.name, lable);
246             //                Shape s = new TitledRectangle(centerX - milestone.width / 2, centerY, milestone.width, milestone.height - 1, toolTip);
247             //                father.graphics2D.fill(s);
248             //            } else {
249             //                Shape s = new Rectangle(centerX - milestone.width / 2, centerY, milestone.width, milestone.height - 1);
250             //                father.graphics2D.fill(s);
251             //            }
252             parent.graphics2D.setColor(parent.graphicsTheme.milestoneTextColor);
253             FontMetrics fm        = parent.graphics2D.getFontMetrics();
254             int         maxAscent = fm.getMaxAscent();
255             {
256                 if (m != null) {
257                     String lable   = DateUtil.createDateString(time, imageMapSdf);
258                     String toolTip = String.format("<b>%s</b> = %s<br>%s.<br>", text, m.name, lable);
259                     parent.graphics2D.drawString(text, x - milestone.width / 2 + 2, centerY + milestone.height / 2 + maxAscent / 2 - 2, toolTip);
260                 } else {
261                     parent.graphics2D.drawString(text, x - milestone.width / 2 + 2, centerY + milestone.height / 2 + maxAscent / 2 - 2);
262                 }
263             }
264             if (m != null) {
265                 // imageMap
266                 String lable = DateUtil.createDateString(time, imageMapSdf);
267                 imageMap += "<area alt=\"<font face=arial>";
268                 imageMap += String.format("<b>%s</b> = %s<br>%s.<br>", text, m.name, lable);
269                 imageMap += String.format("\" shape=\"rect\" coords=\"%d,%d,%d,%d\"", AbstractCanvas.transformToMapX(centerX - milestone.width / 2),
270                         AbstractCanvas.transformToMapY(centerY), AbstractCanvas.transformToMapX(centerX + milestone.width / 2),
271                         AbstractCanvas.transformToMapY(centerY + milestone.height - 1));
272                 imageMap += " >\n";
273             }
274         }
275         if (drawMilestone && drawFlag) {
276             parent.graphics2D.setFont(milestone.flagFont);
277             FontMetrics fm    = parent.graphics2D.getFontMetrics();
278             String      lable = DateUtil.createDateString(time, sdfMMMdd);
279             int         width = fm.stringWidth(lable);
280             // flag background
281             parent.graphics2D.setColor(parent.graphicsTheme.milestoneFlagColor);
282             parent.graphics2D.fillRect(x - milestone.width / 2, flagY, width, milestone.flagHeight);
283             if (calendarAtBottom) {
284                 // flag pole
285                 parent.graphics2D.setColor(flagTextColor);
286                 parent.graphics2D.fillRect(centerX, milestone.flagY + milestone.flagHeight - 4, 1, 3);
287                 // flag
288                 parent.graphics2D.drawString(lable, x - milestone.width / 2, milestone.flagY + milestone.flagHeight - 5);
289             } else {
290                 // flag pole
291                 parent.graphics2D.setColor(flagTextColor);
292                 parent.graphics2D.fillRect(centerX, centerY + milestone.height, 1, 3);
293                 // flag
294                 parent.graphics2D.drawString(lable, x - milestone.width / 2, milestone.flagY + milestone.flagHeight - 1);
295             }
296         }
297         return imageMap;
298     }

308     public void drawTextBox(int x1, int x2, Integer y1, int height, String text, Color textColor, Color backgroundColor, Color borderColor, Font font,
309                             boolean centered) {
310         {
311             parent.graphics2D.setColor(backgroundColor);
312             parent.graphics2D.fillRect(x1, y1, x2 - x1, height - 1);
313         }
314         {
315             parent.graphics2D.setColor(borderColor);
316             parent.graphics2D.fillRect(x1 - 1, y1 - 1, x2 - x1 + 1, 1);
317             parent.graphics2D.fillRect(x2, y1, 1, height - 1);
318         }
319         if (text != null) {
320             parent.graphics2D.setFont(font);
321             FontMetrics fm    = parent.graphics2D.getFontMetrics();
322             int         width = fm.stringWidth(text);
323             parent.graphics2D.setColor(textColor);
324             int maxAscent = fm.getMaxAscent();
325             if (centered) {
326                 parent.graphics2D.drawString(text, x1 + (x2 - x1) / 2 - width / 2, y1 + height / 2 + (maxAscent + 1) / 2 - 2);
327             } else {
328                 parent.graphics2D.drawString(text, x1 + 1 + 1, y1 + height / 2 + (maxAscent + 1) / 2 - 2);
329             }
330         }

333     public int getHeight() {
334         int height = year.getHeight();
335         if (isMonthVisible()) {
336             height += month.getHeight();
337         }
338         if (isWeekVisible()) {
339             height += week.getHeight();
340         }
341         if (isDayOfWeekVisible()) {
342             height += dayOfWeek.getHeight();
343             height += dayOfMonth.getHeight();
344         } else if (milestonesVisible()) {
345             height += milestone.height;
346         }
347         if (milestonesVisible()) {
348             height += milestone.flagHeight;
349         }
350         return height;
351     }

373     public void initPosition(int x, int y) {
374         this.x = x;
375         if (calendarAtBottom) {
376             // flag
377             // milestone, dayOfWeek
378             // dayOfMonth
379             // week
380             // month
381             // year
382 
383             milestone.flagY = y;
384             milestone.y     = milestone.flagY + milestone.flagHeight;
385             dayOfWeek.setY(milestone.y);
386             dayOfMonth.setY(dayOfWeek.getY() + dayOfWeek.getHeight());
387             if (isDayOfWeekVisible()) {
388                 if (isDayOfMonthVisible()) {
389                     week.setY(dayOfMonth.getY() + dayOfMonth.getHeight());
390                 } else {
391                     week.setY(dayOfWeek.getY() + dayOfWeek.getHeight());
392                 }
393             } else {
394                 week.setY(milestone.y + milestone.height);
395             }
396             if (isWeekVisible()) {
397                 month.setY(week.getY() + week.getHeight());
398             } else {
399                 month.setY(dayOfWeek.getY() + dayOfWeek.getHeight());
400             }
401             year.setY(month.getY() + month.getHeight());
402         } else {
403             // year
404             // month
405             // week
406             // dayOfMonth
407             // milestone, dayOfWeek
408             // flag
409             year.setY(y);
410             month.setY(year.getY() + year.getHeight());
411             week.setY(month.getY() + month.getHeight());
412             dayOfMonth.setY(week.getY() + week.getHeight());
413             if (isDayOfMonthVisible()) {
414                 dayOfWeek.setY(dayOfMonth.getY() + dayOfMonth.getHeight());
415             } else {
416                 dayOfWeek.setY(week.getY() + week.getHeight());
417             }
418             milestone.y     = dayOfWeek.getY();
419             milestone.flagY = dayOfWeek.getY() + milestone.height;
420         }
421     }
