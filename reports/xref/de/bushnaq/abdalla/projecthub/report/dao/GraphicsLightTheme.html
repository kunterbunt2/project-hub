197     private int generateColors(int ps, int pg, int ss, int sg, int ts, int tg, int i, int a) {
198         burnDownColor[i++] = new Color(ps, sg, ts, a);
199         burnDownColor[i++] = new Color(ts, pg, ss, a);
200         burnDownColor[i++] = new Color(ss, tg, ps, a);
201         return i;
