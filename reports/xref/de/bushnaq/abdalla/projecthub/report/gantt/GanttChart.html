18  package de.bushnaq.abdalla.projecthub.report.gantt;
19  
20  import de.bushnaq.abdalla.projecthub.Context;
21  import de.bushnaq.abdalla.projecthub.dto.Sprint;
22  import de.bushnaq.abdalla.projecthub.report.AbstractChart;
23  import de.bushnaq.abdalla.projecthub.report.dao.BurnDownGraphicsTheme;
24  
25  import java.time.LocalDateTime;
26  import java.util.List;
27  
28  
29  public class GanttChart extends AbstractChart {
30  
31      public GanttChart(Context context, String projectRequestKey, String relateCssPath, String column, String sprintName, List<Throwable> exception,
32                        LocalDateTime now, boolean completed, Sprint sprint, String cssClass,
33                        BurnDownGraphicsTheme graphicsTheme) throws Exception {
34          super("Gantt Chart", projectRequestKey, relateCssPath, column, sprintName, null, cssClass, graphicsTheme);
35          getRenderers().add(new GanttRenderer(context, sprintName, exception, now, completed, sprint, cssClass, graphicsTheme));
36          this.setChartWidth(getRenderers().getFirst().chartWidth);
37          this.setChartHeight(getRenderers().getFirst().chartHeight + captionElement.height + footerElement.height - 1);
38          captionElement.width = getChartWidth();
39          footerElement.y      = getRenderers().getFirst().chartHeight + captionElement.height;
40      }
41  
42      @Override
43      protected void createReport() throws Exception {
44          getRenderers().getFirst().draw(graphics2D, 0, captionElement.height);
45      }
46  
47  }
