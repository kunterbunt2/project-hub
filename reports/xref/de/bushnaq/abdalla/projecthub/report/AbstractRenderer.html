109     protected int calculateDayX(LocalDate date) {
110         LocalDate firstMilestoneDay = milestones.firstMilestone;
111         int       firstMilestoneX   = firstDayX + calendarXAxes.dayOfWeek.getWidth() / 2;
112         return firstMilestoneX + (DateUtil.calculateDays(firstMilestoneDay, date) + calendarXAxes.getPriRun()) * calendarXAxes.dayOfWeek.getWidth();
113     }
114 
115     protected int calculateMaxDays() {
116         return DateUtil.calculateDays(milestones.firstMilestone, milestones.lastMilestone) + 1 + calendarXAxes.getPriRun() + calendarXAxes.getPostRun();
117     }

119     protected int calculateX(LocalDateTime date, LocalDateTime startTime, long secondsPerDay) {
120         LocalDate firstMilestoneDay = milestones.firstMilestone;
121         int       firstMilestoneX   = firstDayX + calendarXAxes.dayOfWeek.getWidth() / 2;
122         // String createDateTimeString = DateUtil.createDateTimeString(date);
123         //        LocalDateTime timeOfDay = LocalDateTimeUtil.getTimeOfDayInMillis(date);
124         Duration workedToday = Duration.between(startTime, date);
125         int dayX = firstMilestoneX
126                 + (DateUtil.calculateDays(firstMilestoneDay, DateUtil.toDayPrecision(date)) + calendarXAxes.getPriRun()) * calendarXAxes.dayOfWeek.getWidth();
127         int timeOfDayX = (int) (((workedToday.getSeconds()) * calendarXAxes.dayOfWeek.getWidth()) / secondsPerDay);
128         return dayX + timeOfDayX;
129     }

205     protected void drawGraphText(int x, int y, String text, Color textColor, Font font, TextAlignment aligned) {
206         // flag
207         graphics2D.setFont(font);
208         FontMetrics fm    = graphics2D.getFontMetrics();
209         int         width = fm.stringWidth(text);
210         graphics2D.setColor(graphicsTheme.graphTextBackgroundColor);
211         switch (aligned) {
212             case left:
213                 graphics2D.fillRect(x, y - 9 + 2, width, 12);
214                 graphics2D.setColor(textColor);
215                 graphics2D.drawString(text, x, y + 2);
216                 break;
217             case right:
218                 graphics2D.fillRect(x - width, y - 9 + 2, width, 12);
219                 graphics2D.setColor(textColor);
220                 graphics2D.drawString(text, x - width, y + 2);
221                 break;
222         }

224     }

226     protected void drawLegend(int x, int y, Color interpolationColor) {
227         int lineHeight  = 14;
228         int legendY     = y + lineHeight;
229         int legendX1    = x;
230         int legendX2    = legendX1 + 10;
231         int legendTextY = legendY + 1;
232         int legendTextX = legendX2 + 4;
233         int milestoneX  = legendX1 + 5;
234         int milestoneY  = legendY - calendarXAxes.milestone.height / 2;
235 
236         graphics2D.setStroke(new BasicStroke(STANDARD_LINE_STROKE_WIDTH, BasicStroke.CAP_BUTT, BasicStroke.JOIN_BEVEL, 0, new float[]{3}, 0));
237         graphics2D.setColor(graphicsTheme.surroundingSquareColor);
238         graphics2D.drawLine(legendX1, legendY, legendX2, legendY);
239         drawGraphText(legendTextX, legendTextY, "Guideline", graphicsTheme.tickTextColor, calendarXAxes.dayOfWeek.getFont(), TextAlignment.left);

241         legendY += lineHeight;
242         legendTextY += lineHeight;
243         milestoneY += lineHeight;

306     protected void initPosition(int x, int y) throws IOException {
307         firstDayX = x;
308         if (calendarAtBottom) {
309             calendarXAxes.initPosition(x, y);
310             diagram.initPosition(x, y);
311             calendarXAxes.initPosition(x, diagram.y + diagram.height + 1);
312         } else {
313             calendarXAxes.initPosition(x, y);
314             diagram.initPosition(x, calendarXAxes.year.getY() + calendarXAxes.getHeight());
315         }
316     }

318     protected void initSize(int x, int y, boolean calendarAtBottom) throws IOException {
319         this.calendarAtBottom = calendarAtBottom;
320         // this.calendarAtBottom = calendarAtBottom;
321         calculateDayWidth();
322         chartWidth      = calculateChartWidth();
323         chartHeight     = calculateChartHeight();
324         firstDayX       = x;
325         bankHolidayFont = new Font(Font.SANS_SERIF, Font.PLAIN, Math.min(14, (int) (calendarXAxes.dayOfWeek.getWidth() * 1.1)));
326 
327         if (calendarAtBottom) {
328             calendarXAxes.initSize(chartWidth, calendarXAxes.dayOfWeek.getWidth(), calendarAtBottom);
329             diagram.initSize(chartWidth - x, chartHeight - calendarXAxes.getHeight());
330             calendarXAxes.initSize(chartWidth, calendarXAxes.dayOfWeek.getWidth(), calendarAtBottom);
331         } else {
332             calendarXAxes.initSize(chartWidth, calendarXAxes.dayOfWeek.getWidth(), calendarAtBottom);
333             diagram.initSize(chartWidth, chartHeight - calendarXAxes.getHeight());
334         }
335     }
