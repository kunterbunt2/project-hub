88      void drawCalendar(Graphics2D graphics2D, int year, LocalDate today, int x, int y) throws IOException, FontFormatException {
89          int holidays = 0;
90  
91          // Save original font and color
92          Font  originalFont  = graphics2D.getFont();
93          Color originalColor = graphics2D.getColor();
94  
95          // Fonts
96          Font smallFont = new Font(Font.SANS_SERIF, Font.PLAIN, 6);
97  
98          int yearTextHeight = drawYear(graphics2D, year, x, y);
99  
100         // Calculate offset for calendar to position below the year text
101         int yearOffset = yearTextHeight + 20; // Add some padding
102 
103         // For each month
104         for (int monthIndex = 0; monthIndex < 12; monthIndex++) {
105             int row = monthIndex / MONTHS_PER_ROW;
106             int col = monthIndex % MONTHS_PER_ROW;
107 
108             int monthStartX = x + col * (MONTH_WIDTH + MONTH_GAP);
109             int monthStartY = y + yearOffset + row * (MONTH_HEIGHT + MONTH_GAP); // Apply the year offset
110 
111             // Create LocalDate for first day of month
112             LocalDate firstOfMonth = LocalDate.of(year, monthIndex + 1, 1);
113             String    monthName    = firstOfMonth.getMonth().toString();
114 
115             drawMonthName(graphics2D, monthStartX, monthStartY, monthName);
116 
117             drawWeekDays(graphics2D, monthStartX, monthStartY, firstOfMonth);
118 
119             // Determine first day of week for this month (0 = Sunday, 6 = Saturday)
120             int firstDayOfWeek = firstOfMonth.getDayOfWeek().getValue() % 7;
121             if (firstDayOfWeek == 0) firstDayOfWeek = 7;  // Convert Sunday from 0 to 7
122             firstDayOfWeek--; // Adjust to 0-based index
123 
124             // Last day of month
125             int lastDay = firstOfMonth.lengthOfMonth();
126 
127             // Calculate starting Y position for the first week, adding adequate space after weekday headers
128             int daysStartY = monthStartY + HEADER_HEIGHT + WEEKDAY_HEADER_HEIGHT + 10; // Added 10px extra spacing
129 
130             // Draw days from previous month to fill the gaps at the beginning
131             if (firstDayOfWeek > 0) {
132                 // Get the previous month
133                 LocalDate prevMonth        = firstOfMonth.minusMonths(1);
134                 int       prevMonthLastDay = prevMonth.lengthOfMonth();
135 
136                 // Draw the days from the previous month
137                 for (int i = 0; i < firstDayOfWeek; i++) {
138                     int prevDay    = prevMonthLastDay - firstDayOfWeek + i + 1;
139                     int dayX       = monthStartX + 10 + (i * DAY_SIZE);
140                     int dayY       = daysStartY;
141                     int dayCenterX = dayX + (DAY_SIZE / 2);
142                     int dayCenterY = dayY + (DAY_SIZE / 2) - 6;
143 
144                     // Draw with filling text color
145                     graphics2D.setColor(graphicsTheme.calendarFillingDayTextColor);
146                     drawDayOfMonth(graphics2D, dayCenterX, dayCenterY, prevDay);
147                 }
148             }
149 
150             for (int day = 1; day <= lastDay; day++) {
151                 LocalDate currentDate = LocalDate.of(year, monthIndex + 1, day);
152                 int       dayOfWeek   = (firstDayOfWeek + day - 1) % 7;
153                 int       weekNum     = (firstDayOfWeek + day - 1) / 7;
154                 int       dayX        = monthStartX + 10 + (dayOfWeek * DAY_SIZE);
155                 int       dayY        = daysStartY + (weekNum * DAY_SIZE);
156                 // Calculate center position for the day cell
157                 int dayCenterX = dayX + (DAY_SIZE / 2);
158                 int dayCenterY = dayY + (DAY_SIZE / 2) - 6; // Adjusted to center the text better
159 
160                 ProjectCalendar pc        = user.getCalendar();
161                 boolean         isWeekend = !pc.isWorkingDay(currentDate.getDayOfWeek());
162                 // Check if today
163                 boolean isToday = currentDate.equals(today);
164 
165                 // Check off days
166                 Color bgColor   = null;
167                 Color textColor = graphicsTheme.calendarNormalDayTextColor;
168 
169                 if (isWeekend) {
170                     bgColor   = graphicsTheme.calendarWeekendBgColor;
171                     textColor = graphicsTheme.calendarWeekendTextColor;
172                 } else {
173                     ProjectCalendarException exception = pc.getException(currentDate);
174                     if (exception != null) {
175                         String name = exception.getName();
176                         if (name.equals(OffDayType.VACATION.name())) {
177                             bgColor   = graphicsTheme.calendarVacationBgColor;
178                             textColor = graphicsTheme.calendarVacationTextColor;
179                             vacationDays++;
180                         } else if (name.equals(OffDayType.SICK.name())) {
181                             bgColor   = graphicsTheme.calendarSickBgColor;
182                             textColor = graphicsTheme.calendarSickTextColor;
183                             sickDays++;
184                         } else if (name.equals(OffDayType.TRIP.name())) {
185                             bgColor   = graphicsTheme.calendarTripBgColor;
186                             textColor = graphicsTheme.calendarTripTextColor;
187                             tripDays++;
188                         } else {
189                             bgColor   = graphicsTheme.calendarHolidayBgColor;
190                             textColor = graphicsTheme.calendarHolidayTextColor;
191                             this.holidays++;
192                         }
193                     }
194                 }
195 
196                 if (bgColor != null) {
197                     graphics2D.setColor(bgColor);
198                     graphics2D.fillRect(dayCenterX - (DAY_SIZE / 2) + 1, dayCenterY - (DAY_SIZE / 2) + 1, DAY_SIZE - 1, DAY_SIZE - 1);
199                 }
200 
201                 graphics2D.setColor(textColor);
202                 //today can be any day, even weekend, holiday, vacation or sock day.
203                 if (isToday) {
204                     drawToday(graphics2D, dayCenterX, dayCenterY);
205                 }
206                 drawDayOfMonth(graphics2D, dayCenterX, dayCenterY, day);
207             }
208 
209             // Draw days from next month to fill the gaps at the end
210             int lastDayOfWeek = (firstDayOfWeek + lastDay - 1) % 7;
211             if (lastDayOfWeek < 6) { // If the month doesn't end on a Sunday
212                 int daysToAdd = 6 - lastDayOfWeek;
213 
214                 for (int i = 1; i <= daysToAdd; i++) {
215                     int weekNum    = (firstDayOfWeek + lastDay - 1 + i) / 7;
216                     int dayOfWeek  = (firstDayOfWeek + lastDay - 1 + i) % 7;
217                     int dayX       = monthStartX + 10 + (dayOfWeek * DAY_SIZE);
218                     int dayY       = daysStartY + (weekNum * DAY_SIZE);
219                     int dayCenterX = dayX + (DAY_SIZE / 2);
220                     int dayCenterY = dayY + (DAY_SIZE / 2) - 6;
221 
222                     // Draw with filling text color
223                     graphics2D.setColor(graphicsTheme.calendarFillingDayTextColor);
224                     drawDayOfMonth(graphics2D, dayCenterX, dayCenterY, i);
225                 }
226             }
227         }
228         int legendStartY = y + (6 * DAY_SIZE) + yearOffset + 2 * (MONTH_HEIGHT + MONTH_GAP) + HEADER_HEIGHT + WEEKDAY_HEADER_HEIGHT + 10; // Added 10px extra spacing
229         drawLegend(graphics2D, x + 15, legendStartY);
230 
231         // Restore original settings
232         graphics2D.setFont(originalFont);
233         graphics2D.setColor(originalColor);
234     }

236     private static void drawDayOfMonth(Graphics2D graphics2D, int dayCenterX, int dayCenterY, int day) {
237         // Draw day number
238         String      dayStr    = String.valueOf(day);
239         FontMetrics metrics   = graphics2D.getFontMetrics();
240         int         textWidth = metrics.stringWidth(dayStr);
241         Font        dayFont   = new Font(Font.SANS_SERIF, Font.BOLD, 10);
242         graphics2D.setFont(dayFont);
243         graphics2D.drawString(dayStr, dayCenterX - (textWidth / 2), dayCenterY + (metrics.getAscent() / 2));
244     }

246     // Helper method to draw example day numbers in legend
247     private void drawExampleDayNumber(Graphics2D graphics2D, int x, int y, int squareSize, String dayNumber) {
248         Font originalFont = graphics2D.getFont();
249         // Use the same font style as in the actual calendar
250         Font dayFont = new Font(Font.SANS_SERIF, Font.BOLD, 10);
251         graphics2D.setFont(dayFont);
252 
253         FontMetrics fm        = graphics2D.getFontMetrics();
254         int         textWidth = fm.stringWidth(dayNumber);
255         int         yShift    = fm.getAscent() - fm.getHeight() / 2;
256 
257         // Center the day number in the square
258         int centerX = x + squareSize / 2;
259         int centerY = y + squareSize / 2;
260 
261         graphics2D.drawString(dayNumber, centerX - (textWidth / 2), centerY + yShift);
262 
263         // Restore original font
264         graphics2D.setFont(originalFont);
265     }

267     private void drawLegend(Graphics2D graphics2D, int x, int y) {
268         // Save original font and color
269         Font  originalFont  = graphics2D.getFont();
270         Color originalColor = graphics2D.getColor();
271 
272         Font legendFont = new Font(Font.SANS_SERIF, Font.PLAIN, 12);
273         graphics2D.setFont(legendFont);
274 
275         int itemWidth  = 100;
276         int itemHeight = 20;
277         int squareSize = 16;
278         int gap        = 10;
279 
280         // Create legend items
281         List<LegendItem> legendItems = Arrays.asList(
282                 new LegendItem(graphicsTheme.calendarVacationBgColor, graphicsTheme.calendarVacationTextColor, "Vacation", "" + vacationDays),
283                 new LegendItem(graphicsTheme.calendarSickBgColor, graphicsTheme.calendarSickTextColor, "Sick Leave", "" + sickDays),
284                 new LegendItem(graphicsTheme.calendarHolidayBgColor, graphicsTheme.calendarHolidayTextColor, "Holiday", "" + holidays),
285                 new LegendItem(graphicsTheme.calendarTripBgColor, graphicsTheme.calendarTripTextColor, "Business Trip", "" + tripDays)
286         );
287 
288         // Draw each legend item
289         for (int i = 0; i < legendItems.size(); i++) {
290             LegendItem item  = legendItems.get(i);
291             int        itemX = x + (itemWidth * i);
292 
293             // Draw the colored box
294             graphics2D.setColor(item.bgColor);
295             graphics2D.fillRect(itemX, y, squareSize, squareSize);
296 
297             // Draw example day number
298             graphics2D.setColor(item.textColor);
299             drawExampleDayNumber(graphics2D, itemX, y, squareSize, item.exampleNumber);
300 
301             // Draw the label
302             FontMetrics fm     = graphics2D.getFontMetrics();
303             int         yShift = fm.getAscent() - fm.getHeight() / 2;
304             graphics2D.drawString(item.label, itemX + squareSize + 5, y + squareSize / 2 + yShift);
305         }
306 
307         // Restore original settings
308         graphics2D.setFont(originalFont);
309         graphics2D.setColor(originalColor);
310     }

312     private void drawMonthName(Graphics2D graphics2D, int monthStartX, int monthStartY, String monthName) {
313         // Draw month name
314         Font monthFont = new Font(Font.SANS_SERIF, Font.PLAIN, 14);

322     private void drawToday(Graphics2D graphics2D, int dayCenterX, int dayCenterY) {
323         graphics2D.setColor(graphicsTheme.calendarTodayBgColor);
324         graphics2D.fillOval(dayCenterX - (DAY_SIZE / 2) + 1, dayCenterY - (DAY_SIZE / 2) + 1, DAY_SIZE - 1, DAY_SIZE - 1);
325         graphics2D.setColor(graphicsTheme.calendarTodayTextColor);
326     }

328     private void drawWeekDays(Graphics2D graphics2D, int monthStartX, int monthStartY, LocalDate firstOfMonth) {
329         // Draw weekday headers
330         // Day of week labels
331         String[] weekdays    = {"M", "T", "W", "T", "F", "S", "S"};
332         Font     weekdayFont = new Font(Font.SANS_SERIF, Font.BOLD, 10);
333         graphics2D.setFont(weekdayFont);
334         FontMetrics weekdayMetrics = graphics2D.getFontMetrics();
335         int         weekdayAscent  = weekdayMetrics.getAscent();
336         for (int i = 0; i < 7; i++) {
337             int weekdayX = monthStartX + 10 + (i * DAY_SIZE);
338             // Center the weekday text vertically in the weekday header area
339             int weekdayY = monthStartY + HEADER_HEIGHT + (WEEKDAY_HEADER_HEIGHT / 2) + (weekdayAscent / 2);
340             // Get the calendar day type from the user's calendar
341             ProjectCalendar pc      = user.getCalendar();
342             LocalDate       dayDate = firstOfMonth.withDayOfMonth(1).plusDays(i);
343             graphics2D.setColor(graphicsTheme.calendarWeekDayTextColor);
344             graphics2D.drawString(weekdays[i], weekdayX + DAY_SIZE / 2 - 5, weekdayY);
345         }
346     }

348     private int drawYear(Graphics2D graphics2D, int year, int x, int y) throws IOException, FontFormatException {
349         Font yearFont = new Font(Font.SANS_SERIF, Font.PLAIN, 24);
350 //        Font yearFont = loadFont("Inter[opsz,wght].ttf", 24f);
351         graphics2D.setFont(yearFont);
352         graphics2D.setColor(graphicsTheme.calendarYearTextColor);
353         String yearText = String.valueOf(year);
354 
355         // Calculate year text dimensions
356         FontMetrics yearMetrics    = graphics2D.getFontMetrics(yearFont);
357         int         yearTextHeight = yearMetrics.getHeight();
358         int         yearTextWidth  = yearMetrics.stringWidth(yearText);
359 
360         // Center the year text
361         graphics2D.drawString(yearText, x + 5, y + yearMetrics.getAscent());
362         return yearTextHeight;
363     }

365     Font loadFont(String fontName, float size) {
366         try (FileInputStream fontStream = new FileInputStream("resources/fonts/ofl/inter/" + fontName)) {
367             Font font = Font.createFont(Font.TRUETYPE_FONT, fontStream);
368             // Use FontVariation API for Java 17+
369             Map map = Map.of(TextAttribute.WEIGHT, TextAttribute.WEIGHT_LIGHT);
370             return font.deriveFont(map).deriveFont(size);
371 
372         } catch (FontFormatException | IOException e) {
373             logger.error(e.getMessage(), e);
374             return new Font(Font.SANS_SERIF, Font.PLAIN, 24);
375         }
376     }
