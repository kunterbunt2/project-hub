44  @Getter
45  @Setter
46  @NoArgsConstructor
47  @ToString(callSuper = true)
48  @EqualsAndHashCode(of = {"id"}, callSuper = false)

50  public class User extends AbstractTimeAware implements Comparable<User> {

52      private static final long               MAX_PROJECT_LENGTH = 1;
53      @JsonManagedReference
54      private              List<Availability> availabilities     = new ArrayList<>();
55  
56      @JsonIgnore
57      private ProjectCalendar calendar;
58      private Color           color;

69      public void addAvailability(Availability availability) {
70          availabilities.add(availability);
71      }
72  
73      public void addLocation(Location location) {
74          if (location.getStart() == null)
75              throw new IllegalArgumentException("start date is null");
76          if (location.getCountry() == null)
77              throw new IllegalArgumentException("start date is null");
78          if (location.getState() == null)
79              throw new IllegalArgumentException("start date is null");
80          locations.add(location);
81      }
82  
83      public void addOffday(OffDay offDay) {
84          offDays.add(offDay);
85      }

88      public int compareTo(User other) {
89          return this.id.compareTo(other.id);
90      }

92      @JsonIgnore
93      public String getKey() {
94          return "U-" + id;
95      }

97      public void initialize(GanttContext gc) {
98          ProjectCalendar resourceCalendar = getCalendar();
99          if (resourceCalendar == null) {
100             resourceCalendar = gc.getProjectFile().addDefaultDerivedCalendar();
101             resourceCalendar.setParent(gc.getCalendar());
102             resourceCalendar.setName(getName());
103             setCalendar(resourceCalendar);
104         }
105 
106         initializeLocationsAndOffdays();
107     }
108 
109     public void initialize(Sprint sprint) {
110         ProjectCalendar resourceCalendar = getCalendar();
111         if (resourceCalendar == null) {
112             resourceCalendar = sprint.getProjectFile().addDefaultDerivedCalendar();
113             resourceCalendar.setParent(sprint.getCalendar());
114             resourceCalendar.setName(getName());
115             setCalendar(resourceCalendar);
116         }
117 
118         initializeLocationsAndOffdays();
119     }
120 
121     public void initialize() {
122         long            time             = System.currentTimeMillis();
123         ProjectCalendar resourceCalendar = getCalendar();
124         if (resourceCalendar == null) {
125             ProjectFile projectFile = new ProjectFile();
126             CalendarUtil.initializeProjectProperties(projectFile);
127             ProjectCalendar calendar = CalendarUtil.initializeCalendar(projectFile);
128             resourceCalendar = projectFile.addDefaultDerivedCalendar();
129             resourceCalendar.setParent(calendar);
130             resourceCalendar.setName(getName());
131             setCalendar(resourceCalendar);
132         }
133         System.out.println("User.initialize() took " + (System.currentTimeMillis() - time) + " ms for user: " + getName());
134 
135         time = System.currentTimeMillis();
136         initializeLocationsAndOffdays();
137         System.out.println("User.initializeLocationsAndOffdays() took " + (System.currentTimeMillis() - time) + " ms for user: " + getName());
138     }
139 
140     private void initializeLocationsAndOffdays() {
141         //TODO rethink employee leaving company and coming back
142         ProjectCalendar pc = getCalendar();
143         LocalDate       endDateInclusive;
144         for (OffDay offDay : getOffDays()) {
145             ProjectCalendarException pce = pc.addCalendarException(offDay.getFirstDay(), offDay.getLastDay());
146             pce.setName(offDay.getType().name());
147         }
148         for (int i = 0; i < locations.size(); i++) {
149             Location  location           = locations.get(i);
150             LocalDate startDateInclusive = location.getStart();
151             if (i + 1 < locations.size())
152                 endDateInclusive = locations.get(i + 1).getStart();//end of this location is start of next location
153             else
154                 endDateInclusive = ParameterOptions.now.plusYears(MAX_PROJECT_LENGTH).toLocalDate();
155             HolidayManager holidayManager = HolidayManager.getInstance(ManagerParameters.create(location.getCountry()));
156             List<Holiday>  holidays       = holidayManager.getHolidays(startDateInclusive, endDateInclusive, location.getState()).stream().sorted().collect(Collectors.toList());
157             URL            url            = getClass().getClassLoader().getResource("holidays/carnival-holidays.xml");
158             if (url != null && "nw".equals(location.getState())) {
159                 UrlManagerParameter urlManParam        = new UrlManagerParameter(url, new Properties());
160                 HolidayManager      customManager      = HolidayManager.getInstance(urlManParam);
161                 Set<Holiday>        additionalHolidays = customManager.getHolidays(startDateInclusive, endDateInclusive, location.getState());
162                 holidays.addAll(additionalHolidays.stream().toList());
163             }
164             for (Holiday holiday : holidays) {
165                 ProjectCalendarException pce = pc.addCalendarException(holiday.getDate());
166                 pce.setName(holiday.getDescription());
167             }
168         }
169     }
