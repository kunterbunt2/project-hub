18  package de.bushnaq.abdalla.projecthub.ui.dialog;
19  
20  import com.vaadin.flow.component.button.Button;
21  import com.vaadin.flow.component.button.ButtonVariant;
22  import com.vaadin.flow.component.dialog.Dialog;
23  import com.vaadin.flow.component.orderedlayout.FlexComponent;
24  import com.vaadin.flow.component.orderedlayout.HorizontalLayout;
25  import com.vaadin.flow.component.orderedlayout.VerticalLayout;
26  
27  /**
28   * A reusable dialog for confirming delete operations.
29   */
30  public class ConfirmDialog extends Dialog {
31  
32      public static final String CANCEL_BUTTON  = "cancel-confirm-button";
33      public static final String CONFIRM_BUTTON = "confirm-button";
34      public static final String CONFIRM_DIALOG = "confirm-dialog";
35  
36      /**
37       * Creates a confirmation dialog for delete operations.
38       *
39       * @param title             The dialog title
40       * @param message           The confirmation message to display
41       * @param confirmButtonText Text for the delete/confirm button
42       * @param action            Runnable to execute when confirmed
43       */
44      public ConfirmDialog(String title, String message, String confirmButtonText, Runnable action) {
45          setHeaderTitle(title);
46          setId(CONFIRM_DIALOG);
47          setWidth("480px");
48  
49          VerticalLayout dialogLayout = new VerticalLayout();
50          dialogLayout.add(message);
51          dialogLayout.setPadding(true);
52  
53          Button confirmButton = new Button(confirmButtonText, e -> {
54              action.run();
55              close();
56          });
57          confirmButton.setId(CONFIRM_BUTTON);
58          confirmButton.addThemeVariants(ButtonVariant.LUMO_PRIMARY, ButtonVariant.LUMO_ERROR);
59  
60          Button cancelButton = new Button("Cancel", e -> close());
61          cancelButton.setId(CANCEL_BUTTON);
62  
63          HorizontalLayout buttonLayout = new HorizontalLayout();
64          buttonLayout.setJustifyContentMode(FlexComponent.JustifyContentMode.END);
65          buttonLayout.add(cancelButton, confirmButton);
66          buttonLayout.setWidthFull();
67  
68          dialogLayout.add(buttonLayout);
69          add(dialogLayout);
70      }
71  }
