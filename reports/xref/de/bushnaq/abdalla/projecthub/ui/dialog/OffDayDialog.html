32  import com.vaadin.flow.data.binder.ValidationException;

34  import de.bushnaq.abdalla.projecthub.dto.OffDayType;

41  import java.time.LocalDate;

43  public class OffDayDialog extends Dialog {
44      public static final String               CANCEL_BUTTON           = "offday-dialog-cancel";
45      public static final String               CONFIRM_BUTTON          = "offday-dialog-confirm";
46      public static final String               OFFDAY_DIALOG           = "offday-dialog";
47      public static final String               OFFDAY_END_DATE_FIELD   = "offday-end-date-field";
48      public static final String               OFFDAY_START_DATE_FIELD = "offday-start-date-field";
49      public static final String               OFFDAY_TYPE_FIELD       = "offday-type-field";
50      private final       Binder<OffDay>       binder                  = new Binder<>(OffDay.class);
51      private final       DatePicker           firstDayField           = new DatePicker("First Day");
52      private final       boolean              isNewOffDay;
53      private final       DatePicker           lastDayField            = new DatePicker("Last Day");
54      private final       OffDay               offDay;
55      private final       OffDayApi            offDayApi;
56      private final       Runnable             onSaveCallback;
57      private final       ComboBox<OffDayType> typeField               = new ComboBox<>("Type");
58      private final       User                 user;
59  
60      public OffDayDialog(OffDay offDay, User user, OffDayApi offDayApi, Runnable onSaveCallback) {
61          this.offDayApi      = offDayApi;
62          this.user           = user;
63          this.onSaveCallback = onSaveCallback;
64          this.isNewOffDay    = (offDay == null);
65          this.offDay         = isNewOffDay ? new OffDay(LocalDate.now(), LocalDate.now(), OffDayType.VACATION) : offDay;
66  
67          // Set the dialog title with an icon
68          String title = offDay == null ? "Create Off Day" : "Edit Off Day";
69  
70          // Create a custom header with icon
71          HorizontalLayout headerLayout = new HorizontalLayout();
72          headerLayout.setAlignItems(FlexComponent.Alignment.CENTER);
73          headerLayout.setSpacing(true);
74  
75          Icon titleIcon = new Icon(VaadinIcon.CALENDAR);
76          titleIcon.getStyle().set("margin-right", "0.5em");
77  
78          H3 titleLabel = new H3(title);
79          titleLabel.getStyle().set("margin", "0");
80  
81          headerLayout.add(titleIcon, titleLabel);
82  
83          // Set the custom header
84          setHeaderTitle(null); // Clear the default title
85          getHeader().add(headerLayout);
86  
87          setId(OFFDAY_DIALOG);
88          setWidth("480px");
89  //        setCloseOnEsc(true);
90  //        setCloseOnOutsideClick(false);
91  //        setDraggable(true);
92  //        setResizable(true);
93  
94          // Setup form and actions
95          add(createHeader(), createForm(), VaadinUtils.createDialogButtonLayout("Save", CONFIRM_BUTTON, "Cancel", CANCEL_BUTTON, this::save, this));
96          configureFormBinder();
97      }
98  
99      private void configureFormBinder() {
100         // First day binding
101         binder.forField(firstDayField)
102                 .asRequired("First day is required")
103                 .withValidator(date -> date == null || lastDayField.getValue() == null ||
104                                 !date.isAfter(lastDayField.getValue()),
105                         "First day must be before or equal to last day")
106                 .bind(OffDay::getFirstDay, OffDay::setFirstDay);
107 
108         // Last day binding
109         binder.forField(lastDayField)
110                 .asRequired("Last day is required")
111                 .withValidator(date -> date == null || firstDayField.getValue() == null ||
112                                 !date.isBefore(firstDayField.getValue()),
113                         "Last day must be after or equal to first day")
114                 .bind(OffDay::getLastDay, OffDay::setLastDay);
115 
116         // Type binding
117         binder.forField(typeField)
118                 .asRequired("Type is required")
119                 .bind(OffDay::getType, OffDay::setType);
120 
121         // Load data into the form
122         if (!isNewOffDay) {
123             binder.readBean(offDay);
124         }
125     }
126 
127     private FormLayout createForm() {
128         FormLayout formLayout = new FormLayout();
129 
130         // Configure fields
131         firstDayField.setWidthFull();
132         firstDayField.setHelperText("The first day of the off period");
133         firstDayField.setId(OFFDAY_START_DATE_FIELD);
134         firstDayField.setPrefixComponent(new Icon(VaadinIcon.CALENDAR));
135 
136         lastDayField.setWidthFull();
137         lastDayField.setHelperText("The last day of the off period");
138         lastDayField.setId(OFFDAY_END_DATE_FIELD);
139         lastDayField.setPrefixComponent(new Icon(VaadinIcon.CALENDAR));
140 
141         typeField.setWidthFull();
142         typeField.setHelperText("Type of off day");
143         typeField.setItems(OffDayType.values());
144         typeField.setId(OFFDAY_TYPE_FIELD);
145         typeField.setPrefixComponent(new Icon(VaadinIcon.TAGS));
146 
147         // Add validation for date range relationship
148         firstDayField.addValueChangeListener(e -> lastDayField.setMin(e.getValue()));
149         lastDayField.addValueChangeListener(e -> firstDayField.setMax(e.getValue()));
150 
151         formLayout.add(typeField, firstDayField, lastDayField);
152         return formLayout;
153     }
154 
155     private H3 createHeader() {
156         String title  = isNewOffDay ? "Add New Off Day" : "Edit Off Day";
157         H3     header = new H3(title);
158         header.getStyle().set("margin-top", "0");
159         return header;
160     }
161 
162     private void save() {
163         try {
164             // Validate all fields first - this will trigger validation and show error messages
165             if (!binder.validate().isOk()) {
166                 return; // Stop here if validation fails
167             }
168 
169             // Write values to bean
170             binder.writeBean(offDay);
171 
172             // Ensure user association is set
173             offDay.setUser(user);
174 
175             // Save to backend
176             if (isNewOffDay) {
177                 offDayApi.persist(offDay, user.getId());
178             } else {
179                 offDayApi.update(offDay, user.getId());
180             }
181             // Notify success
182             String message = isNewOffDay ? "New off day record added" : "Off day record updated";
183             Notification.show(message, 3000, Notification.Position.MIDDLE);
184 
185             // Trigger refresh callback and close dialog
186             onSaveCallback.run();
187             close();
188         } catch (ValidationException e) {
189             System.out.printf("Validation failed: %s%n", e.getMessage());
190             // Validation errors will already be displayed next to the fields
191         } catch (Exception e) {
192             System.out.printf("Validation failed: %s%n", e.getMessage());
193             if (e instanceof ResponseStatusException && ((ResponseStatusException) e).getStatusCode().equals(HttpStatus.CONFLICT)) {
194                 String errorMessage = ((ResponseStatusException) e).getReason();
195                 firstDayField.setInvalid(errorMessage != null);
196                 firstDayField.setErrorMessage(errorMessage);
197                 lastDayField.setInvalid(errorMessage != null);
198                 lastDayField.setErrorMessage(errorMessage);
199                 // Keep the dialog open so the user can correct the name
200             } else {
201                 // For other errors, show generic message and close dialog
202                 Notification notification = Notification.show("Error saving off day: " + e.getMessage(), 3000, Notification.Position.MIDDLE);
203                 notification.addThemeVariants(NotificationVariant.LUMO_ERROR);
204                 notification.open();
205                 // dialogReference.close();
206                 // Keep the dialog open so the user can correct the name
207             }
208             // Only show notification for unexpected errors
209         }
210     }
