18  package de.bushnaq.abdalla.projecthub.ui.dialog;
19  
20  import com.vaadin.flow.component.datepicker.DatePicker;
21  import com.vaadin.flow.component.dialog.Dialog;
22  import com.vaadin.flow.component.icon.Icon;
23  import com.vaadin.flow.component.icon.VaadinIcon;
24  import com.vaadin.flow.component.notification.Notification;
25  import com.vaadin.flow.component.orderedlayout.FlexComponent;
26  import com.vaadin.flow.component.orderedlayout.HorizontalLayout;
27  import com.vaadin.flow.component.orderedlayout.VerticalLayout;
28  import com.vaadin.flow.component.textfield.EmailField;
29  import com.vaadin.flow.component.textfield.TextField;
30  import de.bushnaq.abdalla.projecthub.dto.User;
31  import de.bushnaq.abdalla.projecthub.ui.util.VaadinUtils;
32  import org.vaadin.addons.tatu.ColorPicker;
33  
34  import java.awt.*;
35  import java.util.List;
36  import java.util.function.Consumer;
37  
38  /**
39   * A reusable dialog for creating and editing users.
40   */
41  public class UserDialog extends Dialog {
42  
43      public static final String         CANCEL_BUTTON                 = "cancel-user-button";
44      public static final String         CONFIRM_BUTTON                = "save-user-button";
45      public static final String         USER_COLOR_PICKER             = "user-color-picker";
46      public static final String         USER_DIALOG                   = "user-dialog";
47      public static final String         USER_EMAIL_FIELD              = "user-email-field";
48      public static final String         USER_FIRST_WORKING_DAY_PICKER = "user-first-working-day-picker";
49      public static final String         USER_LAST_WORKING_DAY_PICKER  = "user-last-working-day-picker";
50      public static final String         USER_NAME_FIELD               = "user-name-field";
51      private final       ColorPicker    colorPicker;
52      private final       EmailField     emailField;
53      private final       DatePicker     firstWorkingDayPicker;
54      private final       boolean        isEditMode;
55      private final       DatePicker     lastWorkingDayPicker;
56      private final       TextField      nameField;
57      private final       Consumer<User> saveCallback;
58      private final       User           user;
59  
60      /**
61       * Creates a dialog for creating or editing a user.
62       *
63       * @param user         The user to edit, or null for creating a new user
64       * @param saveCallback Callback that receives the user with updated values
65       */
66      public UserDialog(User user, Consumer<User> saveCallback) {
67          this.user         = user;
68          this.saveCallback = saveCallback;
69          isEditMode        = user != null;
70  
71          // Set the dialog title with an icon
72          String title = isEditMode ? "Edit User" : "Create User";
73  
74          // Create a custom header with icon
75          HorizontalLayout headerLayout = new HorizontalLayout();
76          headerLayout.setAlignItems(FlexComponent.Alignment.CENTER);
77          headerLayout.setSpacing(true);
78  
79          Icon titleIcon = new Icon(VaadinIcon.USER);
80          titleIcon.getStyle().set("margin-right", "0.5em");
81  
82          com.vaadin.flow.component.html.H3 titleLabel = new com.vaadin.flow.component.html.H3(title);
83          titleLabel.getStyle().set("margin", "0");
84  
85          headerLayout.add(titleIcon, titleLabel);
86  
87          // Set the custom header
88          setHeaderTitle(null); // Clear the default title
89          getHeader().add(headerLayout);
90  
91          setId(USER_DIALOG);
92          setWidth("500px");
93  
94          VerticalLayout dialogLayout = new VerticalLayout();
95          dialogLayout.setPadding(false);
96          dialogLayout.setSpacing(true);
97  
98          // Name field
99          nameField = new TextField("Name");
100         nameField.setId(USER_NAME_FIELD);
101         nameField.setWidthFull();
102         nameField.setRequired(true);
103         nameField.setPrefixComponent(new Icon(VaadinIcon.USER));
104 
105         // Email field
106         emailField = new EmailField("Email");
107         emailField.setId(USER_EMAIL_FIELD);
108         emailField.setWidthFull();
109         emailField.setPrefixComponent(new Icon(VaadinIcon.ENVELOPE));
110 
111         // Color picker (using the community add-on)
112         colorPicker = new ColorPicker();
113         colorPicker.setId(USER_COLOR_PICKER);
114         colorPicker.setWidth("100%");
115 
116         // Set predefined color presets
117         colorPicker.setPresets(List.of(
118                 new ColorPicker.ColorPreset("#FF0000", "Red"),
119                 new ColorPicker.ColorPreset("#0000FF", "Blue"),
120                 new ColorPicker.ColorPreset("#008000", "Green"),
121                 new ColorPicker.ColorPreset("#FFFF00", "Yellow"),
122                 new ColorPicker.ColorPreset("#FFA500", "Orange"),
123                 new ColorPicker.ColorPreset("#800080", "Purple"),
124                 new ColorPicker.ColorPreset("#FFC0CB", "Pink"),
125                 new ColorPicker.ColorPreset("#00FFFF", "Cyan"),
126                 new ColorPicker.ColorPreset("#FF00FF", "Magenta"),
127                 new ColorPicker.ColorPreset("#D3D3D3", "Light Gray"),
128                 new ColorPicker.ColorPreset("#808080", "Gray"),
129                 new ColorPicker.ColorPreset("#A9A9A9", "Dark Gray"),
130                 new ColorPicker.ColorPreset("#000000", "Black")
131         ));
132 
133         // First working day picker
134         firstWorkingDayPicker = new DatePicker("First Working Day");
135         firstWorkingDayPicker.setId(USER_FIRST_WORKING_DAY_PICKER);
136         firstWorkingDayPicker.setWidthFull();
137         firstWorkingDayPicker.setPrefixComponent(new Icon(VaadinIcon.CALENDAR_USER));
138 
139         // Last working day picker
140         lastWorkingDayPicker = new DatePicker("Last Working Day");
141         lastWorkingDayPicker.setId(USER_LAST_WORKING_DAY_PICKER);
142         lastWorkingDayPicker.setWidthFull();
143         lastWorkingDayPicker.setPrefixComponent(new Icon(VaadinIcon.CALENDAR_USER));
144 
145         // Add validation to ensure last working day is after first working day
146         firstWorkingDayPicker.addValueChangeListener(event -> {
147             if (event.getValue() != null) {
148                 lastWorkingDayPicker.setMin(event.getValue());
149             }
150         });
151 
152         lastWorkingDayPicker.addValueChangeListener(event -> {
153             if (event.getValue() != null) {
154                 firstWorkingDayPicker.setMax(event.getValue());
155             }
156         });
157 
158         if (isEditMode) {
159             nameField.setValue(user.getName() != null ? user.getName() : "");
160             emailField.setValue(user.getEmail() != null ? user.getEmail() : "");
161             if (user.getColor() != null) {
162                 String colorHex = "#" + Integer.toHexString(user.getColor().getRGB()).substring(2);
163                 colorPicker.setValue(colorHex);
164             }
165             firstWorkingDayPicker.setValue(user.getFirstWorkingDay());
166             lastWorkingDayPicker.setValue(user.getLastWorkingDay());
167         }
168 
169         dialogLayout.add(
170                 nameField,
171                 emailField,
172                 colorPicker,
173                 firstWorkingDayPicker,
174                 lastWorkingDayPicker
175         );
176 
177         dialogLayout.add(VaadinUtils.createDialogButtonLayout("Save", CONFIRM_BUTTON, "Cancel", CANCEL_BUTTON, this::save, this));
178         add(dialogLayout);
179     }
180 
181     private void save() {
182         if (nameField.getValue().trim().isEmpty()) {
183             Notification.show("Please enter a user name", 3000, Notification.Position.MIDDLE);
184             return;
185         }
186 
187         User userToSave;
188         if (isEditMode) {
189             userToSave = user;
190         } else {
191             userToSave = new User();
192         }
193 
194         userToSave.setName(nameField.getValue().trim());
195         userToSave.setEmail(emailField.getValue().trim());
196 
197         // Convert Vaadin color string to AWT Color
198         String colorValue = colorPicker.getValue();
199         if (colorValue != null && !colorValue.isEmpty()) {
200             userToSave.setColor(Color.decode(colorValue));
201         }
202 
203         userToSave.setFirstWorkingDay(firstWorkingDayPicker.getValue());
204         userToSave.setLastWorkingDay(lastWorkingDayPicker.getValue());
205 
206         saveCallback.accept(userToSave);
207         close();
208 
209     }
210 }
