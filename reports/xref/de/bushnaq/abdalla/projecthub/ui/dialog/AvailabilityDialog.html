42  public class AvailabilityDialog extends Dialog {
43      public static final String               AVAILABILITY_DIALOG           = "availability-dialog";
44      public static final String               AVAILABILITY_PERCENTAGE_FIELD = "availability-percentage-field";
45      public static final String               AVAILABILITY_START_DATE_FIELD = "availability-start-date-field";
46      public static final String               CANCEL_BUTTON                 = "availability-dialog-cancel";
47      public static final String               CONFIRM_BUTTON                = "availability-dialog-confirm";
48      private final       Availability         availability;
49      private final       AvailabilityApi      availabilityApi;
50      private final       IntegerField         availabilityField             = new IntegerField("Availability (%)");
51      private final       Binder<Availability> binder                        = new Binder<>(Availability.class);
52      private final       boolean              isNewAvailability;
53      private final       Runnable             onSaveCallback;
54      private final       DatePicker           startDateField                = new DatePicker("Start Date");
55      private final       User                 user;
56  
57      public AvailabilityDialog(Availability availability, User user, AvailabilityApi availabilityApi, Runnable onSaveCallback) {
58          this.availabilityApi   = availabilityApi;
59          this.user              = user;
60          this.onSaveCallback    = onSaveCallback;
61          this.isNewAvailability = (availability == null);
62          this.availability      = isNewAvailability ? new Availability() : availability;
63  
64          // Set the dialog title with an icon
65          String title = availability == null ? "Create Availability" : "Edit Availability";
66  
67          // Create a custom header with icon
68          HorizontalLayout headerLayout = new HorizontalLayout();
69          headerLayout.setAlignItems(FlexComponent.Alignment.CENTER);
70          headerLayout.setSpacing(true);
71  
72          Icon titleIcon = new Icon(VaadinIcon.CHART);
73          titleIcon.getStyle().set("margin-right", "0.5em");
74  
75          H3 titleLabel = new H3(title);
76          titleLabel.getStyle().set("margin", "0");
77  
78          headerLayout.add(titleIcon, titleLabel);
79  
80          // Set the custom header
81          setHeaderTitle(null); // Clear the default title
82          getHeader().add(headerLayout);
83  
84          setId(AVAILABILITY_DIALOG);
85          setWidth("480px");
86  //        setCloseOnEsc(true);
87  //        setCloseOnOutsideClick(false);
88  //        setDraggable(true);
89  //        setResizable(true);
90  
91          // Setup form and actions
92          add(createHeader(), createForm(), VaadinUtils.createDialogButtonLayout("Save", CONFIRM_BUTTON, "Cancel", CANCEL_BUTTON, this::save, this));
93          configureFormBinder();
94      }
95  
96      private void configureFormBinder() {
97          // Start date binding with custom validator to ensure uniqueness
98          binder.forField(startDateField)
99                  .asRequired("Start date is required")
100                 .withValidator(this::validateStartDateUniqueness, "Another availability record already exists with this date")
101                 .withValidationStatusHandler(status -> {
102                     startDateField.setInvalid(status.getMessage().isPresent());
103                     startDateField.setErrorMessage(status.getMessage().orElse(""));
104                 })
105                 .bind(Availability::getStart, Availability::setStart);
106 
107         // Availability binding (as percentage, converting between 0-150% display and 0-1.5 stored value)
108         binder.forField(availabilityField)
109                 .asRequired("Availability is required")
110                 .withConverter(
111                         value -> value != null ? value.floatValue() / 100 : 0.0f,  // Double to float and convert from percentage
112                         value -> value != null ? (int) (value * 100) : 0  // float to Double and convert to percentage
113                 )
114                 .withValidator(new RangeValidator<>("Availability must be between 0% and 150%", Float::compare, 0.0f, 1.5f))
115                 .withValidationStatusHandler(status -> {
116                     availabilityField.setInvalid(status.getMessage().isPresent());
117                     availabilityField.setErrorMessage(status.getMessage().orElse(""));
118                 })
119                 .bind(Availability::getAvailability, Availability::setAvailability);
120 
121         // Load data into the form
122         if (!isNewAvailability) {
123             binder.readBean(availability);
124             // If editing, convert the stored value (0-1.5) to percentage display (0-150)
125             availabilityField.setValue((int) (availability.getAvailability() * 100));
126         }
127     }
128 
129     private FormLayout createForm() {
130         FormLayout formLayout = new FormLayout();
131 
132         // Configure fields
133         startDateField.setWidthFull();
134         startDateField.setHelperText("The date when this availability level begins");
135         startDateField.setId(AVAILABILITY_START_DATE_FIELD);
136         startDateField.setPrefixComponent(new Icon(VaadinIcon.CALENDAR));
137 
138         availabilityField.setWidthFull();
139         availabilityField.setHelperText("Value between 0-150% (e.g., 100 = full time)");
140         availabilityField.setSuffixComponent(new Span("%"));
141         availabilityField.setPrefixComponent(new Icon(VaadinIcon.CHART));
142         availabilityField.setId(AVAILABILITY_PERCENTAGE_FIELD);
143 
144         formLayout.add(startDateField, availabilityField);
145         return formLayout;
146     }
147 
148     private H3 createHeader() {
149         String title  = isNewAvailability ? "Add New Availability" : "Edit Availability";
150         H3     header = new H3(title);
151         header.getStyle().set("margin-top", "0");
152         return header;
153     }
154 
155     private void save() {
156         try {
157             // Validate all fields first - this will trigger validation and show error messages
158             if (!binder.validate().isOk()) {
159                 return; // Stop here if validation fails
160             }
161 
162             // Write values to bean
163             binder.writeBean(availability);
164 
165             // Ensure user association is set
166             availability.setUser(user);
167 
168             // Save to backend
169             if (isNewAvailability) {
170                 availabilityApi.persist(availability, user.getId());
171             } else {
172                 availabilityApi.update(availability, user.getId());
173             }
174 
175             // Notify success
176             String message = isNewAvailability ? "New availability record added" : "Availability record updated";
177             Notification.show(message, 3000, Notification.Position.MIDDLE);
178 
179             // Trigger refresh callback and close dialog
180             onSaveCallback.run();
181             close();
182         } catch (ValidationException e) {
183             System.out.printf("Validation failed: %s%n", e.getMessage());
184             // Validation errors will already be displayed next to the fields
185             // No need for additional notification
186         } catch (Exception e) {
187             System.out.printf("Validation failed: %s%n", e.getMessage());
188             // Only show notification for unexpected errors
189             Notification notification = Notification.show("Error saving availability: " + e.getMessage(), 3000, Notification.Position.MIDDLE);
190             notification.addThemeVariants(NotificationVariant.LUMO_ERROR);
191         }
192     }
193 
194     private boolean validateStartDateUniqueness(LocalDate date) {
195         if (date == null) return true; // Let the required validator handle null values
196 
197         // If editing existing record and date hasn't changed, it's valid
198         if (!isNewAvailability && availability.getStart().equals(date)) {
199             return true;
200         }
201 
202         // Check if any other availability record exists with the same date
203         return user.getAvailabilities().stream().noneMatch(a -> date.equals(a.getStart()) && !a.equals(availability));
204     }
