51  @Layout
52  @PermitAll // When security is enabled, allow all authenticated users
53  public final class MainLayout extends AppLayout {
54  
55      private final Breadcrumbs      breadcrumbs  = new Breadcrumbs();
56      private final Map<Tab, String> tabToPathMap = new HashMap<>();
57      private final Tabs             tabs         = new Tabs();
58  
59      MainLayout() {
60          setPrimarySection(Section.NAVBAR);
61  
62          // Create main navigation bar components
63          HorizontalLayout navbarLayout = new HorizontalLayout();
64          navbarLayout.setWidthFull();
65          navbarLayout.setJustifyContentMode(FlexComponent.JustifyContentMode.BETWEEN);
66          navbarLayout.setAlignItems(FlexComponent.Alignment.CENTER);
67          navbarLayout.addClassName("navbar-custom");
68  
69          // Add logo and app name to the left
70          Image logoLayout = createLogo();
71  
72          // Add navigation tabs to the center
73          createNavTabs();
74          tabs.addClassNames(Margin.Horizontal.MEDIUM);
75          ThemeToggle themeToggle = new ThemeToggle();
76  
77          // Add user menu to the right
78          Component userMenu = createUserMenu();
79  
80          navbarLayout.add(logoLayout, tabs, themeToggle, userMenu);
81          navbarLayout.expand(tabs);
82  
83          Div breadcrumbContainer = new Div(breadcrumbs);
84          breadcrumbContainer.addClassNames(
85                  Padding.Horizontal.MEDIUM,
86                  Padding.Vertical.XSMALL,
87                  Width.FULL,
88                  Background.CONTRAST_5
89          );
90  
91          var navAndBreadcrumbs = new VerticalLayout();
92          navAndBreadcrumbs.setPadding(false);
93          navAndBreadcrumbs.setSpacing(false);
94          navAndBreadcrumbs.setMargin(false);
95  
96          navAndBreadcrumbs.add(breadcrumbContainer, navbarLayout);
97  
98          // Add the combined layout to the navbar area
99          addToNavbar(true, navAndBreadcrumbs);
100     }
101 
102     private Image createLogo() {
103         Image logo = new Image("images/logo.svg", "Kassandra Logo");
104         logo.setHeight("32px");
105         return logo;
106     }
107 
108     private void createNavTabs() {
109         tabs.setOrientation(Tabs.Orientation.HORIZONTAL);
110 
111         MenuConfiguration.getMenuEntries().forEach(entry -> {
112             Tab tab = createTab(entry);
113             tabToPathMap.put(tab, entry.path());
114             tabs.add(tab);
115         });
116 
117         tabs.addSelectedChangeListener(event -> {
118             Tab    selectedTab = event.getSelectedTab();
119             String targetPath  = tabToPathMap.get(selectedTab);
120             if (targetPath != null && !targetPath.isEmpty()) {
121                 getUI().ifPresent(ui -> ui.navigate(targetPath));
122             }
123         });
124     }
125 
126     private Tab createTab(MenuEntry menuEntry) {
127         Icon icon = null;
128         if (menuEntry.icon() != null) {
129             icon = new Icon(menuEntry.icon());
130             icon.addClassNames(Margin.Right.XSMALL);
131         }
132 
133         Span text = new Span(menuEntry.title());
134 
135         if (icon != null) {
136             return new Tab(new HorizontalLayout(icon, text));
137         } else {
138             return new Tab(text);
139         }
140     }
141 
142     private Component createUserMenu() {
143         final String username = getUserName();
144 
145         var avatar = new Avatar(username);
146         avatar.addThemeVariants(AvatarVariant.LUMO_XSMALL);
147         avatar.addClassNames(Margin.Right.SMALL);
148         avatar.setColorIndex(5);
149 
150         var userMenu = new MenuBar();
151         userMenu.addThemeVariants(MenuBarVariant.LUMO_TERTIARY_INLINE);
152         userMenu.addClassNames(Margin.Right.MEDIUM);
153 
154         var userMenuItem = userMenu.addItem(avatar);
155         userMenuItem.add(username);
156         userMenuItem.getSubMenu().addItem("Manage Availability", e -> navigateToAvailability(username));
157         userMenuItem.getSubMenu().addItem("Manage Location", e -> navigateToLocation(username));
158         userMenuItem.getSubMenu().addItem("Manage Off Days", e -> navigateToOffDays(username));
159         userMenuItem.getSubMenu().addItem("View Profile").setEnabled(false);
160         userMenuItem.getSubMenu().addItem("Manage Settings").setEnabled(false);
161         userMenuItem.getSubMenu().addItem("Logout", e -> logout());
162 
163         return userMenu;
164     }
165 
166     // Method to get the breadcrumbs component (to be used by views)
167     public Breadcrumbs getBreadcrumbs() {
168         return breadcrumbs;
169     }
170 
171     private String getUserName() {
172         Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
173         String         userName       = authentication != null ? authentication.getName() : "Guest";
174 
175         // If using OIDC, try to get the email address from authentication details
176         if (authentication != null && authentication.getPrincipal() instanceof org.springframework.security.oauth2.core.oidc.user.OidcUser oidcUser) {
177             String email = oidcUser.getEmail();
178             if (email != null && !email.isEmpty()) {
179                 userName = email;
180             }
181         }
182         return userName;
183     }
184 
185     private void logout() {
186         SecurityContextLogoutHandler logoutHandler = new SecurityContextLogoutHandler();
187         logoutHandler.logout(SecurityUtils.getHttpServletRequest(), SecurityUtils.getHttpServletResponse(), SecurityContextHolder.getContext().getAuthentication());
188         getUI().ifPresent(ui -> ui.getPage().setLocation("/ui/login"));
189     }
190 
191     private void navigateToAvailability(String username) {
192         getUI().ifPresent(ui -> ui.navigate("availability/" + username));
193     }
194 
195     private void navigateToLocation(String username) {
196         getUI().ifPresent(ui -> ui.navigate("location/" + username));
197     }
198 
199     private void navigateToOffDays(String username) {
200         getUI().ifPresent(ui -> ui.navigate("offday/" + username));
201     }
202 }
