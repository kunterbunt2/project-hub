18  package de.bushnaq.abdalla.projecthub.ui.view;
19  
20  import com.vaadin.flow.component.html.*;
21  import com.vaadin.flow.component.login.LoginForm;
22  import com.vaadin.flow.component.orderedlayout.VerticalLayout;
23  import com.vaadin.flow.router.BeforeEnterEvent;
24  import com.vaadin.flow.router.BeforeEnterObserver;
25  import com.vaadin.flow.router.PageTitle;
26  import com.vaadin.flow.router.Route;
27  import com.vaadin.flow.server.auth.AnonymousAllowed;
28  import com.vaadin.flow.theme.lumo.LumoUtility.Margin;
29  import org.springframework.core.env.Environment;
30  
31  
32  /**
33   * Login view that supports both Basic Authentication and OIDC authentication.
34   */
35  @Route("login")
36  @PageTitle("Login | Project Hub")
37  @AnonymousAllowed
38  public class LoginView extends VerticalLayout implements BeforeEnterObserver {
39      public static final String    LOGIN_VIEW        = "login-view";
40      // ID for OIDC login button used in tests
41      public static final String    OIDC_LOGIN_BUTTON = "oidc-login-button";
42      public static final String    ROUTE             = "login";
43      private final       LoginForm loginForm         = new LoginForm();
44      private final       boolean   oidcEnabled;
45  
46      public LoginView(Environment environment) {
47          this.oidcEnabled = environment.getProperty("spring.security.oauth2.client.registration.keycloak.client-id") != null;
48  
49          setId(LOGIN_VIEW);
50          addClassName(LOGIN_VIEW);
51          setSizeFull();
52          setAlignItems(Alignment.CENTER);
53          setJustifyContentMode(JustifyContentMode.CENTER);
54  
55          loginForm.setAction("login");
56          loginForm.setForgotPasswordButtonVisible(false);
57  
58          H1 title = new H1("Project Hub");
59          title.addClassNames(Margin.Bottom.MEDIUM);
60  
61          add(title);
62  
63          // Create a container for authentication options
64          VerticalLayout authContainer = new VerticalLayout();
65          authContainer.setWidthFull();
66          authContainer.setMaxWidth("400px");
67          authContainer.setAlignItems(Alignment.CENTER);
68          authContainer.getStyle().set("margin", "0 auto");
69  
70          // Only add basic auth form if OIDC is not the only authentication method
71          authContainer.add(new H3("Login with username and password"));
72          authContainer.add(loginForm);
73  
74          Paragraph hint = new Paragraph("Default credentials: admin / admin123");
75          hint.addClassName(Margin.Top.SMALL);
76          authContainer.add(hint);
77  
78          // Add OIDC login option if it's enabled
79          if (oidcEnabled) {
80              Div separator = new Div();
81              separator.setText("OR");
82              separator.getStyle().set("margin", "20px 0");
83              separator.getStyle().set("text-align", "center");
84              separator.getStyle().set("width", "100%");
85              authContainer.add(separator);
86  
87              // Create OIDC login button
88              Anchor loginWithOidcButton = new Anchor("/oauth2/authorization/keycloak", "Login with SSO");
89              loginWithOidcButton.setRouterIgnore(true); // Prevent Vaadin from intercepting the link
90              loginWithOidcButton.setId(OIDC_LOGIN_BUTTON);
91              loginWithOidcButton.getStyle().set("background-color", "var(--lumo-primary-color)");
92              loginWithOidcButton.getStyle().set("color", "var(--lumo-primary-contrast-color)");
93              loginWithOidcButton.getStyle().set("padding", "10px 20px");
94              loginWithOidcButton.getStyle().set("border-radius", "4px");
95              loginWithOidcButton.getStyle().set("cursor", "pointer");
96              loginWithOidcButton.getStyle().set("text-decoration", "none");
97              loginWithOidcButton.getStyle().set("display", "block");
98              loginWithOidcButton.getStyle().set("text-align", "center");
99              loginWithOidcButton.getStyle().set("margin-top", "20px");
100 
101             authContainer.add(loginWithOidcButton);
102         }
103 
104         add(authContainer);
105     }
106 
107     @Override
108     public void beforeEnter(BeforeEnterEvent beforeEnterEvent) {
109         // Check if authentication error occurred
110         if (beforeEnterEvent.getLocation()
111                 .getQueryParameters()
112                 .getParameters()
113                 .containsKey("error")) {
114             loginForm.setError(true);
115         }
116     }
117 }
