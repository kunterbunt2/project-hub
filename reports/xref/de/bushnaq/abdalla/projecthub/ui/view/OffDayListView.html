91      @Override
92      public void beforeEnter(BeforeEnterEvent event) {
93          // Get username from URL parameter or use the currently authenticated user
94          String usernameParam = event.getRouteParameters().get("username").orElse(null);
95  
96          Authentication authentication  = SecurityContextHolder.getContext().getAuthentication();
97          String         currentUsername = authentication != null ? authentication.getName() : null;
98  
99          // If no username is provided, use the current authenticated user
100         // Store in a final variable to use in lambda
101         final String username = (usernameParam == null && currentUsername != null) ?
102                 currentUsername : usernameParam;
103 
104         if (username != null) {
105             try {
106                 // Find user by username using the direct getByName method
107                 currentUser = userApi.getByName(username);
108                 currentUser.initialize();
109             } catch (ResponseStatusException ex) {
110                 if (ex.getStatusCode().equals(HttpStatus.NOT_FOUND)) {
111                     // Create a new user since one wasn't found
112                     User newUser = createDefaultUser(username);
113                     currentUser = userApi.persist(newUser);
114                     currentUser.initialize();
115                     Notification notification = Notification.show("Created new user: " + username, 3000, Notification.Position.MIDDLE);
116                     notification.addThemeVariants(NotificationVariant.LUMO_SUCCESS);
117                 } else {
118                     throw ex;
119                 }
120             }
121         } else {
122             // Redirect to main page if no username and not authenticated
123             event.forwardTo("");
124         }
125         createCalendar();
126     }

128     private void confirmDelete(OffDay offDay) {
129         ConfirmDialog dialog = new ConfirmDialog("Confirm Delete", "Are you sure you want to delete this off day record?", "Delete",
130                 () -> {
131                     try {
132                         offDayApi.deleteById(currentUser, offDay);
133                         refreshOffDayGrid();
134                         // Update the calendar with the fresh user data
135                         if (yearCalendar != null) {
136                             yearCalendar.updateCalendar(currentUser);
137                         }
138                         Notification.show("Off day deleted", 3000, Notification.Position.MIDDLE);
139                     } catch (Exception ex) {
140                         Notification notification = Notification.show("Failed to delete: " + ex.getMessage(), 3000, Notification.Position.MIDDLE);
141                         notification.addThemeVariants(NotificationVariant.LUMO_ERROR);
142                     }
143                 });
144         dialog.open();
145     }

147     private YearCalendarComponent createCalendar() {
148         // Create a new YearCalendarComponent with the current user and the current year
149         if (yearCalendar != null)
150             return yearCalendar;
151         yearCalendar = new YearCalendarComponent(currentUser, java.time.LocalDate.now().getYear(), this::handleCalendarDayClick);
152         return yearCalendar;
153     }

155     /**
156      * Creates a default user with standard settings
157      *
158      * @param username The name for the new user
159      * @return A new User object with default settings
160      */
161     private User createDefaultUser(String username) {
162         User user = new User();
163         user.setName(username);
164         user.setEmail(username);
165         user.setFirstWorkingDay(java.time.LocalDate.now());
166         user.setColor(new java.awt.Color(51, 102, 204));
167         Availability availability = new Availability(1.0f, java.time.LocalDate.now());
168         user.addAvailability(availability);
169         de.bushnaq.abdalla.projecthub.dto.Location location = new Location("DE", "nw", java.time.LocalDate.now());
170         user.addLocation(location);
171         return user;
172     }

174     private Grid<OffDay> createGrid() {
175         grid.setId(OFFDAY_GRID);
176         grid.setWidthFull();
177         grid.addClassNames(LumoUtility.Border.ALL, LumoUtility.BorderColor.CONTRAST_10);
178 
179         // Add columns
180         grid.addColumn(new ComponentRenderer<>(offDay -> {
181                     DateTimeFormatter formatter    = DateTimeFormatter.ofPattern("yyyy-MM-dd");
182                     String            startDateStr = offDay.getFirstDay().format(formatter);
183                     Span              span         = new Span(startDateStr);
184                     span.setId(OFFDAY_GRID_START_DATE_PREFIX + offDay.getId());
185                     return span;
186                 }))
187                 .setHeader(createHeaderWithIcon(VaadinIcon.CALENDAR, "First Day"))
188                 .setSortable(true)
189                 .setKey("firstDay");
190 
191         grid.addColumn(new ComponentRenderer<>(offDay -> {
192                     DateTimeFormatter formatter  = DateTimeFormatter.ofPattern("yyyy-MM-dd");
193                     String            endDateStr = offDay.getLastDay().format(formatter);
194                     Span              span       = new Span(endDateStr);
195                     span.setId(OFFDAY_GRID_END_DATE_PREFIX + offDay.getId());
196                     return span;
197                 }))
198                 .setHeader(createHeaderWithIcon(VaadinIcon.CALENDAR, "Last Day"))
199                 .setSortable(true)
200                 .setKey("lastDay");
201 
202         grid.addColumn(new ComponentRenderer<>(offDay -> {
203                     OffDayType type = offDay.getType();
204                     Span       span = new Span(type.name());
205                     span.setId(OFFDAY_GRID_TYPE_PREFIX + offDay.getId());
206                     return span;
207                 }))
208                 .setHeader(createHeaderWithIcon(VaadinIcon.TAGS, "Type"))
209                 .setSortable(true)
210                 .setKey("type");
211 
212         // Add action column
213         grid.addColumn(new ComponentRenderer<>(offDay -> {
214             HorizontalLayout layout = new HorizontalLayout();
215             layout.setAlignItems(FlexComponent.Alignment.CENTER);
216             layout.setSpacing(true);
217 
218             Button editButton = new Button(new Icon(VaadinIcon.EDIT));
219             editButton.setId(OFFDAY_GRID_EDIT_BUTTON_PREFIX + offDay.getId());
220             editButton.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_TERTIARY);
221             editButton.addClickListener(e -> openOffDayDialog(offDay));
222             editButton.getElement().setAttribute("title", "Edit");
223 
224             Button deleteButton = new Button(new Icon(VaadinIcon.TRASH));
225             deleteButton.setId(OFFDAY_GRID_DELETE_BUTTON_PREFIX + offDay.getId());
226             deleteButton.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_TERTIARY, ButtonVariant.LUMO_ERROR);
227             deleteButton.addClickListener(e -> confirmDelete(offDay));
228             deleteButton.getElement().setAttribute("title", "Delete");
229 
230             layout.add(editButton, deleteButton);
231             return layout;
232         })).setHeader(createHeaderWithIcon(VaadinIcon.COG, "Actions")).setFlexGrow(0).setWidth("120px");
233 
234         return grid;
235     }

251     /**
252      * Handles clicks on calendar days that have off days
253      *
254      * @param date The date that was clicked
255      */
256     private void handleCalendarDayClick(java.time.LocalDate date) {
257         if (currentUser != null) {
258             // Find the off day that contains this date
259             OffDay matchingOffDay = currentUser.getOffDays().stream()
260                     .filter(offDay ->
261                             (date.equals(offDay.getFirstDay()) || date.isAfter(offDay.getFirstDay())) &&
262                                     (date.equals(offDay.getLastDay()) || date.isBefore(offDay.getLastDay())))
263                     .findFirst()
264                     .orElse(null);
265 
266             if (matchingOffDay != null) {
267                 // Open the dialog to edit this off day
268                 openOffDayDialog(matchingOffDay);
269             }
270         }
271     }

273     private void openOffDayDialog(OffDay offDay) {
274         // Create new or edit existing off day
275         OffDayDialog dialog = new OffDayDialog(
276                 offDay,
277                 this.currentUser,
278                 this.offDayApi,
279                 () -> {
280                     refreshOffDayGrid();
281                     if (yearCalendar != null) {
282                         // Pass the updated user to the calendar
283                         yearCalendar.updateCalendar(currentUser);
284                     }
285                 });
286         dialog.open();
287     }

290     private void refreshOffDayGrid() {
291         if (currentUser != null) {
292             // Reload the user to get fresh data
293             currentUser = userApi.getById(currentUser.getId());
294             currentUser.initialize();
295 
296             // Sort off days by start date (newest to oldest)
297             List<OffDay> sortedOffDays = currentUser.getOffDays().stream().sorted(Comparator.comparing(OffDay::getFirstDay).reversed()).collect(Collectors.toList());
298 
299             grid.setItems(sortedOffDays);
300 
301             yearCalendar.updateCalendar(currentUser);
302         }
303     }
