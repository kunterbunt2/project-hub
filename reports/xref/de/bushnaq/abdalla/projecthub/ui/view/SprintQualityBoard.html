181     private Div createFieldDisplay(String label, String value, String status) {
182         Div container = new Div();
183         container.getStyle()
184                 .set("display", "flex")
185                 .set("flex-direction", "column")
186                 .set("padding", "var(--lumo-space-s)")
187                 .set("background", "var(--lumo-base-color)");
188 
189         // Create a wrapper div for the value part that doesn't stretch
190         Div valueWrapper = new Div();
191         valueWrapper.getStyle()
192                 .set("display", "flex") // Use flex to allow inner content to determine size
193                 .set("width", "auto");  // Don't stretch to full width
194 
195         // Value part (top)
196         Span valueSpan = new Span(value);
197         valueSpan.getStyle()
198                 .set("font-weight", "500")
199                 .set("font-size", "var(--lumo-font-size-xl)");
200 
201         // Apply status-based styling if status is provided
202         if (status != null) {
203             valueSpan.getStyle()
204                     .set("padding", "0 var(--lumo-space-xs)")
205                     .set("border-radius", "var(--lumo-border-radius-s)");
206 
207             switch (status) {
208                 case "GOOD":
209                     valueSpan.getStyle().set("background-color", "var(--lumo-success-color)");
210                     break;
211                 case "NORMAL":
212                     // Default styling
213                     break;
214                 case "WARNING":
215                     valueSpan.getStyle().set("background-color", "rgba(255, 179, 0, 1)");
216                     break;
217                 case "CRITICAL":
218                     valueSpan.getStyle().set("background-color", "var(--lumo-error-color)");
219                     break;
220             }
221         }
222 
223         valueWrapper.add(valueSpan);
224 
225         // Name part with HTML support for special characters
226         Span nameSpan = new Span();
227         nameSpan.getElement().setProperty("innerHTML", label);
228         nameSpan.getStyle()
229                 .set("font-size", "var(--lumo-font-size-xs)")
230                 .set("color", "var(--lumo-secondary-text-color)")
231                 .set("margin-top", "var(--lumo-space-xs)");
232 
233         container.add(valueWrapper, nameSpan);
234         return container;
235     }

299 
300         // First row
301         gridContainer.add(createFieldDisplay("Sprint Name", sprint.getName()));//column 1
302         gridContainer.add(createFieldDisplay("Sprint Start Date", DateUtil.createDateString(sprint.getStart(), dtfymd)));//column 2

306         gridContainer.add(createFieldDisplay("&Sigma; Effort Spent", DateUtil.createDurationString(sprint.getWorked(), false, true, false)));//column 7
307

312         gridContainer.add(createFieldDisplay("Progress", String.format("%.0f%%", 100 * ReportUtil.calcualteProgress(sprint.getWorked(), estimation)), status));//column 4
313         gridContainer.add(createFieldDisplay("Current Schedule Delay", DateUtil.createDurationString(ReportUtil.calcualteWorkDaysMiliseconsDelay(sprint.getStart(), now, sprint.getEnd(), sprint.getWorked(), sprint.getOriginalEstimation(), estimation, sprint.getRemaining()), false, true, false), status));//column 6

316

323         }
324         gridContainer.add(createFieldDisplay("Optimal Efficiency", ReportUtil.createPersonDayEfficiencyString(ReportUtil.calcualteOptimaleEfficiency(sprint.getStart(), sprint.getEnd(), DateUtil.add(sprint.getWorked(), sprint.getRemaining())))));//column 3

337         gridContainer.add(createFieldDisplay("4.3", "a"));//column 3

406         return burndownChart;

418     private void loadData() {
419         //- populate grid with tasks of the sprint
420         long time = System.currentTimeMillis();
421 
422         // Capture the security context from the current thread
423         Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
424 
425         // Load in parallel with security context propagation
426         CompletableFuture<Sprint> sprintFuture = CompletableFuture.supplyAsync(() -> {
427             // Set security context in this thread
428             SecurityContext context = SecurityContextHolder.createEmptyContext();
429             context.setAuthentication(authentication);
430             SecurityContextHolder.setContext(context);
431             try {
432                 Sprint s = sprintApi.getById(sprintId);
433                 s.initialize();
434                 return s;
435             } finally {
436                 SecurityContextHolder.clearContext();// Clear the security context after execution
437             }
438         });
439 
440         CompletableFuture<List<User>> usersFuture = CompletableFuture.supplyAsync(() -> {
441             // Set security context in this thread
442             SecurityContext context = SecurityContextHolder.createEmptyContext();
443             context.setAuthentication(authentication);
444             SecurityContextHolder.setContext(context);
445             try {
446                 return userApi.getAll(sprintId);
447             } finally {
448                 SecurityContextHolder.clearContext();// Clear the security context after execution
449 
450             }
451         });
452 
453         CompletableFuture<List<Task>> tasksFuture = CompletableFuture.supplyAsync(() -> {
454             // Set security context in this thread
455             SecurityContext context = SecurityContextHolder.createEmptyContext();
456             context.setAuthentication(authentication);
457             SecurityContextHolder.setContext(context);
458             try {
459                 return taskApi.getAll(sprintId);
460             } finally {
461                 SecurityContextHolder.clearContext();// Clear the security context after execution
462             }
463         });
464 
465         CompletableFuture<List<Worklog>> worklogsFuture = CompletableFuture.supplyAsync(() -> {
466             // Set security context in this thread
467             SecurityContext context = SecurityContextHolder.createEmptyContext();
468             context.setAuthentication(authentication);
469             SecurityContextHolder.setContext(context);
470             try {
471                 return worklogApi.getAll(sprintId);
472             } finally {
473                 SecurityContextHolder.clearContext();// Clear the security context after execution
474             }
475         });
476 
477         // Wait for all futures and combine results
478         try {
479             sprint = sprintFuture.get();
480             logger.info("sprint loaded and initialized in {} ms", System.currentTimeMillis() - time);
481             time = System.currentTimeMillis();
482             sprint.initUserMap(usersFuture.get());
483             sprint.initTaskMap(tasksFuture.get(), worklogsFuture.get());
484             logger.info("sprint user, task and worklog maps initialized in {} ms", System.currentTimeMillis() - time);
485             sprint.recalculate(ParameterOptions.getLocalNow());
486         } catch (InterruptedException | ExecutionException e) {
487             logger.error("Error loading sprint data", e);
488             // Handle exception appropriately
489         }
490     }

518             throw new RuntimeException(e);

531                 logger.warn("Failed to close output stream", closeException);
532             }
533             throw new RuntimeException(e);
534         }
535     }

537     private Image renderGanttImage(GanttChart chart) {
538         StreamResource resource = new StreamResource("gantt.svg", () -> {
539             try (ByteArrayOutputStream outputStream = renderGanttChart(chart)) {
540                 // Convert ByteArrayOutputStream to ByteArrayInputStream
541                 return new ByteArrayInputStream(outputStream.toByteArray());
542             } catch (Exception e) {
543                 logger.error("Error creating gantt chart", e);
544                 return new ByteArrayInputStream(new byte[0]); // Empty stream in case of error
545             }
546         });
547         // Create an image component with the SVG resource
548         Image ganttChart = new Image(resource, "Sprint Gantt Chart");
549         return ganttChart;
