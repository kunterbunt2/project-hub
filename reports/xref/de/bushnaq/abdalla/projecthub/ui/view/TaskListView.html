53  @Route("task-list")
54  @PageTitle("Task List Page")
55  @CssImport("./styles/grid-styles.css")
56  @PermitAll // When security is enabled, allow all authenticated users
57  @RolesAllowed({"USER", "ADMIN"}) // Allow access to users with specific roles
58  public class TaskListView extends Main implements AfterNavigationObserver {
59      public static final String     TASK_GRID_NAME_PREFIX = "task-grid-name-";
60      private final       Clock      clock;
61      private final       Grid<Task> grid;
62      private final       H2         pageTitle;
63      private             Long       productId;
64      private             Long       projectId;
65      private             Sprint     sprint;
66      private final       SprintApi  sprintApi;
67      private             Long       sprintId;
68      private final       TaskApi    taskApi;
69      private final       UserApi    userApi;
70      private             Long       versionId;

72      public TaskListView(TaskApi taskApi, SprintApi sprintApi, UserApi userApi, Clock clock) {
73          this.taskApi   = taskApi;
74          this.sprintApi = sprintApi;
75          this.userApi   = userApi;
76          this.clock     = clock;
77  
78          // Create title layout with icon
79          HorizontalLayout titleLayout = new HorizontalLayout();
80          titleLayout.setSpacing(true);
81          titleLayout.setAlignItems(FlexComponent.Alignment.CENTER);
82  
83          Icon taskIcon = new Icon(VaadinIcon.TASKS);
84          pageTitle = new H2("Tasks");
85          pageTitle.addClassNames(
86                  LumoUtility.Margin.Top.MEDIUM,
87                  LumoUtility.Margin.Bottom.SMALL
88          );
89  
90          titleLayout.add(taskIcon, pageTitle);
91  
92          grid = new Grid<>();
93          setupGridColumns();
94          // Add borders between columns
95          grid.addThemeVariants(GridVariant.LUMO_COLUMN_BORDERS, GridVariant.LUMO_ROW_STRIPES);
96  
97          // Add custom styling for more JIRA-like appearance
98          grid.addClassName("jira-style-grid");
99  
100         // Apply custom styling directly to make borders lighter gray
101         grid.getElement().getStyle().set("--lumo-contrast-10pct", "#e0e0e0");
102         grid.setSizeFull();
103 
104         setSizeFull();
105         addClassNames(LumoUtility.BoxSizing.BORDER, LumoUtility.Display.FLEX, LumoUtility.FlexDirection.COLUMN);
106 
107         add(titleLayout, grid);
108     }

110     @Override
111     public void afterNavigation(AfterNavigationEvent event) {
112         //- Get query parameters
113         Location        location        = event.getLocation();
114         QueryParameters queryParameters = location.getQueryParameters();
115         if (queryParameters.getParameters().containsKey("product")) {
116             this.productId = Long.parseLong(queryParameters.getParameters().get("product").getFirst());
117         }
118         if (queryParameters.getParameters().containsKey("version")) {
119             this.versionId = Long.parseLong(queryParameters.getParameters().get("version").getFirst());
120         }
121         if (queryParameters.getParameters().containsKey("project")) {
122             this.projectId = Long.parseLong(queryParameters.getParameters().get("project").getFirst());
123         }
124         if (queryParameters.getParameters().containsKey("sprint")) {
125             this.sprintId = Long.parseLong(queryParameters.getParameters().get("sprint").getFirst());
126             pageTitle.setText("Task of Sprint ID: " + sprintId);
127         }
128 
129         //- Update breadcrumbs
130         getElement().getParent().getComponent()
131                 .ifPresent(component -> {
132                     if (component instanceof MainLayout mainLayout) {
133                         mainLayout.getBreadcrumbs().clear();
134                         mainLayout.getBreadcrumbs().addItem("Products", ProductListView.class);
135                         {
136                             Map<String, String> params = new HashMap<>();
137                             params.put("product", String.valueOf(productId));
138                             mainLayout.getBreadcrumbs().addItem("Versions", VersionListView.class, params);
139                         }
140                         {
141                             Map<String, String> params = new HashMap<>();
142                             params.put("product", String.valueOf(productId));
143                             params.put("version", String.valueOf(versionId));
144                             mainLayout.getBreadcrumbs().addItem("Projects", FeatureListView.class, params);
145                         }
146                         {
147                             Map<String, String> params = new HashMap<>();
148                             params.put("product", String.valueOf(productId));
149                             params.put("version", String.valueOf(versionId));
150                             params.put("project", String.valueOf(projectId));
151                             mainLayout.getBreadcrumbs().addItem("Sprints", SprintListView.class, params);
152                         }
153                         {
154                             Map<String, String> params = new HashMap<>();
155                             params.put("product", String.valueOf(productId));
156                             params.put("version", String.valueOf(versionId));
157                             params.put("project", String.valueOf(projectId));
158                             params.put("sprint", String.valueOf(sprintId));
159                             mainLayout.getBreadcrumbs().addItem("Tasks", TaskListView.class, params);
160                         }
161                     }
162                 });
163 
164         //- populate grid
165         sprint = sprintApi.getById(sprintId);
166         sprint.initUserMap(userApi.getAll(sprintId));
167         sprint.initTaskMap(taskApi.getAll(sprintId), null);
168         pageTitle.setText("Task of Sprint ID: " + sprintId);
169         grid.setItems(sprint.getTasks());
170     }

172     private void setupGridColumns() {
173         DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofLocalizedDateTime(FormatStyle.LONG).withZone(clock.getZone()).withLocale(getLocale());
174 
175         grid.addColumn(Task::getKey).setHeader("Key").setAutoWidth(true);
176         grid.addColumn(task -> task.getParentTask() != null ? task.getParentTask().getKey() : "").setHeader("Parent").setAutoWidth(true);
177         grid.addColumn(new ComponentRenderer<>(task -> {
178             Div div    = new Div();
179             Div square = new Div();
180             square.setMinHeight("16px");
181             square.setMaxHeight("16px");
182             square.setMinWidth("16px");
183             square.setMaxWidth("16px");
184             square.getStyle().set("float", "left");
185             square.getStyle().set("margin", "1px");
186             div.add(square);
187             div.add(task.getName());
188             div.setId(TASK_GRID_NAME_PREFIX + task.getName());
189             return div;
190         })).setHeader("Name").setAutoWidth(true);
191         grid.addColumn(task -> task.getResourceId() != null ? sprint.getuser(task.getResourceId()).getName() : "").setHeader("Assigned").setAutoWidth(true);
192         grid.addColumn(task -> dateTimeFormatter.format(task.getStart())).setHeader("Start").setAutoWidth(true);
193         grid.addColumn(task -> dateTimeFormatter.format(task.getFinish())).setHeader("End").setAutoWidth(true);
194         grid.addColumn(task -> !task.getOriginalEstimate().equals(Duration.ZERO) ? DateUtil.createDurationString(task.getOriginalEstimate(), false, true, false) : "").setHeader("Original Estimate").setAutoWidth(true);
195         grid.addColumn(task -> !task.getOriginalEstimate().equals(Duration.ZERO) ? DateUtil.createDurationString(task.getTimeSpent(), false, true, false) : "").setHeader("Time Spent").setAutoWidth(true);
196         grid.addColumn(task -> !task.getOriginalEstimate().equals(Duration.ZERO) ? DateUtil.createDurationString(task.getRemainingEstimate(), false, true, false) : "").setHeader("Remaining Estimate").setAutoWidth(true);
197         grid.addColumn(task -> !task.getOriginalEstimate().equals(Duration.ZERO) ? String.format("%2.0f%%", task.getProgress().doubleValue() * 100) : "").setHeader("Progress").setAutoWidth(true);
198         grid.addColumn(task -> {
199             List<Relation> relations = task.getPredecessors();
200             if (relations == null || relations.isEmpty()) {
201                 return "";
202             }
203 
204             return relations.stream()
205                     .map(relation -> {
206                         Task predecessor = sprint.getTaskById(relation.getPredecessorId());
207                         return predecessor != null ? predecessor.getKey() : "";
208                     })
209                     .filter(key -> !key.isEmpty())
210                     .collect(Collectors.joining(", "));
211         }).setHeader("Dependency").setAutoWidth(true);
212         grid.addColumn(task -> task.getTaskMode().name()).setHeader("Mode").setAutoWidth(true);
213 
214     }
