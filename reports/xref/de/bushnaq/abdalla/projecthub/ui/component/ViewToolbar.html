18  package de.bushnaq.abdalla.projecthub.ui.component;
19  
20  import com.vaadin.flow.component.Component;
21  import com.vaadin.flow.component.Composite;
22  import com.vaadin.flow.component.applayout.DrawerToggle;
23  import com.vaadin.flow.component.html.Div;
24  import com.vaadin.flow.component.html.H1;
25  import com.vaadin.flow.component.html.Header;
26  import com.vaadin.flow.theme.lumo.LumoUtility.*;
27  
28  public final class ViewToolbar extends Composite<Header> {
29  
30      public ViewToolbar(String viewTitle, Component... components) {
31          addClassNames(Display.FLEX, FlexDirection.COLUMN, JustifyContent.BETWEEN, AlignItems.STRETCH, Gap.MEDIUM,
32                  FlexDirection.Breakpoint.Medium.ROW, AlignItems.Breakpoint.Medium.CENTER);
33  
34          var drawerToggle = new DrawerToggle();
35          drawerToggle.addClassNames(Margin.NONE);
36  
37          var title = new H1(viewTitle);
38          title.addClassNames(FontSize.XLARGE, Margin.NONE, FontWeight.LIGHT);
39  
40          var toggleAndTitle = new Div(drawerToggle, title);
41          toggleAndTitle.addClassNames(Display.FLEX, AlignItems.CENTER);
42          getContent().add(toggleAndTitle);
43  
44          if (components.length > 0) {
45              var actions = new Div(components);
46              actions.addClassNames(Display.FLEX, FlexDirection.COLUMN, JustifyContent.BETWEEN, Flex.GROW, Gap.SMALL,
47                      FlexDirection.Breakpoint.Medium.ROW);
48              getContent().add(actions);
49          }
50      }
51  
52      public static Component group(Component... components) {
53          var group = new Div(components);
54          group.addClassNames(Display.FLEX, FlexDirection.COLUMN, AlignItems.STRETCH, Gap.SMALL,
55                  FlexDirection.Breakpoint.Medium.ROW, AlignItems.Breakpoint.Medium.CENTER);
56          return group;
57      }
58  }
