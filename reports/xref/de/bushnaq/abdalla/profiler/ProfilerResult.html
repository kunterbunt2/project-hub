18  package de.bushnaq.abdalla.profiler;
19  
20  public class ProfilerResult {
21      long      delta;
22      SampleSet sampleSet;
23      long      totalDelta;
24  
25      public ProfilerResult(long totalStart, SampleSet sampleMap) {
26          this.sampleSet = sampleMap;
27          totalDelta     = System.nanoTime() - totalStart;
28  
29          long sum = 0;
30          for (SampleType type : sampleMap.sampleKeySet()) {
31              Sample profilingCounter = sampleMap.getSample(type);
32              sum += profilingCounter.getTimeNanoSec();
33          }
34          delta = totalDelta - sum;
35      }
36  
37  }
