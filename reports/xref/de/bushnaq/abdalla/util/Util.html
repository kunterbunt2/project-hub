55      public static Double add(Double value1, Double value2) {
56          if (value1 == null && value2 == null) {
57              return null;
58          } else if (value1 == null) {
59              return value2;
60          } else if (value2 == null) {
61              return value1;
62          } else {
63              return value1 + value2;
64          }
65      }

67      public static String columnIdtoExcelColumnName(int number) {
68          final StringBuilder sb = new StringBuilder();
69  
70          int num = number - 1;
71          while (num >= 0) {
72              int numChar = (num % 26) + 65;
73              sb.append((char) numChar);
74              num = (num / 26) - 1;
75          }
76          return sb.reverse().toString();
77      }

79      public static Set<String> createColumnSet(PreparedStatement preparedStatement) throws SQLException {
80          Set<String>       columns = new HashSet<>();
81          ResultSetMetaData md      = preparedStatement.getMetaData();
82          int               numCol  = md.getColumnCount();
83          for (int i = 1; i <= numCol; i++) {
84              columns.add(md.getColumnName(i));
85          }
86          return columns;
87      }

93      public static String extractParameters(final String aStringToParse, final Map<String, Integer> aParameterList) {
94          int _lastParameterIndex = 1;
95          aParameterList.clear();
96          final StringBuffer _stringToParse      = new StringBuffer(aStringToParse);
97          int                _foundColumnAtIndex = -1;
98          do {
99              _foundColumnAtIndex = _stringToParse.indexOf(":");
100             if (_foundColumnAtIndex > -1) {
101                 final int _foundSpaceAtIndex = findParameterEnd(_stringToParse, _foundColumnAtIndex + 1);
102                 if (_foundSpaceAtIndex > -1) {
103                     final String _parameterName = _stringToParse.substring(_foundColumnAtIndex + 1, _foundSpaceAtIndex);
104                     aParameterList.put(_parameterName, Integer.valueOf(_lastParameterIndex++));
105                     _stringToParse.replace(_foundColumnAtIndex, _foundSpaceAtIndex, "?");
106                 } else {
107                     // ---Finished
108                     break;
109                 }
110             } else {
111                 // ---Finished
112                 break;
113             }
114         } while (_foundColumnAtIndex != -1);
115         return _stringToParse.toString();
116     }

124     public static int findParameterEnd(final StringBuffer aStringToParse, final int aStartIndex) {
125         final String[] _token = {",", " ", "\r", ")"};
126         int            _index = aStringToParse.length();
127         for (int i = 0; i < _token.length; i++) {
128             final int _tempIndex = aStringToParse.indexOf(_token[i], aStartIndex);
129             if ((_tempIndex != -1) && (_tempIndex < _index)) {
130                 _index = _tempIndex;
131             }
132         }
133         return _index;
134     }

150     public static Range getIntersection(LocalDateTime os, LocalDateTime of, LocalDateTime ps, LocalDateTime pf) {
151         //o   |      |
152         //i   ------
153         //p |      |
154         if (!os.isBefore(ps) && !os.isAfter(pf)) {
155             return new Range(os, pf);
156         }
157         //o|  |
158         //i ---
159         //p |      |
160         if (!of.isBefore(ps) && !of.isAfter(pf)) {
161             return new Range(ps, of);
162         }
163         //o|        |
164         //i  -------
165         //p |      |
166         if (!os.isAfter(ps) && !of.isBefore(pf)) {
167             return new Range(ps, pf);
168         }
169         //o  |    |
170         //i  ------
171         //p |      |
172         if (os.isAfter(ps) && of.isBefore(pf)) {
173             return new Range(os, of);
174         }
175         return null;
176     }

178     public static String longToString(final Long aValue, final boolean aCreateLeadingZero) {
179         if (aValue != null) {
180             if (!aCreateLeadingZero || (aValue > 9)) {
181                 return Long.toString(aValue);
182             } else {
183                 return "0" + aValue;
184             }
185         } else {
186             return "";
187         }
188     }

190     public static String putCsvList(String[] list) {
191         StringBuilder result = new StringBuilder();
192         if (list != null) {
193             for (int i = 0; i < list.length; i++) {
194                 if (i != 0) {
195                     result.append("\\|");
196                 }
197                 result.append(list[i]);
198             }
199         } else {
200             result.append("null");
201         }
202         return result.toString();
203     }
