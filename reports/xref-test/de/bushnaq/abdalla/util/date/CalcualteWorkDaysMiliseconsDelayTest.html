18  package de.bushnaq.abdalla.util.date;
19  
20  import org.junit.jupiter.api.Test;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import java.time.Duration;
25  import java.time.LocalDateTime;
26  import java.time.format.DateTimeFormatter;
27  
28  import static org.junit.jupiter.api.Assertions.assertEquals;
29  
30  public class CalcualteWorkDaysMiliseconsDelayTest {
31      static        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss");
32      private final Logger            logger    = LoggerFactory.getLogger(this.getClass());
33  
34      @Test
35      public void ct2Test() {
36          //start=2018.Aug.28 11:17 now=2018.Sep.14 08:39 end=2018.Okt.08 18:00 worked=01w 02d 05h 39m remaining=03w 01d 02h 31m
37  
38          LocalDateTime start = LocalDateTime.parse("2018-08-28 11:17:00", formatter);
39          //        Calendar start = new GregorianCalendar(2018, Calendar.AUGUST, 28, 11, 17);
40          LocalDateTime now = LocalDateTime.parse("2018-09-13 08:39:00", formatter);
41          //        Calendar now = new GregorianCalendar(2018, Calendar.SEPTEMBER, 13, 8, 39);
42          LocalDateTime end = LocalDateTime.parse("2018-10-08 18:00:00", formatter);
43          //        Calendar end = new GregorianCalendar(2018, Calendar.OCTOBER, 8, 18, 0);
44          Duration worked = Duration.ofMinutes(7 * ReportUtil.ONE_WORKING_DAY_MINUTES).plusHours(5).plusMinutes(39);
45          //        Long worked = new Long(7 * Util.ONE_WORKING_DAY_MILLIS + (5 * 60 + 39) * 60 * 1000);
46          Duration remaining = Duration.ofMinutes(16 * ReportUtil.ONE_WORKING_DAY_MINUTES).plusHours(2).plusMinutes(31);
47          //        Long remaining = new Long(16 * Util.ONE_WORKING_DAY_MILLIS + (2 * 60 + 31) * 60 * 1000);
48          logger.info(String.format("start =     %s", start.format(formatter)));
49          logger.info(String.format("now =       %s", now.format(formatter)));
50          logger.info(String.format("end =       %s", end.format(formatter)));
51          logger.info(String.format("worked =    %s", DateUtil.createDurationString(worked, false, true, true)));
52          logger.info(String.format("remaining = %s", DateUtil.createDurationString(remaining, false, true, true)));
53  
54          Duration calcualteWorkDaysMiliseconsDelay = ReportUtil.calcualteWorkDaysMiliseconsDelay(start, now, end, worked, null, null, remaining);
55  
56          logger.info(String.format("Effort Estimate = %s", DateUtil.createWorkDayDurationString(worked.plus(remaining), false, true, false)));
57  
58          logger.info(String.format("Current sprint delay = %s", DateUtil.createDurationString(calcualteWorkDaysMiliseconsDelay, false, true, false)));
59  
60          assertEquals("4d 03h 46m", DateUtil.createDurationString(calcualteWorkDaysMiliseconsDelay, false, true, false));
61      }
