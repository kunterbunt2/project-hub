18  package de.bushnaq.abdalla.projecthub.util;
19  
20  import jakarta.persistence.EntityManager;
21  import org.junit.jupiter.api.AfterEach;
22  import org.junit.jupiter.api.BeforeEach;
23  import org.junit.jupiter.api.TestInfo;
24  import org.springframework.beans.factory.annotation.Autowired;
25  
26  import java.util.List;
27  
28  public class AbstractTestUtil extends DTOAsserts {
29      @Autowired
30      protected EntityManager entityManager;
31  
32      @AfterEach
33      protected void afterEach(TestInfo testInfo) throws Exception {
34          logger.info("----------------------------------------------");
35          logger.info("stop " + testInfo.getDisplayName());
36          logger.info("==============================================");
37      }
38  
39      @BeforeEach
40      protected void beforeEach(TestInfo testInfo) {
41          logger.info("==============================================");
42          logger.info("start " + testInfo.getDisplayName());
43          logger.info("----------------------------------------------");
44          List<String> tableNames = getAllTableNames();
45          entityManager.createNativeQuery("SET REFERENTIAL_INTEGRITY FALSE").executeUpdate();
46          for (String tableName : tableNames) {
47              entityManager.createNativeQuery("TRUNCATE TABLE " + tableName).executeUpdate();
48          }
49          entityManager.createNativeQuery("SET REFERENTIAL_INTEGRITY TRUE").executeUpdate();
50          entityManager.flush();
51      }
52  
53      protected List<String> getAllTableNames() {
54          // Get all table names
55          @SuppressWarnings("unchecked") List<String> tableNames = entityManager.createNativeQuery("SELECT TABLE_NAME FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_SCHEMA = SCHEMA()").getResultList();
56  
57          // Filter out system tables
58          tableNames = tableNames.stream().filter(name -> !name.equals("SPATIAL_REF_SYS") && !name.equals("GEOMETRY_COLUMNS")).toList();
59          return tableNames;
60      }
61  
62      protected void printTables() {
63          System.out.println("\n\n\n");
64          try {
65              String        separator = "+";
66              StringBuilder output    = new StringBuilder("\n=== Database Content ===\n");
67  
68              List<String> tableNames = getAllTableNames();
69  
70              for (String tableName : tableNames) {
71                  // Get column names
72                  @SuppressWarnings("unchecked") List<String> columnNames = entityManager.createNativeQuery("SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS " + "WHERE TABLE_SCHEMA = SCHEMA() AND TABLE_NAME = '" + tableName + "' " + "ORDER BY ORDINAL_POSITION").getResultList();
73  
74                  // Get all data
75                  String  dataQuery = "SELECT * FROM " + tableName;
76                  List<?> results   = entityManager.createNativeQuery(dataQuery).getResultList();
77  
78                  // Calculate max width for each column
79                  int[] maxWidths = new int[columnNames.size()];
80                  // Initialize with column name lengths
81                  for (int i = 0; i < columnNames.size(); i++) {
82                      maxWidths[i] = columnNames.get(i).length();
83                  }
84  
85                  // Check data widths
86                  for (Object row : results) {
87                      if (row instanceof Object[] cells) {
88                          for (int i = 0; i < cells.length; i++) {
89                              String value = cells[i] != null ? cells[i].toString() : "null";
90                              maxWidths[i] = Math.max(maxWidths[i], value.length());
91                          }
92                      } else {
93                          maxWidths[0] = Math.max(maxWidths[0], row != null ? row.toString().length() : 4);
94                      }
95                  }
96  
97                  // Create dynamic separator
98                  separator = "+";
99                  for (int width : maxWidths) {
100                     separator += "-".repeat(width + 2) + "+";
101                 }
102                 separator += "\n";
103 
104 
105                 if (!results.isEmpty()) {
106                     // Print table
107                     output.append("\n").append(tableName).append(":\n").append(separator);
108                     // Print header
109                     output.append("|");
110                     for (int i = 0; i < columnNames.size(); i++) {
111                         output.append(String.format(" %-" + maxWidths[i] + "s |", columnNames.get(i)));
112                     }
113                     output.append("\n").append(separator);
114                     // Print data
115                     for (Object row : results) {
116                         output.append("|");
117                         if (row instanceof Object[] cells) {
118                             for (int i = 0; i < cells.length; i++) {
119                                 String value = cells[i] != null ? cells[i].toString() : "null";
120                                 output.append(String.format(" %-" + maxWidths[i] + "s |", value));
121                             }
122                         } else {
123                             output.append(String.format(" %-" + maxWidths[0] + "s |", row != null ? row.toString() : "null"));
124                         }
125                         output.append("\n");
126                     }
127                     output.append(separator);
128                 }
129 
130             }
131 
132             System.out.println(output);
133         } catch (Exception e) {
134             System.err.println("Error printing tables: " + e.getMessage());
135             e.printStackTrace();
136         }
137     }
