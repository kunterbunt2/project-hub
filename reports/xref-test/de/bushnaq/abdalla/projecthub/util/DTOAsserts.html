31      private static void assertAvailabilityEquals(Availability expected, Availability actual) {
32          assertEquals(expected.getCreated(), actual.getCreated(), String.format("Availability '%d' created date does not match", actual.getId()));
33  //        assertEquals(expected.getUpdated(), actual.getUpdated(), String.format("Availability '%d' updated date does not match", actual.getId()));
34          assertEquals(expected.getId(), actual.getId(), "Availability IDs do not match");
35          assertEquals(expected.getAvailability(), actual.getAvailability(), "Availability values do not match");
36          assertEquals(expected.getStart(), actual.getStart(), "Availability start dates do not match");
37      }

39      protected static void assertFeatureEquals(Feature expected, Feature actual) {
40          assertFeatureEquals(expected, actual, false);
41      }
42  
43      protected static void assertFeatureEquals(Feature expected, Feature actual, boolean shallow) {
44          assertEquals(expected.getCreated(), actual.getCreated(), String.format("Feature '%s' created date does not match", actual.getName()));
45  //        assertEquals(expected.getUpdated(), actual.getUpdated(), String.format("Project '%s' updated date does not match", actual.getName()));
46          assertEquals(expected.getId(), actual.getId(), "Feature IDs do not match");
47          assertEquals(expected.getName(), actual.getName(), "Feature names do not match");
48          if (!shallow) {
49              assertEquals(expected.getSprints().size(), actual.getSprints().size(), String.format("Number of sprints in feature '%s' do not match", actual.getName()));
50              for (int i = 0; i < expected.getSprints().size(); i++) {
51                  assertSprintEquals(expected.getSprints().get(i), actual.getSprints().get(i));
52              }
53          }
54      }

56      protected static void assertLocalDateTimeEquals(LocalDateTime expected, LocalDateTime actual, String name) {
57          if (expected == null && actual == null) {
58              return;
59          }
60          assertEquals(expected, actual, String.format("%s LocalDateTime mismatch.", name));
61  //        assertTrue(Math.abs(ChronoUnit.MICROS.between(expected, actual)) < 1, () -> String.format("Expected %s but was %s", expected, actual));
62      }

108     protected static void assertSprintEquals(Sprint expected, Sprint actual, boolean shallow) {
109         assertEquals(expected.getCreated(), actual.getCreated(), String.format("Sprint '%s' created date does not match", actual.getName()));
110 //        assertEquals(expected.getUpdated(), actual.getUpdated(), String.format("Sprint '%s' updated date does not match", actual.getName()));
111         assertEquals(expected.getEnd(), actual.getEnd(), "Sprint end dates do not match");
112         assertEquals(expected.getId(), actual.getId(), "Sprint IDs do not match");
113         assertEquals(expected.getName(), actual.getName(), "Sprint names do not match");
114         assertEquals(expected.getStart(), actual.getStart(), "Sprint start dates do not match");
115         assertEquals(expected.getStatus(), actual.getStatus(), "Sprint status values do not match");
116 
117         if (!shallow) {
118             assertEquals(expected.getTasks().size(), actual.getTasks().size(), "Number of tasks in sprint do not match");
119             for (int i = 0; i < expected.getTasks().size(); i++) {
120                 assertTaskEquals(expected.getTasks().get(i), actual.getTasks().get(i));
121             }
122         }
123     }

125     protected static void assertTaskEquals(Task expected, Task actual) {
126         assertEquals(expected.getChildTasks().size(), actual.getChildTasks().size(), "Number of child tasks do not match");
127         for (int i = 0; i < expected.getChildTasks().size(); i++) {
128             assertTaskEquals(expected.getChildTasks().get(i), actual.getChildTasks().get(i));
129         }
130 
131         assertEquals(expected.getDuration(), actual.getDuration(), "Task durations do not match");
132         assertLocalDateTimeEquals(expected.getFinish(), actual.getFinish(), "Task finish");
133         assertEquals(expected.getId(), actual.getId(), "Task IDs do not match");
134         assertEquals(expected.getName(), actual.getName(), "Task names do not match");
135         assertEquals(expected.getParentTask(), actual.getParentTask(), "Task parents do not match");
136 
137         assertEquals(expected.getPredecessors().size(), actual.getPredecessors().size(), "Number of task predecessors do not match");
138         for (int i = 0; i < expected.getPredecessors().size(); i++) {
139             assertRelationEquals(expected.getPredecessors().get(i), actual.getPredecessors().get(i));
140         }
141 
142         assertEquals(expected.getResourceId(), actual.getResourceId(), "Task resource IDs do not match");
143         assertLocalDateTimeEquals(expected.getStart(), actual.getStart(), "Task start");
144     }
