95      private void compareResults(TestInfo testInfo) throws IOException {
96          String expectedJson = Files.readString(Paths.get(testReferenceResultFolder, TestInfoUtil.getTestMethodName(testInfo) + ".json"));
97          String actualJson   = Files.readString(Paths.get(testResultFolder, TestInfoUtil.getTestMethodName(testInfo) + ".json"));
98  
99          try {
100             TypeReference<Map<String, Object>> typeRef      = new TypeReference<>() {
101             };
102             Map<String, Object>                referenceMap = objectMapper.readValue(expectedJson, typeRef);
103             Map<String, Object>                map          = objectMapper.readValue(actualJson, typeRef);
104 
105             // Compare users
106             Map<String, User> referenceUsers = objectMapper.convertValue(referenceMap.get("users"), new TypeReference<Map<String, User>>() {
107             });
108             Map<String, User> users = objectMapper.convertValue(map.get("users"), new TypeReference<Map<String, User>>() {
109             });
110             assertEquals(referenceUsers.size(), users.size(), "Number of users differs");
111             for (String key : referenceUsers.keySet()) {
112                 assertTrue(users.containsKey(key), "Missing user: " + referenceUsers.get(key).getName());
113                 assertUserEquals(referenceUsers.get(key), users.get(key));
114             }
115 
116             // Compare sprints
117             Sprint referenceSprint = objectMapper.convertValue(referenceMap.get("sprint"), Sprint.class);
118             Sprint sprint          = objectMapper.convertValue(map.get("sprint"), Sprint.class);
119             assertSprintEquals(referenceSprint, sprint);
120 
121             // Compare tasks
122             List<Task> referenceTasks = objectMapper.convertValue(referenceMap.get("tasks"), new TypeReference<List<Task>>() {
123             });
124             List<Task> tasks = objectMapper.convertValue(map.get("tasks"), new TypeReference<List<Task>>() {
125             });
126             for (Task task : tasks)
127                 sprint.addTask(task);
128             for (Task task : referenceTasks)
129                 referenceSprint.addTask(task);
130             {
131                 GanttContext gc = new GanttContext();
132                 gc.allUsers   = new ArrayList<>(referenceUsers.values());
133                 gc.allSprints = List.of(referenceSprint);
134                 gc.allTasks   = referenceTasks;
135                 gc.initialize();
136             }
137             {
138                 GanttContext gc = new GanttContext();
139                 gc.allUsers   = new ArrayList<>(users.values());
140                 gc.allSprints = List.of(sprint);
141                 gc.allTasks   = tasks;
142                 gc.initialize();
143             }
144 
145             logProjectTasks(testResultFolder + "/" + TestInfoUtil.getTestMethodName(testInfo) + ".json", sprint, testReferenceResultFolder + "/" + TestInfoUtil.getTestMethodName(testInfo) + ".json", referenceSprint);
146             compareTasks(tasks, referenceTasks);
147         } catch (JsonProcessingException e) {
148             fail("Failed to parse JSON: " + e.getMessage());
149         }
150 
151     }

227     protected void generateProductsIfNeeded(TestInfo testInfo, RandomCase randomCase) throws Exception {
228         String testCaseName = this.getClass().getName() + "-" + testInfo.getTestMethod().get().getName() + "-" + randomCase.getTestCaseIndex();
229         // Create a snapshot name based on the test case
230         String snapshotName = testInfo.getTestClass().get().getSimpleName() + "-" + randomCase.getTestCaseIndex();
231         // Try to find and load an existing database snapshot
232         String  latestSnapshot = databaseStateManager.findLatestSnapshot(snapshotName);
233         boolean dataLoaded     = false;
234         if (latestSnapshot != null) {
235             logger.info("Found existing database snapshot: {}. Attempting to load...", latestSnapshot);
236             dataLoaded = databaseStateManager.importDatabaseSnapshot(latestSnapshot);
237         }
238         // If no snapshot was found or loading failed, generate data the regular way
239         if (!dataLoaded) {
240             logger.info("Generating fresh test data (this might take a few minutes)...");
241             generateProductsInternal(testInfo, randomCase);
242             // After successful data generation, export a snapshot for future test runs
243             databaseStateManager.exportDatabaseSnapshot(snapshotName);
244         } else {
245             logger.info("Successfully loaded test data from snapshot. Skipping data generation.");
246         }
247     }

279     private void generateSprint(TestInfo testInfo, RandomCase randomCase, Feature project) throws Exception {
280         int numberOfUsers = randomCase.getMaxNumberOfUsers();
281 //        System.out.println("Number of users=" + numberOfUsers);
282         int           numberOfStories = random.nextInt(randomCase.getMaxNumberOfStories()) + 1;
283         LocalDateTime startDateTime   = randomCase.getMinStartDate().plusDays(random.nextInt((int) randomCase.getMaxStartDateShift().toDays())).atStartOfDay().plusHours(8);
284         try (Profiler pc1 = new Profiler(SampleType.JPA)) {
285             Sprint savedSprint    = addRandomSprint(project);
286             Sprint sprint         = sprintApi.getById(savedSprint.getId());
287             Task   startMilestone = addTask(sprint, null, "Start", startDateTime, Duration.ZERO, null, null, TaskMode.MANUALLY_SCHEDULED, true);
288             for (int f = 0; f < numberOfStories; f++) {
289                 String storyName     = nameGenerator.generateStoryName(f);
290                 Task   story         = addParentTask(storyName, sprint, null, startMilestone);
291                 int    numberOfTasks = random.nextInt(randomCase.getMaxNumberOfTasks()) + 1;
292                 for (int t = 0; t < numberOfTasks; t++) {
293                     int userIndex = random.nextInt(numberOfUsers);
294 //                    System.out.println("User index=" + userIndex);
295                     User   user             = expectedUsers.stream().toList().get(userIndex);
296                     String duration         = String.format("%dh", (int) (random.nextFloat(randomCase.getMaxDurationDays() * 7.5f) + 1));
297                     String workName         = NameGenerator.generateWorkName(storyName, t);
298                     Task   depenedenycyTask = null;
299                     if (random.nextFloat(1) > 0.5f) {
300                         int tries = 8;
301                         do {
302                             depenedenycyTask = sprint.getTasks().get(random.nextInt(sprint.getTasks().size()));
303                             //make sure this task is not a parent of our parent and not a milestone
304                             if (depenedenycyTask.isMilestone() || depenedenycyTask.isAncestor(story)) {
305                                 depenedenycyTask = null;
306                                 tries--;
307                             }
308                         }
309                         while (depenedenycyTask == null && tries > 0);
310                     }
311                     addTask(workName, duration, user, sprint, story, depenedenycyTask);
312                 }
313             }
314             try (Profiler pc2 = new Profiler(SampleType.CPU)) {
315                 sprint.initialize();
316             }
317             sprint.initUserMap(userApi.getAll(sprint.getId()));
318             sprint.initTaskMap(taskApi.getAll(sprint.getId()), worklogApi.getAll(sprint.getId()));
319             try (Profiler pc3 = new Profiler(SampleType.CPU)) {
320                 levelResources(testInfo, sprint, null);
321             }
322             generateWorklogs(sprint, ParameterOptions.getLocalNow());
323         }
324         Profiler.log("generateProductsIfNeeded-" + randomCase.getTestCaseIndex());
325     }

327     protected String generateTestCaseName(TestInfo testInfo) {
328         String displayName = testInfo.getDisplayName();
329         String methodName  = TestInfoUtil.getTestMethodName(testInfo);
330         if (displayName.startsWith("[")) {
331             //parametrized test case
332             //[1] mppFileName=references\CREQ11793 Siemens OMS 2.0 Ph2 SMIME-rcp.mpp
333             String bullet = displayName.substring(0, displayName.indexOf(' '));
334             if (displayName.contains("mppFileName=")) {
335                 String mppFileName = displayName.substring(displayName.indexOf('\\') + 1, displayName.lastIndexOf("-rcp."));
336                 if (mppFileName.length() > 10) {
337                     mppFileName = mppFileName.substring(0, 7) + "...";
338                 }
339                 return bullet + " " + methodName + " (" + mppFileName + ")";
340             } else {
341                 return bullet + methodName;
342             }
343         } else {
344             return methodName;
345         }

355         try (Profiler pc = new Profiler(SampleType.CPU)) {
356 
357             final long SECONDS_PER_WORKING_DAY = 75 * 6 * 60;
358             final long SECONDS_PER_HOUR        = 60 * 60;
359             long       oneDay                  = 75 * SECONDS_PER_HOUR / 10;
360             Duration   rest                    = Duration.ofSeconds(1);
361             //- iterate over the days of the sprint
362             for (LocalDate day = sprint.getStart().toLocalDate(); !rest.equals(Duration.ZERO) && now.toLocalDate().isAfter(day); day = day.plusDays(1)) {
363                 LocalDateTime startOfDay     = day.atStartOfDay().plusHours(8);
364                 LocalDateTime endOfDay       = day.atStartOfDay().plusHours(16).plusMinutes(30);
365                 LocalDateTime lunchStartTime = DateUtil.calculateLunchStartTime(day.atStartOfDay());
366                 LocalDateTime lunchStopTime  = DateUtil.calculateLunchStopTime(day.atStartOfDay());
367                 rest = Duration.ZERO;
368                 for (Task task : sprint.getTasks()) {
369                     if (task.getChildTasks().isEmpty() && task.getOriginalEstimate() != null && !task.getOriginalEstimate().isZero()) {
370                         Number availability = task.getAssignedUser().getAvailabilities().getLast().getAvailability();
371                         if (task.getChildTasks().isEmpty()) {
372                             if (!day.isBefore(task.getStart().toLocalDate()) /*&& !day.isAfter(task.getFinish().toLocalDate())*/) {

374

377                                         if (!task.getRemainingEstimate().isZero()) {
378                                             // we have the whole day
379                                             double   minPerformance = 0.6f;

385                                             } else {
386                                                 w = task.getRemainingEstimate();
387                                             }

390                                             task.removeRemainingEstimate(w);
391                                             task.recalculate();
392                                         }
393                                     }

395                             }
396                         }
397                     }
398                     rest = rest.plus(task.getRemainingEstimate());//accumulate the rest

474     private void logTask(Task task, Task referenceTask, int maxNameLength) {
475         String   buffer         = "";
476         String   criticalString = task.isCritical() ? "Y" : "N";
477         String   startString    = DateUtil.createDateString(task.getStart(), dtfymdhmss);
478         String   finishString   = DateUtil.createDateString(task.getFinish(), dtfymdhmss);
479         String   durationString = null;
480         Duration duration       = task.getDuration();
481         if (duration != null) {
482             //            int minutes = (int) ((duration.getDuration() * 7.5 * 60 * 60) / 60);
483             //            double seconds = (duration.getDuration() * 7.5 * 60 * 60 - minutes * 60);
484             durationString = DateUtil.createDurationString(duration, true, true, true);
485         }
486         Duration referenceDuration = null;
487         if (referenceTask != null) {
488             referenceDuration = referenceTask.getDuration();
489             //            int minutes = (int) ((duration.getDuration() * 7.5 * 60 * 60) / 60);
490             //            double seconds = (duration.getDuration() * 7.5 * 60 * 60 - minutes * 60);
491 //            String referenceDurationString = DateUtil.createDurationString(referenceDuration, true, true, true);
492         }
493         String          criticalFlag = ANSI_GREEN;
494         String          startFlag    = ANSI_GREEN;
495         String          finishFlag   = ANSI_GREEN;
496         String          durationFlag = ANSI_GREEN;
497         ProjectCalendar calendar     = GanttUtil.getCalendar(task);
498         if (referenceTask != null) {
499             if (task.getChildTasks().isEmpty() && task.isCritical() != referenceTask.isCritical()) {
500                 criticalFlag = ANSI_RED;
501             }
502             if (task.getStart() == null) {
503                 startFlag = ANSI_RED;
504             } else if (!GanttUtil.equals(calendar, task.getStart(), referenceTask.getStart())) {
505                 startFlag = ANSI_RED;
506             } else if (!task.getStart().equals(referenceTask.getStart())) {
507                 startFlag = ANSI_YELLOW;
508             }
509             if (task.getFinish() == null) {
510                 finishFlag = ANSI_RED;
511             } else if (!GanttUtil.equals(calendar, task.getFinish(), referenceTask.getFinish())) {
512                 finishFlag = ANSI_RED;
513 
514             } else if (!task.getFinish().equals(referenceTask.getFinish())) {
515                 finishFlag = ANSI_YELLOW;
516             }
517             if (task.getDuration() == null) {
518                 durationFlag = ANSI_RED;
519             } else if (!GanttUtil.equals(task.getDuration(), referenceTask.getDuration())) {
520                 durationFlag = ANSI_RED;
521             }
522 
523         }
524         buffer += String.format("[%2d] N='%-" + maxNameLength + "s' C=%s%s%s S='%s%20s%s' D='%s%-19s%s' F='%s%20s%s'", task.getId(),//
525                 task.getName(),//
526                 criticalFlag, criticalString, ANSI_RESET,//
527                 startFlag, startString, ANSI_RESET,//
528                 durationFlag, durationString, ANSI_RESET,//
529                 finishFlag, finishString, ANSI_RESET);
530         logger.trace(buffer);
531     }
