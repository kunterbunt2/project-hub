138     // Configure Spring Security to use the Keycloak container
139     @DynamicPropertySource
140     static void registerKeycloakProperties(DynamicPropertyRegistry registry) {
141         // Start the container
142         keycloak.start();
143 
144         // Get the actual URL that's accessible from outside the container
145         String externalUrl = getPublicFacingUrl(keycloak);
146         System.out.println("Keycloak External URL: " + externalUrl);
147 
148         // Log all container environment information for debugging
149         System.out.println("Keycloak Container:");
150         System.out.println("  Auth Server URL: " + keycloak.getAuthServerUrl());
151         System.out.println("  Container IP: " + keycloak.getHost());
152         System.out.println("  HTTP Port Mapping: " + keycloak.getMappedPort(8080));
153         System.out.println("  HTTPS Port Mapping: " + keycloak.getMappedPort(8443));
154 
155         // Override the authServerUrl with our public-facing URL
156         String publicAuthServerUrl = externalUrl + "/";
157 
158         // Create properties with the public URL
159         Map<String, String> props = new HashMap<>();
160         props.put("spring.security.oauth2.client.provider.keycloak.issuer-uri", publicAuthServerUrl + "realms/project-hub-realm");
161         props.put("spring.security.oauth2.client.provider.keycloak.authorization-uri", publicAuthServerUrl + "realms/project-hub-realm/protocol/openid-connect/auth");
162         props.put("spring.security.oauth2.client.provider.keycloak.token-uri", publicAuthServerUrl + "realms/project-hub-realm/protocol/openid-connect/token");
163         props.put("spring.security.oauth2.client.provider.keycloak.user-info-uri", publicAuthServerUrl + "realms/project-hub-realm/protocol/openid-connect/userinfo");
164         props.put("spring.security.oauth2.client.provider.keycloak.jwk-set-uri", publicAuthServerUrl + "realms/project-hub-realm/protocol/openid-connect/certs");
165 
166         props.put("spring.security.oauth2.client.registration.keycloak.client-id", "project-hub-client");
167         props.put("spring.security.oauth2.client.registration.keycloak.client-secret", "test-client-secret");
168         props.put("spring.security.oauth2.client.registration.keycloak.scope", "openid,profile,email");
169         props.put("spring.security.oauth2.client.registration.keycloak.authorization-grant-type", "authorization_code");
170         props.put("spring.security.oauth2.client.registration.keycloak.redirect-uri", "{baseUrl}/login/oauth2/code/{registrationId}");
171 
172         props.put("spring.security.oauth2.resourceserver.jwt.issuer-uri", publicAuthServerUrl + "realms/project-hub-realm");
173 
174         // Register all properties
175         props.forEach((key, value) -> registry.add(key, () -> value));
176     }
