99       * <p>
100      * This method is useful for verifying that a specific element appears exactly the expected
101      * number of times in a grid, such as checking that no duplicate products were created.
102      *
103      * @param gridId         the ID of the grid WebElement to check
104      * @param gridNamePrefix the prefix used in the element IDs within the grid
105      * @param name           the name suffix to search for in the grid element IDs
106      * @param expectedCount  the expected number of occurrences of the element in the grid
107      */

109         {
110             waitForElementToBeLocated(gridId);
111             WebElement grid = getDriver().findElement(By.id(gridId));
112             // Wait a moment for any UI updates to complete
113             wait(500);
114             // Check for the div containing the product name
115             List<WebElement> matchingElements = grid.findElements(By.cssSelector("[id^='" + gridNamePrefix + name + "']"));
116             assertEquals(expectedCount, matchingElements.size(), "Expected 1 element with name '" + name + "' but found " + matchingElements.size());
117         }
118     }

197      * Gets the current value from a color picker component.
198      * <p>
199      * This method retrieves the color value from a Vaadin ColorPicker component
200      * using JavaScript. It accesses the component's shadow DOM to get the color
201      * input element's value. Returns the hex color code with # prefix.
202      *
203      * @param colorPickerId the ID of the color picker component
204      * @return the current color value as a hex string (e.g. "#FF5733"), or null if not available
205      */
206     public String getColorPickerValue(String colorPickerId) {
207         // Wait for the color picker element to be present
208         waitUntil(ExpectedConditions.presenceOfElementLocated(By.id(colorPickerId)));
209 
210         // Access the color input element through the shadow DOM, similar to the set method
211         String colorScript =//
212                 "var picker = document.getElementById('" + colorPickerId + "');" +//
213                         "if (picker) {" +//
214                         "  var inputElement = picker.shadowRoot.querySelector('input[type=\"color\"]');" +//
215                         "  if (inputElement) {" +//
216                         "    return inputElement.value;" +//
217                         "  }" +//
218                         "  return picker.value;" + // Fallback to component's value if input not found
219                         "}" +//
220                         "return null;";
221 
222         return (String) executeJavaScript(colorScript);
223     }

260      * Gets the current value from a date picker component.
261      * <p>
262      * This method retrieves the date value from a Vaadin DatePicker component
263      * using JavaScript and converts it to a LocalDate object.
264      *
265      * @param datePickerId the ID of the date picker component
266      * @return the current date value as LocalDate, or null if no date is set or component not found

268     public LocalDate getDatePickerValue(String datePickerId) {
269         // Wait for the date picker element to be present
270         waitUntil(ExpectedConditions.presenceOfElementLocated(By.id(datePickerId)));
271 
272         String dateScript =//
273                 "var datePicker = document.getElementById('" + datePickerId + "');" +//
274                         "if (datePicker && datePicker.value) {" +//
275                         "  return datePicker.value;" +//

287     public WebDriver getDriver() {
288         if (driver != null) {
289             return driver;
290         }
291 
292         WebDriverManager.chromedriver().setup();
293 
294         ChromeOptions options = new ChromeOptions();
295 
296         // Check if we're running in headless mode (for CI environment)
297         boolean headlessMode = isSeleniumHeadless();
298         if (headlessMode) {
299             logger.info("creating selenium driver, Running Chrome in headless mode");
300             options.addArguments("--headless=new");
301             options.addArguments("--disable-gpu");
302             options.addArguments("--no-sandbox");
303             options.addArguments("--disable-dev-shm-usage");
304             options.addArguments("--window-size=1920,1080");
305             options.addArguments("--disable-extensions");
306             options.addArguments("--disable-browser-side-navigation");
307             options.addArguments("--disable-web-security");
308             options.addArguments("--dns-prefetch-disable");
309             // Add a longer timeout for the page load
310             options.setPageLoadTimeout(Duration.ofSeconds(60));
311             // Disable the "Save password?" prompt
312             options.setExperimentalOption("prefs", Map.of(
313                     "credentials_enable_service", false,
314                     "profile.password_manager_enabled", false
315             ));
316         } else {
317             logger.info("creating selenium driver");
318         }
319 
320         options.addArguments("--remote-allow-origins=*");
321         options.setCapability(CapabilityType.ACCEPT_INSECURE_CERTS, true);
322 
323         // Set a higher script timeout to prevent connection issues
324         driver = new ChromeDriver(options);
325         driver.manage().timeouts().scriptTimeout(Duration.ofSeconds(30));
326         setImplicitWaitDuration(implicitWaitDuration);
327         wait = new WebDriverWait(driver, waitDuration);
328 
329         // Set default window size if not in headless mode
330         if (!headlessMode) {
331             if (windowSize != null) {
332                 getDriver().manage().window().setSize(windowSize);
333                 getDriver().manage().window().setPosition(new Point(33, 22));
334             } else {
335                 //maximize window by default
336                 getDriver().manage().window().maximize();
337             }
338         }
339         return driver;
340     }

342     /**
343      * Gets the error message from a Vaadin text field element by accessing the error-message slot.
344      * <p>
345      * This method retrieves the error message that appears below a form field when validation fails.
346      * It waits for the field to become invalid and for an error message to appear before returning it.
347      * It uses JavaScript to access the shadow DOM and find the error message slot content.
348      *
349      * @param fieldId the ID of the field element
350      * @return the error message text, or null if no error message is present after waiting
351      */
352     public String getFieldErrorMessage(String fieldId) {
353         pushWaitDuration(Duration.ofSeconds(3));
354         waitUntil(ExpectedConditions.presenceOfElementLocated(By.id(fieldId)));
355 
356         // Wait for the field to become invalid (with a timeout)
357         try {
358             // Use waitUntil with a custom condition to wait for field to become invalid and have an error message
359             waitUntil(driver -> {
360                 Boolean isInvalid = (Boolean) executeJavaScript(
361                         "var field = document.getElementById('" + fieldId + "');" +//
362                                 "return field && field.invalid;");
363 
364                 if (Boolean.TRUE.equals(isInvalid)) {
365                     // Now check if the error message element exists
366                     String errorMsg = (String) executeJavaScript(//
367                             "var field = document.getElementById('" + fieldId + "');" +//
368                                     "var errorElement = field.querySelector('[slot=\"error-message\"]');" +//
369                                     "return errorElement ? errorElement.textContent : null;");
370 
371                     return errorMsg != null && !errorMsg.isEmpty();
372                 }
373                 return false;
374             });
375         } catch (org.openqa.selenium.TimeoutException e) {
376             logger.warn("Timed out waiting for error message on field: {}", fieldId);
377             // We'll continue and try to get whatever message might be there
378         }
379         popWaitDuration();
380         // Now that we've waited, get the actual error message
381         String script =
382                 "var field = document.getElementById('" + fieldId + "');" +//
383                         "if (field && field.invalid) {" +//
384                         "  var errorElement = field.querySelector('[slot=\"error-message\"]');" +//
385                         "  return errorElement ? errorElement.textContent : null;" +//
386                         "}" +//
387                         "return null;";
388 
389         return (String) executeJavaScript(script);
390     }

531     public void selectGridRow(String gridRowBaseId, Class<?> viewClass, String rowName) {
532         String  url             = getRouteValue(viewClass);
533         boolean outerTesting    = true;
534         int     outerIterations = 12;//5 seconds
535         //try several times
536         do {
537             WebElement row = findElement(By.id(gridRowBaseId + rowName));
538             try {
539                 row.click();
540                 logger.info("Clicked row: " + gridRowBaseId + rowName);
541             } catch (StaleElementReferenceException e) {
542                 //ignore and retry
543             }
544             //check several times
545             int     innerIterations = 10;//5 seconds
546             boolean innerTesting    = true;
547             do {
548                 if (ExpectedConditions.urlContains(url).apply(getDriver())) {
549                     outerTesting = false;
550                 }
551                 if (--innerIterations == 0) {
552                     innerTesting = false;
553                 }
554                 if (outerTesting) {
555                     try {
556                         Thread.sleep(100);
557                     } catch (InterruptedException e) {
558                         //ignore
559                     }
560                 }
561             } while (innerTesting && outerTesting);
562             if (--outerIterations == 0) {
563                 outerTesting = false;
564             }
565             if (outerTesting) {
566                 try {
567                     Thread.sleep(50);
568                 } catch (InterruptedException e) {
569                     //ignore
570                 }
571             }
572         } while (outerTesting);
573     }

642     /**
643      * Sets a value to a color picker component.
644      * <p>
645      * This method provides a reliable way to set color values on Vaadin ColorPicker
646      * components. It uses JavaScript to set the value directly and ensures the component
647      * properly registers the color change by triggering necessary events.
648      *
649      * @param colorPickerId the ID of the color picker component
650      * @param colorValue    the color value to set (hex format with # prefix, e.g., "#FF5733")
651      * @return true if the color was successfully set, false otherwise
652      */
653     public boolean setColorPickerValue(String colorPickerId, String colorValue) {
654         // Wait for the color picker element to be present
655         waitUntil(ExpectedConditions.presenceOfElementLocated(By.id(colorPickerId)));
656 
657         // First attempt: set the value and trigger necessary events
658         String colorScript =//
659                 "var picker = document.getElementById('" + colorPickerId + "');" +//
660                         "if (picker) {" +//
661                         "  picker.value = '" + colorValue + "';" +//
662                         "  if (typeof picker._valueChanged === 'function') {" +//
663                         "    picker._valueChanged('" + colorValue + "');" +//
664                         "  }" +//
665                         "  var inputElement = picker.shadowRoot.querySelector('input[type=\"color\"]');" +//
666                         "  if (inputElement) {" +//
667                         "    inputElement.value = '" + colorValue + "';" +//
668                         "    inputElement.dispatchEvent(new Event('input', {bubbles: true}));" +//
669                         "    inputElement.dispatchEvent(new Event('change', {bubbles: true}));" +//
670                         "  }" +//
671                         "  return picker.value === '" + colorValue + "';" +//
672                         "}" +//

931     /**
932      * Wait for an element to become interactable (enabled and visible)
933      * This is useful for elements like ComboBoxes that may be disabled initially
934      * and only become enabled after other UI actions.
935      *
936      * @param id The ID of the element to wait for
937      */
938     public void waitForElementToBeInteractable(String id) {
939         try {
940             // Wait for the element to be visible first
941             waitUntil(ExpectedConditions.visibilityOfElementLocated(By.id(id)));
942 
943             // Wait for the element to not have the 'disabled' attribute
944             waitUntil(driver -> {
945                 WebElement element = driver.findElement(By.id(id));
946                 return element.getAttribute("disabled") == null &&
947                         !"true".equals(element.getAttribute("aria-disabled"));
948             });
949 
950             // Finally, wait for it to be clickable
951             waitUntil(ExpectedConditions.elementToBeClickable(By.id(id)));
952 
953             // Small additional delay to ensure UI is fully ready
954             try {
955                 Thread.sleep(300);
956             } catch (InterruptedException e) {
957                 Thread.currentThread().interrupt();
958             }
959         } catch (Exception e) {
960             logger.error("Error waiting for element to be interactable: " + id, e);
961             throw e;
962         }
963     }

1028      * Waits until the browser window is closed either manually by the user
1029      * or by another process.
1030      *
1031      * @param timeoutMillis Maximum time to wait in milliseconds, or 0 for infinite wait
1032      * @return true if browser was closed, false if timeout occurred
1033      */
1034     public boolean waitUntilBrowserClosed(long timeoutMillis) {
1035         long startTime = System.currentTimeMillis();
1036 
1037         while (true) {
1038             try {
1039                 // Try to execute a simple script or get window handles to verify browser is open
1040                 getDriver().getWindowHandles();
1041                 // Short sleep to prevent high CPU usage
1042                 try {
1043                     Thread.sleep(500);
1044                 } catch (InterruptedException e) {
1045                     Thread.currentThread().interrupt();
1046                     if (videoRecorder.isRecording()) {
1047                         stopRecording();
1048                     }
1049                     driver.quit();//quit the driver
1050                     driver = null;
1051                     return false;
1052                 }
1053 
1054                 // Check for timeout if specified
1055                 if (timeoutMillis > 0 && System.currentTimeMillis() - startTime > timeoutMillis) {
1056                     if (videoRecorder.isRecording()) {
1057                         stopRecording();
1058                     }
1059                     driver.quit();//quit the driver
1060                     driver = null;
1061                     return false;
1062                 }
1063             } catch (WebDriverException e) {
1064                 if (e instanceof NoSuchSessionException) {
1065                     // Browser has been closed if we get a NoSuchSessionException
1066                 } else {
1067                     throw e;
1068                 }
1069                 if (videoRecorder.isRecording()) {
1070                     stopRecording();
1071                 }
1072                 driver.quit();//quit the driver
1073                 driver = null;
1074                 return true;
1075             }
1076         }
