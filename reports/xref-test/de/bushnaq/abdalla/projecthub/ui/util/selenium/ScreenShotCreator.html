30  public class ScreenShotCreator {
31      private static final Logger logger = LoggerFactory.getLogger(ScreenShotCreator.class);
32  
33      static String fileNameGenerator(String displayName, String testmethod) {
34          displayName = displayName.replace("'", "");
35          String path = getScreenshotFolder();
36          return path + testmethod + "(" + displayName + ")" + ".png";
37      }
38  
39      private static String getScreenshotFolder() {
40          return "target" + File.separator + "externalFiles" + File.separator + "screenshots" + File.separator;
41      }
42  
43      /**
44       * Takes a screenshot of a specific element identified by its ID
45       *
46       * @param driver
47       * @param overlayElement The element to capture
48       * @param fileName       The filename where the screenshot should be saved
49       */
50      public static void takeElementScreenshot(WebDriver driver, WebElement overlayElement, String dialogId, String fileName) {
51          try {
52              if (overlayElement != null) {
53  
54                  // Take screenshot of the entire page
55                  wait(1000);
56                  File screenshot = ((TakesScreenshot) driver).getScreenshotAs(OutputType.FILE);
57                  wait(1000);
58                  // Read the screenshot file into a BufferedImage
59                  BufferedImage fullImg = ImageIO.read(screenshot);
60  
61                  // Get the location and size of the element
62                  org.openqa.selenium.Point     location = overlayElement.getLocation();
63                  org.openqa.selenium.Dimension size     = overlayElement.getSize();
64  
65                  logger.info("Taking screenshot of dialog overlay: {} at x: {}, y: {}, width: {}, height: {}", dialogId, location.x, location.y, size.getWidth(), size.getHeight());
66                  // Make sure width and height are positive
67                  if (size.getWidth() <= 0 || size.getHeight() <= 0) {
68                      logger.warn("Dialog overlay has invalid dimensions: {}x{}", size.getWidth(), size.getHeight());
69                      // Fallback to taking a full screenshot
70                      takeScreenShot(driver, fileName);
71                      return;
72                  }
73                  // Crop the image to only include the dialog
74                  BufferedImage elementImg = fullImg.getSubimage(
75                          location.getX(),
76                          location.getY(),
77                          size.getWidth(),
78                          size.getHeight()
79                  );
80  
81                  // Ensure the directory exists
82                  File file = new File(fileName);
83                  File dir  = file.getParentFile();
84                  if (dir != null && !dir.exists()) {
85                      dir.mkdirs();
86                  }
87  
88                  // Save the cropped image
89                  ImageIO.write(elementImg, "png", file);
90                  logger.info("Dialog screenshot saved to: {}", fileName);
91              } else {
92                  logger.warn("Could not find overlay element for dialog: {}", dialogId);
93                  takeScreenShot(driver, fileName);
94              }
95          } catch (Exception e) {
96              logger.error("Failed to capture dialog screenshot", e);
97              // Fallback to taking a full screenshot
98              try {
99                  takeScreenShot(driver, fileName);
100             } catch (Exception ex) {
101                 logger.error("Failed to take fallback screenshot", ex);
102             }
103         }
104     }
105 
106     public static void takeScreenShot(WebDriver driver, String fileName) {
107         try {
108             File screenshotFile = ((TakesScreenshot) driver).getScreenshotAs(OutputType.FILE);
109             FileUtils.copyFile(screenshotFile, new File(fileName));
110             logger.info("Screenshot saved to: {}", fileName);
111         } catch (IOException | NoSuchSessionException e) {
112             throw new RuntimeException("Could not make screenshot {}" + fileName, e);
113         }
114     }

127     static void wait(int milliseconds) {
128         try {
129             Thread.sleep(milliseconds);
130         } catch (InterruptedException e) {
131             Thread.currentThread().interrupt();
132             logger.error("Thread interrupted while waiting", e);
133         }
134     }
