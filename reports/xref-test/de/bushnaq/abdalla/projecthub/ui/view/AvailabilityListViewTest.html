18  package de.bushnaq.abdalla.projecthub.ui.view;
19  
20  import de.bushnaq.abdalla.projecthub.ui.util.AbstractUiTestUtil;
21  import de.bushnaq.abdalla.projecthub.ui.util.selenium.SeleniumHandler;
22  import de.bushnaq.abdalla.projecthub.ui.view.util.AvailabilityListViewTester;
23  import org.junit.jupiter.api.BeforeEach;
24  import org.junit.jupiter.api.Test;
25  import org.junit.jupiter.api.TestInfo;
26  import org.junit.jupiter.api.extension.ExtendWith;
27  import org.springframework.beans.factory.annotation.Autowired;
28  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
29  import org.springframework.boot.test.context.SpringBootTest;
30  import org.springframework.test.context.junit.jupiter.SpringExtension;
31  import org.springframework.transaction.annotation.Transactional;
32  
33  import java.time.LocalDate;
34  
35  /**
36   * Integration test for the AvailabilityListView UI component.
37   * Tests CRUD (Create, Read, Update, Delete) operations for availability records in the UI.
38   * <p>
39   * These tests verify that:
40   * - Availability records can be created with appropriate details
41   * - Created records appear correctly in the list
42   * - Records can be edited and changes are reflected in the UI
43   * - Records can be deleted from the system
44   * - Cancellation of operations works as expected
45   * - Validation rules are enforced (unique start dates, valid percentage range)
46   * <p>
47   * The tests account for the fact that each user already has an initial availability record
48   * for the current date when they're created.
49   * <p>
50   * The tests use {@link AvailabilityListViewTester} to interact with the UI elements
51   * and verify the expected behavior.
52   */
53  @ExtendWith(SpringExtension.class)
54  @SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
55  @AutoConfigureMockMvc
56  @Transactional
57  public class AvailabilityListViewTest extends AbstractUiTestUtil {
58      @Autowired
59      private       AvailabilityListViewTester availabilityListViewTester;
60      private final int                        availabilityPercent    = 80;
61      // Initial availability record that exists when the user is created (today's date)
62      private final LocalDate                  initialDate            = LocalDate.now();
63      private final int                        initialPercent         = 100;
64      // Invalid test data for validation tests
65      private final int                        invalidHighPercent     = 160; // Above 150% limit
66      private final int                        invalidLowPercent      = -10; // Below 0% limit
67      private final int                        newAvailabilityPercent = 50;
68      // Test data for updating availability records
69      private final LocalDate                  newStartDate           = LocalDate.of(2025, 8, 1);
70      private final int                        secondAvailPercent     = 75;
71      // Test data for the second availability record (for create operations)
72      private final LocalDate                  secondStartDate        = LocalDate.of(2025, 7, 1);
73      @Autowired
74      private       SeleniumHandler            seleniumHandler;
75      // Test data for the first new availability record
76      private final LocalDate                  startDate              = LocalDate.of(2025, 6, 1);
77      private final String                     testUsername           = "availability-test-user";
78  
79      @BeforeEach
80      public void setupTest(TestInfo testInfo) {
81          availabilityListViewTester.switchToAvailabilityListView(
82                  testInfo.getTestClass().get().getSimpleName(),
83                  generateTestCaseName(testInfo),
84                  testUsername);
85      }
86  
87      /**
88       * Tests that users cannot delete their only availability record.
89       * <p>
90       * Verifies that the delete button for the initial availability record
91       * is disabled, preventing users from deleting their only record.
92       */
93      @Test
94      public void testCannotDeleteOnlyAvailability() {
95          // Verify user cannot delete their only availability record (the initial one)
96          availabilityListViewTester.verifyCannotDeleteOnlyAvailability(initialDate);
97      }
98  
99      /**
100      * Tests the behavior when creating an availability record but canceling the operation.
101      * <p>
102      * Verifies that when a user clicks the create availability button, enters data, and then
103      * cancels the operation, no record is created in the list.
104      */
105     @Test
106     public void testCreateCancel() {
107         availabilityListViewTester.createAvailabilityCancel(startDate, availabilityPercent);
108     }
109 
110     /**
111      * Tests the behavior when successfully creating an availability record.
112      * <p>
113      * Verifies that when a user clicks the create availability button, enters all required fields,
114      * and confirms the creation, the record appears in the list with the correct values.
115      */
116     @Test
117     public void testCreateConfirm() {
118         availabilityListViewTester.createAvailabilityConfirm(startDate, availabilityPercent);
119     }
120 
121     /**
122      * Tests the behavior when attempting to delete an availability record but canceling the operation.
123      * <p>
124      * Creates a record, then attempts to delete it but cancels the confirmation dialog.
125      * Verifies that the record remains in the list.
126      * <p>
127      * Note: This test assumes the user already has the initial default availability record.
128      */
129     @Test
130     public void testDeleteCancel() {
131         // Create a second availability record
132         availabilityListViewTester.createAvailabilityConfirm(startDate, availabilityPercent);
133         // Try to delete but cancel
134         availabilityListViewTester.deleteAvailabilityCancel(startDate);
135     }
136 
137     /**
138      * Tests the behavior when successfully deleting an availability record.
139      * <p>
140      * Creates a record, then deletes it by confirming the deletion in the confirmation dialog.
141      * Verifies that the record is removed from the list.
142      * <p>
143      * Note: This test assumes the user already has the initial default availability record.
144      */
145     @Test
146     public void testDeleteConfirm() {
147         // Create a second availability record
148         availabilityListViewTester.createAvailabilityConfirm(startDate, availabilityPercent);
149         // Delete the newly created record
150         availabilityListViewTester.deleteAvailabilityConfirm(startDate);
151     }
152 
153     /**
154      * Tests that validation prevents creation of duplicate availability records with the same start date.
155      * <p>
156      * Attempts to create a record with the same start date as the initial record but different
157      * availability percentage. Verifies that an error is shown and the duplicate is not created.
158      */
159     @Test
160     public void testDuplicateStartDate() {
161         // Try to create a duplicate with the same start date as the initial record
162         availabilityListViewTester.createDuplicateDateAvailability(initialDate, newAvailabilityPercent);
163     }
164 
165     /**
166      * Tests the behavior when attempting to edit an availability record but canceling the operation.
167      * <p>
168      * Creates a record, attempts to edit all its fields (start date, availability percentage),
169      * but cancels the edit dialog.
170      * Verifies that the original record details remain unchanged and the new values are not applied.
171      * <p>
172      * Note: This test uses the initial availability record that exists when the user is created.
173      */
174     @Test
175     public void testEditCancel() {
176         // Edit initial record but cancel
177         availabilityListViewTester.editAvailabilityCancel(initialDate, newStartDate, initialPercent, newAvailabilityPercent);
178     }
179 
180     /**
181      * Tests the behavior when successfully editing an availability record.
182      * <p>
183      * Edits all fields of the initial availability record and confirms the edit.
184      * Verifies that the record with the new values appears in the list
185      * and the old values are no longer present.
186      */
187     @Test
188     public void testEditConfirm() {
189         // Edit initial record and confirm
190         availabilityListViewTester.editAvailabilityConfirm(initialDate, newStartDate, newAvailabilityPercent);
191     }
192 
193     /**
194      * Tests that validation prevents creation of availability records with invalid percentage values.
195      * <p>
196      * Attempts to create records with percentage values outside the allowed range (0% to 150%).
197      * Verifies that errors are shown and the invalid records are not created.
198      */
199     @Test
200     public void testInvalidPercentage() {
201         // Try to create record with too high percentage
202         availabilityListViewTester.createInvalidPercentageAvailability(startDate, invalidHighPercent);
203 
204         // Try to create record with negative percentage
205         availabilityListViewTester.createInvalidPercentageAvailability(newStartDate, invalidLowPercent);
206     }
207 }
