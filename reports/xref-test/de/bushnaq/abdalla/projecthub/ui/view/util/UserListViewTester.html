35  /**
36   * Test helper class for interacting with the User UI components.
37   * <p>
38   * This class provides methods to test user-related operations in the UI such as

48      /**

52       * @param port            the port on which the application server is running

54      public UserListViewTester(SeleniumHandler seleniumHandler, @Value("${local.server.port:8080}") int port) {
55          this.seleniumHandler = seleniumHandler;
56          this.port            = port;

59      /**
60       * Tests the creation of a user where the user cancels the operation.
61       * <p>
62       * Opens the user creation dialog, enters values for all user fields, then cancels
63       * the dialog. Verifies that no user with the specified name appears in the user list
64       * and that the cancellation didn't affect any existing data.
65       *
66       * @param name            the name of the user to attempt to create
67       * @param email           the email of the user to attempt to create
68       * @param color           the hex color code (with # prefix) for the user
69       * @param firstWorkingDay the first working day for the user
70       * @param lastWorkingDay  the last working day for the user
71       */
72      public void createUserCancel(String name, String email, String color, LocalDate firstWorkingDay, LocalDate lastWorkingDay) {
73          seleniumHandler.click(UserListView.CREATE_USER_BUTTON);
74          seleniumHandler.setTextField(UserDialog.USER_NAME_FIELD, name);
75          seleniumHandler.setTextField(UserDialog.USER_EMAIL_FIELD, email);
76          seleniumHandler.setColorPickerValue(UserDialog.USER_COLOR_PICKER, color);
77          seleniumHandler.setDatePickerValue(UserDialog.USER_FIRST_WORKING_DAY_PICKER, firstWorkingDay);
78          seleniumHandler.setDatePickerValue(UserDialog.USER_LAST_WORKING_DAY_PICKER, lastWorkingDay);
79          seleniumHandler.click(UserDialog.CANCEL_BUTTON);
80          seleniumHandler.ensureIsNotInList(UserListView.USER_GRID_NAME_PREFIX, name);
81      }

143     /**
144      * Tests user editing where the user cancels the edit operation.
145      * <p>
146      * Clicks the edit button for the specified user,
147      * enters new values for all fields, then cancels the edit dialog. Verifies that the user
148      * still exists with its original originalName and no user with the new originalName exists.
149      *
150      * @param originalName     the original originalName of the user to edit
151      * @param newName          the new originalName to attempt to assign to the user
152      * @param originalEmail    the original email to verify remains unchanged after cancellation
153      * @param newEmail         the email to attempt to assign to the user
154      * @param originalColor    the original color to verify remains unchanged after cancellation
155      * @param newColor         the hex color code to attempt to assign to the user
156      * @param originalFirstDay the original first working day to verify remains unchanged after cancellation
157      * @param newFirstDay      the first working day to attempt to assign to the user
158      * @param originalLastDay  the original last working day to verify remains unchanged after cancellation
159      * @param newLastDay       the last working day to attempt to assign to the user
160      */
161     public void editUserCancel(String originalName, String newName,
162                                String originalEmail, String newEmail,
163                                String originalColor, String newColor,
164                                LocalDate originalFirstDay, LocalDate newFirstDay,
165                                LocalDate originalLastDay, LocalDate newLastDay) {
166         // First verify the current field values before starting the edit
167         verifyUserDialogFields(originalName, originalEmail, originalColor, originalFirstDay, originalLastDay);
168 
169         seleniumHandler.click(UserListView.USER_GRID_EDIT_BUTTON_PREFIX + originalName);
170         seleniumHandler.setTextField(UserDialog.USER_NAME_FIELD, newName);
171         seleniumHandler.setTextField(UserDialog.USER_EMAIL_FIELD, newEmail);
172         seleniumHandler.setColorPickerValue(UserDialog.USER_COLOR_PICKER, newColor);
173         seleniumHandler.setDatePickerValue(UserDialog.USER_FIRST_WORKING_DAY_PICKER, newFirstDay);
174         seleniumHandler.setDatePickerValue(UserDialog.USER_LAST_WORKING_DAY_PICKER, newLastDay);
175 
176         // Short pause to ensure UI updates
177         try {
178             Thread.sleep(300);
179         } catch (InterruptedException e) {
180             Thread.currentThread().interrupt();
181         }
182         seleniumHandler.click(UserDialog.CANCEL_BUTTON);
183         seleniumHandler.ensureIsInList(UserListView.USER_GRID_NAME_PREFIX, originalName);
184         seleniumHandler.ensureIsNotInList(UserListView.USER_GRID_NAME_PREFIX, newName);
185     }

214     /**
215      * Navigates to the UserListView.
216      * <p>
217      * Opens the user list URL directly and waits for the page to load
218      * by checking for the presence of the page title element.
219      */
220     public void switchToUserListView(String recordingFolderName, String testName) {
221         //- Check if we need to log in
222         if (!seleniumHandler.getCurrentUrl().contains("/ui/")) {
223             seleniumHandler.getAndCheck("http://localhost:" + port + "/ui/" + LoginView.ROUTE);
224             seleniumHandler.startRecording(recordingFolderName, testName);
225             seleniumHandler.setLoginUser("admin-user");
226             seleniumHandler.setLoginPassword("test-password");
227             seleniumHandler.loginSubmit();
228             seleniumHandler.waitUntil(ExpectedConditions.elementToBeClickable(By.id(ProductListView.PRODUCT_LIST_PAGE_TITLE)));
229         }
230         seleniumHandler.getAndCheck("http://localhost:" + port + "/ui/" + UserListView.ROUTE);
231         seleniumHandler.waitUntil(ExpectedConditions.elementToBeClickable(By.id(UserListView.USER_LIST_PAGE_TITLE)));
232     }

234     /**
235      * Verifies that the form is reset after cancellation.
236      * <p>
237      * Opens a new user dialog and verifies that it doesn't contain any values from
238      * a previous cancelled operation. This method is used to ensure that cancelling
239      * a user creation or edit truly discards all changes.
240      */
241     public void verifyFormIsReset() {
242         // Open a new user dialog
243         seleniumHandler.click(UserListView.CREATE_USER_BUTTON);
244         // Read form values
245         String    formName     = seleniumHandler.getTextField(UserDialog.USER_NAME_FIELD);
246         String    formEmail    = seleniumHandler.getTextField(UserDialog.USER_EMAIL_FIELD);
247         String    formColor    = seleniumHandler.getColorPickerValue(UserDialog.USER_COLOR_PICKER);
248         LocalDate formFirstDay = seleniumHandler.getDatePickerValue(UserDialog.USER_FIRST_WORKING_DAY_PICKER);
249         LocalDate formLastDay  = seleniumHandler.getDatePickerValue(UserDialog.USER_LAST_WORKING_DAY_PICKER);
250         // Close the dialog
251         seleniumHandler.click(UserDialog.CANCEL_BUTTON);
252         // Assert that all form fields are empty or default values
253         Assertions.assertTrue(formName.isEmpty(), "Name field should be empty in new dialog");
254         Assertions.assertTrue(formEmail.isEmpty(), "Email field should be empty in new dialog");
255         Assertions.assertEquals("#000000", formColor, "Color picker should have default value black");
256         Assertions.assertNull(formFirstDay, "First working day picker should be null in new dialog");
257         Assertions.assertNull(formLastDay, "Last working day picker should be null in new dialog");
258     }

260     /**
261      * Verifies all fields in the UserDialog for a specific user.
262      * <p>
263      * Opens the edit dialog for the specified user and verifies that all fields
264      * (name, email, color, first working day, last working day) have the expected values.
265      * Cancels the dialog after verification to avoid making changes.
266      *
267      * @param name             the name of the user to verify
268      * @param expectedEmail    the expected email of the user
269      * @param expectedColor    the expected hex color code (with # prefix) for the user
270      * @param expectedFirstDay the expected first working day for the user
271      * @param expectedLastDay  the expected last working day for the user
272      */
273     public void verifyUserDialogFields(String name, String expectedEmail, String expectedColor, LocalDate expectedFirstDay, LocalDate expectedLastDay) {
274         // Open the edit dialog for the specified user
275         seleniumHandler.click(UserListView.USER_GRID_EDIT_BUTTON_PREFIX + name);
276         // Read actual values from the dialog
277         String    actualName     = seleniumHandler.getTextField(UserDialog.USER_NAME_FIELD);
278         String    actualEmail    = seleniumHandler.getTextField(UserDialog.USER_EMAIL_FIELD);
279         String    actualColor    = seleniumHandler.getColorPickerValue(UserDialog.USER_COLOR_PICKER);
280         LocalDate actualFirstDay = seleniumHandler.getDatePickerValue(UserDialog.USER_FIRST_WORKING_DAY_PICKER);
281         LocalDate actualLastDay  = seleniumHandler.getDatePickerValue(UserDialog.USER_LAST_WORKING_DAY_PICKER);
282         // Cancel the dialog to avoid making changes
283         seleniumHandler.click(UserDialog.CANCEL_BUTTON);
284         Assertions.assertEquals(name, actualName, "Name mismatch");
285         Assertions.assertEquals(expectedEmail, actualEmail, "Email mismatch");
286         Assertions.assertEquals(expectedColor, actualColor, "Color mismatch");
287         Assertions.assertEquals(expectedFirstDay, actualFirstDay, "First working day mismatch");
288         Assertions.assertEquals(expectedLastDay, actualLastDay, "Last working day mismatch");
289     }
