65      /**
66       * Tests attempting to create a duplicate location record with the same start date.
67       * <p>
68       * Opens the location creation dialog, enters a start date that already exists,
69       * then attempts to confirm. Verifies that an error message is shown and the duplicate
70       * record is not created.
71       *
72       * @param existingStartDate the start date that already has a location record
73       * @param country           the country code for the duplicate location
74       * @param state             the state/region code for the duplicate location
75       */
76      public void createDuplicateDateLocation(LocalDate existingStartDate, String country, String state) {
77          String existingDateStr = existingStartDate.format(dateFormatter);
78  
79          // Try to create a duplicate record
80          seleniumHandler.click(LocationListView.CREATE_LOCATION_BUTTON);
81          seleniumHandler.setDatePickerValue(LocationDialog.LOCATION_START_DATE_FIELD, existingStartDate);
82          seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_COUNTRY_FIELD, country);
83          seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_STATE_FIELD, state);
84          seleniumHandler.click(LocationDialog.CONFIRM_BUTTON);
85  
86          // Verify the validation error occurs (dialog should still be open)
87          seleniumHandler.waitUntil(ExpectedConditions.visibilityOfElementLocated(
88                  By.cssSelector("vaadin-notification-card")));
89  
90          // Cancel the dialog since we expect it to still be open
91          seleniumHandler.click(LocationDialog.CANCEL_BUTTON);
92      }

94      /**
95       * Tests the creation of a location record where the user cancels the operation.
96       * <p>
97       * Opens the location creation dialog, enters values for start date, country, and state,
98       * then cancels the dialog. Verifies that no record with the specified start date appears in the list
99       * and that the cancellation didn't affect any existing data.
100      *
101      * @param startDate the start date for the location record
102      * @param country   the country code for the location (ISO 3166-1 alpha-2)
103      * @param state     the state/region code for the location
104      */
105     public void createLocationCancel(LocalDate startDate, String country, String state) {
106         seleniumHandler.click(LocationListView.CREATE_LOCATION_BUTTON);
107         seleniumHandler.setDatePickerValue(LocationDialog.LOCATION_START_DATE_FIELD, startDate);
108         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_COUNTRY_FIELD, country);
109         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_STATE_FIELD, state);
110         seleniumHandler.click(LocationDialog.CANCEL_BUTTON);
111 
112         // Verify the record doesn't appear in the list
113         String startDateStr = startDate.format(dateFormatter);
114         seleniumHandler.ensureIsNotInList(LocationListView.LOCATION_GRID_START_DATE_PREFIX, startDateStr);
115     }

175     /**
176      * Tests the editing of a location record where the user cancels the edit operation.
177      * <p>
178      * Clicks the edit button for the location record with the specified start date,
179      * enters new values, then cancels the edit dialog. Verifies that the record
180      * still maintains its original values.
181      *
182      * @param originalStartDate the original start date of the record to edit
183      * @param newStartDate      the new start date to attempt to assign
184      * @param originalCountry   the original country code to verify remains unchanged after cancellation
185      * @param newCountry        the country code to attempt to assign
186      * @param originalState     the original state/region code to verify remains unchanged after cancellation
187      * @param newState          the state/region code to attempt to assign
188      */
189     public void editLocationCancel(LocalDate originalStartDate, LocalDate newStartDate,
190                                    String originalCountry, String newCountry,
191                                    String originalState, String newState) {
192         String originalDateStr = originalStartDate.format(dateFormatter);
193 
194         // Verify original values first
195         verifyLocationValues(originalDateStr, originalCountry, originalState);
196 
197         // Edit the record but cancel
198         seleniumHandler.click(LocationListView.LOCATION_GRID_EDIT_BUTTON_PREFIX + originalDateStr);
199         seleniumHandler.setDatePickerValue(LocationDialog.LOCATION_START_DATE_FIELD, newStartDate);
200         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_COUNTRY_FIELD, newCountry);
201         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_STATE_FIELD, newState);
202         seleniumHandler.click(LocationDialog.CANCEL_BUTTON);
203 
204         // Verify original record still exists and new one doesn't
205         seleniumHandler.ensureIsInList(LocationListView.LOCATION_GRID_START_DATE_PREFIX, originalDateStr);
206         if (!originalStartDate.equals(newStartDate)) {
207             String newDateStr = newStartDate.format(dateFormatter);
208             seleniumHandler.ensureIsNotInList(LocationListView.LOCATION_GRID_START_DATE_PREFIX, newDateStr);
209         }
210 
211         // Verify the values didn't change
212         verifyLocationValues(originalDateStr, originalCountry, originalState);
213     }

215     /**
216      * Tests the successful editing of a location record.
217      * <p>
218      * Clicks the edit button for the location record with the specified start date,
219      * enters new values, then confirms the edit. Verifies that the record with the new values
220      * appears in the list and the record with the old values is gone.
221      *
222      * @param originalStartDate the original start date of the record to edit
223      * @param newStartDate      the new start date to assign
224      * @param newCountry        the new country code to assign
225      * @param newState          the new state/region code to assign
226      */
227     public void editLocationConfirm(LocalDate originalStartDate, LocalDate newStartDate, String newCountry, String newState) {
228 //        seleniumHandler.waitUntilBrowserClosed(0);
229         String originalDateStr = originalStartDate.format(dateFormatter);
230 
231         // Edit the record and confirm
232         seleniumHandler.click(LocationListView.LOCATION_GRID_EDIT_BUTTON_PREFIX + originalDateStr);
233         seleniumHandler.setDatePickerValue(LocationDialog.LOCATION_START_DATE_FIELD, newStartDate);
234         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_COUNTRY_FIELD, newCountry);
235         seleniumHandler.setComboBoxValue(LocationDialog.LOCATION_STATE_FIELD, newState);
236         seleniumHandler.click(LocationDialog.CONFIRM_BUTTON);
237 
238         String newDateStr = newStartDate.format(dateFormatter);
239 
240         // Verify the new record exists and the old one is gone (if dates are different)
241         seleniumHandler.ensureIsInList(LocationListView.LOCATION_GRID_START_DATE_PREFIX, newDateStr);
242         if (!originalStartDate.equals(newStartDate)) {
243             seleniumHandler.ensureIsNotInList(LocationListView.LOCATION_GRID_START_DATE_PREFIX, originalDateStr);
244         }
245 
246         // Verify the updated values
247         verifyLocationValues(newDateStr, newCountry, newState);
248     }

250     /**
251      * Navigates to the LocationListView for a specific user.
252      * <p>
253      * Opens the location list URL directly, logs in if needed, and waits for the page to load
254      * by checking for the presence of the page title element.
255      *
256      * @param recordingFolderName The folder name for recording the test
257      * @param testName            The name of the test for recording
258      * @param username            The username for which to view locations (optional, uses current user if null)
259      */
260     public void switchToLocationListView(String recordingFolderName, String testName, String username) {
261         //- Check if we need to log in
262         if (!seleniumHandler.getCurrentUrl().contains("/ui/")) {
263             seleniumHandler.getAndCheck("http://localhost:" + port + "/ui/" + LoginView.ROUTE);
264             seleniumHandler.startRecording(recordingFolderName, testName);
265             seleniumHandler.setLoginUser("admin-user");
266             seleniumHandler.setLoginPassword("test-password");
267             seleniumHandler.loginSubmit();
268             seleniumHandler.waitUntil(ExpectedConditions.elementToBeClickable(By.id(ProductListView.PRODUCT_LIST_PAGE_TITLE)));
269         }
270 
271         // Navigate to the location view for the specific user or current user
272         String url = "http://localhost:" + port + "/ui/" + LocationListView.ROUTE;
273         if (username != null) {
274             url += "/" + username;
275         }
276 
277         seleniumHandler.getAndCheck(url);
278         seleniumHandler.waitUntil(ExpectedConditions.elementToBeClickable(By.id(LocationListView.LOCATION_LIST_PAGE_TITLE)));
279     }

298     /**
299      * Verifies the country and state values for a specific location record.
300      * <p>
301      * Opens the edit dialog for the record with the specified start date and verifies
302      * that the country and state fields have the expected values. Cancels the dialog after verification
303      * to avoid making changes.
304      *
305      * @param startDateStr    the formatted start date string of the record to verify
306      * @param expectedCountry the expected country code
307      * @param expectedState   the expected state/region code
308      */
309     private void verifyLocationValues(String startDateStr, String expectedCountry, String expectedState) {
310         seleniumHandler.click(LocationListView.LOCATION_GRID_EDIT_BUTTON_PREFIX + startDateStr);
311         String actualCountry = seleniumHandler.getComboBoxValue(LocationDialog.LOCATION_COUNTRY_FIELD);
312         String actualState   = seleniumHandler.getComboBoxValue(LocationDialog.LOCATION_STATE_FIELD);
313         seleniumHandler.click(LocationDialog.CANCEL_BUTTON);
314         Assertions.assertEquals(expectedCountry, actualCountry, "Country mismatch for record with start date: " + startDateStr);
315         Assertions.assertEquals(expectedState, actualState, "State/region mismatch for record with start date: " + startDateStr);
316     }
