215     /**
216      * Lists files in the specified directory with an extension filter
217      *
218      * @param directoryPath The path to the directory
219      * @param extension     The file extension to filter by (e.g., ".java")
220      * @return List of matching file paths
221      * @throws IOException If an I/O error occurs
222      */
223     private static List<Path> listFilesByExtension(String directoryPath, String extension) throws IOException {
224         try (Stream<Path> stream = Files.list(Paths.get(directoryPath))) {
225             return stream
226                     .filter(Files::isRegularFile)
227                     .filter(path -> path.toString().endsWith(extension))
228                     .sorted()
229                     .collect(Collectors.toList());
230         }
231     }
